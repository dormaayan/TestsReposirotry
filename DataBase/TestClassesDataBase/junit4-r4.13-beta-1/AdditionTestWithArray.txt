@RunWith(Parameterized.class) public static class AdditionTestWithArray {
  @Parameters(name="{index}: {0} + {1} = {2}") public static Object[][] data(){
    return new Object[][]{{0,0,0},{1,1,2},{3,2,5},{4,3,7}};
  }
  @Parameter(0) public int firstSummand;
  @Parameter(1) public int secondSummand;
  @Parameter(2) public int sum;
  @Test public void test(){
    assertEquals(sum,firstSummand + secondSummand);
  }
}
