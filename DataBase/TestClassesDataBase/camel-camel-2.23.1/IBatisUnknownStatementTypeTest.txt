public class IBatisUnknownStatementTypeTest extends CamelTestSupport {
  @Override public boolean isUseRouteBuilder(){
    return false;
  }
  @Test public void testStatementTypeNotSet() throws Exception {
    context.addRoutes(new RouteBuilder(){
      @Override public void configure() throws Exception {
        from("direct:start").to("ibatis:selectAllAccounts");
      }
    }
);
    try {
      context.start();
      fail("Should have thrown exception");
    }
 catch (    FailedToCreateRouteException e) {
      assertIsInstanceOf(IllegalArgumentException.class,e.getCause().getCause());
      assertEquals("statementType must be specified on: ibatis://selectAllAccounts",e.getCause().getCause().getMessage());
    }
  }
}
