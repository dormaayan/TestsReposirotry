public class PluginTest {
  @Test public void mapPluginShouldInterceptGet(){
    Map map=new HashMap();
    map=(Map)new AlwaysMapPlugin().plugin(map);
    assertEquals("Always",map.get("Anything"));
  }
  @Test public void shouldNotInterceptToString(){
    Map map=new HashMap();
    map=(Map)new AlwaysMapPlugin().plugin(map);
    assertFalse("Always".equals(map.toString()));
  }
@Intercepts({@Signature(type=Map.class,method="get",args={Object.class})}) public static class AlwaysMapPlugin implements Interceptor {
    @Override public Object intercept(    Invocation invocation) throws Throwable {
      return "Always";
    }
    @Override public Object plugin(    Object target){
      return Plugin.wrap(target,this);
    }
    @Override public void setProperties(    Properties properties){
    }
  }
}
