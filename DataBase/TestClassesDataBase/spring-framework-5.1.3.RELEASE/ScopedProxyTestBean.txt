@Component @Scope(proxyMode=ScopedProxyMode.INTERFACES) static class ScopedProxyTestBean extends AbstractIdentifiable implements SimpleService {
  @Autowired private EventCollector eventCollector;
  @Autowired private CountDownLatch countDownLatch;
  @EventListener @Override public void handleIt(  TestEvent event){
    this.eventCollector.addEvent(this,event);
  }
  @EventListener @Async public void handleAsync(  AnotherTestEvent event){
    assertTrue(!Thread.currentThread().getName().equals(event.content));
    this.eventCollector.addEvent(this,event);
    this.countDownLatch.countDown();
  }
}
