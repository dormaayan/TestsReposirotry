@Controller static class MyRelativeMethodPathDispatchingController {
  @RequestMapping("*/myHandle") public void myHandle(  HttpServletResponse response) throws IOException {
    response.getWriter().write("myView");
  }
  @RequestMapping("/*/*Other") public void myOtherHandle(  HttpServletResponse response) throws IOException {
    response.getWriter().write("myOtherView");
  }
  @RequestMapping("*/myLang") public void myLangHandle(  HttpServletResponse response) throws IOException {
    response.getWriter().write("myLangView");
  }
  @RequestMapping("/*/surprise") public void mySurpriseHandle(  HttpServletResponse response) throws IOException {
    response.getWriter().write("mySurpriseView");
  }
}
