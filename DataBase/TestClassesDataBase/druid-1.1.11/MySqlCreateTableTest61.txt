public class MySqlCreateTableTest61 extends MysqlTest {
  @Test public void test_one() throws Exception {
    String sql="create table t (pk int auto_increment primary key,a timestamp null default current_timestamp on update current_timestamp, b timestamp null)";
    MySqlStatementParser parser=new MySqlStatementParser(sql);
    SQLStatement stmt=parser.parseCreateTable();
    MySqlSchemaStatVisitor visitor=new MySqlSchemaStatVisitor();
    stmt.accept(visitor);
{
      String output=SQLUtils.toMySqlString(stmt);
      Assert.assertEquals("CREATE TABLE t (" + "\n\tpk int PRIMARY KEY AUTO_INCREMENT," + "\n\ta timestamp NULL DEFAULT current_timestamp ON UPDATE current_timestamp,"+ "\n\tb timestamp NULL"+ "\n)",output);
    }
{
      String output=SQLUtils.toMySqlString(stmt,SQLUtils.DEFAULT_LCASE_FORMAT_OPTION);
      Assert.assertEquals("create table t (" + "\n\tpk int primary key auto_increment," + "\n\ta timestamp null default current_timestamp on update current_timestamp,"+ "\n\tb timestamp null"+ "\n)",output);
    }
  }
}
