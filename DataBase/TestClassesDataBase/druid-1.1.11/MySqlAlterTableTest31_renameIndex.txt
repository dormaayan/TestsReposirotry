public class MySqlAlterTableTest31_renameIndex extends TestCase {
  public void test_alter_add_key() throws Exception {
    String sql="ALTER TABLE test_table_normal\n" + "RENAME INDEX old_index_name TO new_index_name";
    MySqlStatementParser parser=new MySqlStatementParser(sql);
    SQLStatement stmt=parser.parseStatementList().get(0);
    parser.match(Token.EOF);
    String output=SQLUtils.toMySqlString(stmt);
    Assert.assertEquals("ALTER TABLE test_table_normal\n" + "\tRENAME INDEX old_index_name TO new_index_name",output);
  }
}
