public class OdpsAlterTableChangeColumnTest2 extends TestCase {
  public void test_select() throws Exception {
    String sql="alter table sale_detail change column old_col_name rename to new_col_name;";
    Assert.assertEquals("ALTER TABLE sale_detail\n" + "\tCHANGE COLUMN old_col_name RENAME TO new_col_name;",SQLUtils.formatOdps(sql));
    Assert.assertEquals("alter table sale_detail\n" + "\tchange column old_col_name rename to new_col_name;",SQLUtils.formatOdps(sql,SQLUtils.DEFAULT_LCASE_FORMAT_OPTION));
    List<SQLStatement> statementList=SQLUtils.parseStatements(sql,JdbcConstants.ODPS);
    SQLStatement stmt=statementList.get(0);
    Assert.assertEquals(1,statementList.size());
    SchemaStatVisitor visitor=SQLUtils.createSchemaStatVisitor(JdbcConstants.ODPS);
    stmt.accept(visitor);
    Assert.assertEquals(1,visitor.getTables().size());
    Assert.assertEquals(2,visitor.getColumns().size());
    Assert.assertEquals(0,visitor.getConditions().size());
  }
}
