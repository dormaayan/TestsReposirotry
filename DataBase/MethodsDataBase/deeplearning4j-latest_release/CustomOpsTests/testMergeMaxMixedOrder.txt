@Test public void testMergeMaxMixedOrder(){
  val array0=Nd4j.rand('f',5,2).addi(1);
  val array1=array0.dup('c').addi(5);
  array1.put(0,0,0);
  val exp=array1.dup();
  exp.putScalar(0,0,array0.getDouble(0,0));
  val zF=Nd4j.zeros(array0.shape(),'f');
  CustomOp op=DynamicCustomOp.builder("mergemax").addInputs(array0,array1).addOutputs(zF).callInplace(false).build();
  Nd4j.getExecutioner().exec(op);
  assertEquals(exp,zF);
}
