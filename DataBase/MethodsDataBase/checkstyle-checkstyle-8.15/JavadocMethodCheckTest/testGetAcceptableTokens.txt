@Test public void testGetAcceptableTokens(){
  final JavadocMethodCheck javadocMethodCheck=new JavadocMethodCheck();
  final int[] actual=javadocMethodCheck.getAcceptableTokens();
  final int[] expected={TokenTypes.PACKAGE_DEF,TokenTypes.IMPORT,TokenTypes.CLASS_DEF,TokenTypes.ENUM_DEF,TokenTypes.INTERFACE_DEF,TokenTypes.METHOD_DEF,TokenTypes.CTOR_DEF,TokenTypes.ANNOTATION_FIELD_DEF};
  assertArrayEquals("Default acceptable tokens are invalid",expected,actual);
}
