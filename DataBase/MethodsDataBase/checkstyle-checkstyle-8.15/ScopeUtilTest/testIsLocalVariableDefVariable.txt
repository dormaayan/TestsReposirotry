@Test public void testIsLocalVariableDefVariable(){
  assertTrue("invalid result",ScopeUtil.isLocalVariableDef(getNode(TokenTypes.SLIST,TokenTypes.VARIABLE_DEF)));
  assertTrue("invalid result",ScopeUtil.isLocalVariableDef(getNode(TokenTypes.FOR_INIT,TokenTypes.VARIABLE_DEF)));
  assertTrue("invalid result",ScopeUtil.isLocalVariableDef(getNode(TokenTypes.FOR_EACH_CLAUSE,TokenTypes.VARIABLE_DEF)));
  assertFalse("invalid result",ScopeUtil.isLocalVariableDef(getNode(TokenTypes.CLASS_DEF,TokenTypes.VARIABLE_DEF)));
}
