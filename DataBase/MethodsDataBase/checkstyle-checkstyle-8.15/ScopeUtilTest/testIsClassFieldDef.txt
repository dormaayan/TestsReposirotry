@Test public void testIsClassFieldDef(){
  assertTrue("Should return true when passed is class field def",ScopeUtil.isClassFieldDef(getNode(TokenTypes.CLASS_DEF,TokenTypes.OBJBLOCK,TokenTypes.VARIABLE_DEF)));
  assertFalse("Should return false when passed is unexpected",ScopeUtil.isClassFieldDef(getNode(TokenTypes.CLASS_DEF)));
  assertFalse("Should return false when passed is method variable def",ScopeUtil.isClassFieldDef(getNode(TokenTypes.METHOD_DEF,TokenTypes.SLIST,TokenTypes.VARIABLE_DEF)));
}
