@Test public void testisInInterfaceOrAnnotationBlock(){
  assertTrue("Should return true when node is in interface or annotation block",ScopeUtil.isInInterfaceOrAnnotationBlock(getNode(TokenTypes.ANNOTATION_DEF,TokenTypes.OBJBLOCK)));
  assertTrue("Should return true when node is in interface or annotation block",ScopeUtil.isInInterfaceOrAnnotationBlock(getNode(TokenTypes.INTERFACE_DEF,TokenTypes.OBJBLOCK)));
  assertFalse("Should return false when node is not in interface or annotation block",ScopeUtil.isInInterfaceOrAnnotationBlock(getNode(TokenTypes.CLASS_DEF,TokenTypes.OBJBLOCK)));
  assertFalse("Should return false when node is not in interface or annotation block",ScopeUtil.isInInterfaceOrAnnotationBlock(getNode(TokenTypes.LITERAL_NEW,TokenTypes.IDENT)));
  assertFalse("Should return false when node is not in interface or annotation block",ScopeUtil.isInInterfaceOrAnnotationBlock(getNode(TokenTypes.ENUM_DEF,TokenTypes.OBJBLOCK)));
}
