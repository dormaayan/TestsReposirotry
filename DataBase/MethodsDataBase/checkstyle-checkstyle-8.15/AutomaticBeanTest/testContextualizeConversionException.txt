@Test public void testContextualizeConversionException(){
  final TestBean testBean=new TestBean();
  final DefaultContext context=new DefaultContext();
  context.add("val","some string");
  try {
    testBean.contextualize(context);
    fail("InvocationTargetException is expected");
  }
 catch (  CheckstyleException ex) {
    final String expected="illegal value ";
    assertTrue("Invalid exception cause, should be: ConversionException",ex.getCause() instanceof ConversionException);
    assertTrue("Invalid exception message, should start with: " + expected,ex.getMessage().startsWith(expected));
  }
}
