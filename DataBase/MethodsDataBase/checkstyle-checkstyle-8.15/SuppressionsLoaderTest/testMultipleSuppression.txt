@Test public void testMultipleSuppression() throws Exception {
  final FilterSet fc=SuppressionsLoader.loadSuppressions(getPath("InputSuppressionsLoaderMultiple.xml"));
  final FilterSet fc2=new FilterSet();
  final SuppressElement se0=new SuppressElement("file0","check0",null,null,null,null);
  fc2.addFilter(se0);
  final SuppressElement se1=new SuppressElement("file1","check1",null,null,"1,2-3",null);
  fc2.addFilter(se1);
  final SuppressElement se2=new SuppressElement("file2","check2",null,null,null,"1,2-3");
  fc2.addFilter(se2);
  final SuppressElement se3=new SuppressElement("file3","check3",null,null,"1,2-3","1,2-3");
  fc2.addFilter(se3);
  final SuppressElement se4=new SuppressElement(null,null,"message0",null,null,null);
  fc2.addFilter(se4);
  assertEquals("Multiple suppressions were loaded incorrectly",fc2,fc);
}
