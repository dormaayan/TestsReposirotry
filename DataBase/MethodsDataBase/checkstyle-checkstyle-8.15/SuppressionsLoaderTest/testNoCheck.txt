@Test public void testNoCheck() throws IOException {
  final String fn=getPath("InputSuppressionsLoaderNoCheck.xml");
  try {
    SuppressionsLoader.loadSuppressions(fn);
    fail("Exception is expected");
  }
 catch (  CheckstyleException ex) {
    final String messageStart="Unable to parse " + fn;
    assertTrue("Exception message should start with: " + messageStart,ex.getMessage().startsWith(messageStart));
    assertTrue("Exception message should contain \"checks\"",ex.getMessage().contains("\"checks\""));
    assertTrue("Exception message should contain \"suppress\"",ex.getMessage().contains("\"suppress\""));
  }
}
