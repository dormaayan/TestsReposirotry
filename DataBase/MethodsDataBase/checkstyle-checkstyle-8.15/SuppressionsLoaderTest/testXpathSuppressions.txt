@Test public void testXpathSuppressions() throws Exception {
  final String fn=getPath("InputSuppressionsLoaderXpathCorrect.xml");
  final Set<TreeWalkerFilter> filterSet=SuppressionsLoader.loadXpathSuppressions(fn);
  final Set<TreeWalkerFilter> expectedFilterSet=new HashSet<>();
  final XpathFilter xf0=new XpathFilter("file1","test",null,"id1","/CLASS_DEF");
  expectedFilterSet.add(xf0);
  final XpathFilter xf1=new XpathFilter(null,null,"message1",null,"/CLASS_DEF");
  expectedFilterSet.add(xf1);
  assertEquals("Multiple xpath suppressions were loaded incorrectly",expectedFilterSet,filterSet);
}
