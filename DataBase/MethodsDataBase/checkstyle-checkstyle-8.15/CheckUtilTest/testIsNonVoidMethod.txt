@Test public void testIsNonVoidMethod() throws Exception {
  final DetailAST nonVoidMethod=getNodeFromFile(TokenTypes.METHOD_DEF);
  final DetailAST voidMethod=nonVoidMethod.getNextSibling();
  assertTrue("Invalid result: AST provided is void method",CheckUtil.isNonVoidMethod(nonVoidMethod));
  assertFalse("Invalid result: AST provided is non void method",CheckUtil.isNonVoidMethod(voidMethod));
}
