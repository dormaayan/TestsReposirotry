@Test public void testParseDoubleFloatingPointValues(){
  assertEquals("Invalid parse result",-0.05,CheckUtil.parseDouble("-0.05f",TokenTypes.NUM_FLOAT),0);
  assertEquals("Invalid parse result",10.0,CheckUtil.parseDouble("10.0",TokenTypes.NUM_DOUBLE),0);
  assertEquals("Invalid parse result",1230,CheckUtil.parseDouble("1.23e3",TokenTypes.NUM_DOUBLE),0);
  assertEquals("Invalid parse result",-321,CheckUtil.parseDouble("-3.21E2",TokenTypes.NUM_DOUBLE),0);
  assertEquals("Invalid parse result",-0.0,CheckUtil.parseDouble("-0.0",TokenTypes.NUM_DOUBLE),0);
  assertEquals("Invalid parse result",Double.NaN,CheckUtil.parseDouble("NaN",TokenTypes.NUM_DOUBLE),0);
}
