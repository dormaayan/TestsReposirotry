/** 
 * Test CommonUtil.countCharInString.
 */
@Test public void testLengthExpandedTabs(){
  final String s1="\t";
  assertEquals("Invalid expanded tabs length",8,CommonUtil.lengthExpandedTabs(s1,s1.length(),8));
  final String s2="  \t";
  assertEquals("Invalid expanded tabs length",8,CommonUtil.lengthExpandedTabs(s2,s2.length(),8));
  final String s3="\t\t";
  assertEquals("Invalid expanded tabs length",16,CommonUtil.lengthExpandedTabs(s3,s3.length(),8));
  final String s4=" \t ";
  assertEquals("Invalid expanded tabs length",9,CommonUtil.lengthExpandedTabs(s4,s4.length(),8));
  assertEquals("Invalid expanded tabs length",0,CommonUtil.lengthMinusTrailingWhitespace(""));
  assertEquals("Invalid expanded tabs length",0,CommonUtil.lengthMinusTrailingWhitespace(" \t "));
  assertEquals("Invalid expanded tabs length",3,CommonUtil.lengthMinusTrailingWhitespace(" 23"));
  assertEquals("Invalid expanded tabs length",3,CommonUtil.lengthMinusTrailingWhitespace(" 23 \t "));
}
