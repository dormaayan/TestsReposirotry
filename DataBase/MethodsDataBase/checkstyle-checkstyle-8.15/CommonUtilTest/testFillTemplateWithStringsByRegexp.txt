@Test public void testFillTemplateWithStringsByRegexp(){
  assertEquals("invalid result","template",CommonUtil.fillTemplateWithStringsByRegexp("template","lineToPlaceInTemplate",Pattern.compile("NO MATCH")));
  assertEquals("invalid result","before word after",CommonUtil.fillTemplateWithStringsByRegexp("before $0 after","word",Pattern.compile("\\w+")));
  assertEquals("invalid result","before word 123 after1 word after2 123 after3",CommonUtil.fillTemplateWithStringsByRegexp("before $0 after1 $1 after2 $2 after3","word 123",Pattern.compile("(\\w+) (\\d+)")));
}
