@Test public void testFileExtensions(){
  final String[] fileExtensions={"java"};
  final File pdfFile=new File("file.pdf");
  assertFalse("Invalid file extension",CommonUtil.matchesFileExtension(pdfFile,fileExtensions));
  assertTrue("Invalid file extension",CommonUtil.matchesFileExtension(pdfFile));
  assertTrue("Invalid file extension",CommonUtil.matchesFileExtension(pdfFile,(String[])null));
  final File javaFile=new File("file.java");
  assertTrue("Invalid file extension",CommonUtil.matchesFileExtension(javaFile,fileExtensions));
  final File emptyExtensionFile=new File("file.");
  assertTrue("Invalid file extension",CommonUtil.matchesFileExtension(emptyExtensionFile,""));
  assertFalse("Invalid file extension",CommonUtil.matchesFileExtension(pdfFile,".noMatch"));
  assertTrue("Invalid file extension",CommonUtil.matchesFileExtension(pdfFile,".pdf"));
}
