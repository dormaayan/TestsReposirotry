@Test public void testFailureForMultilineRegexp() throws Exception {
  final DefaultConfiguration checkConfig=createModuleConfig(RegexpHeaderCheck.class);
  checkConfig.addAttribute("header","^(.*\\n.*)");
  try {
    createChecker(checkConfig);
    fail("Checker creation should not succeed when regexp spans multiple lines");
  }
 catch (  CheckstyleException ex) {
    assertEquals("Invalid exception message","cannot initialize module" + " com.puppycrawl.tools.checkstyle.checks.header.RegexpHeaderCheck" + " - Cannot set property 'header' to '^(.*\\n.*)' in module"+ " com.puppycrawl.tools.checkstyle.checks.header.RegexpHeaderCheck",ex.getMessage());
  }
}
