@Test public void testEmptyFilename() throws Exception {
  final DefaultConfiguration checkConfig=createModuleConfig(RegexpHeaderCheck.class);
  checkConfig.addAttribute("headerFile","");
  try {
    createChecker(checkConfig);
    fail("Checker creation should not succeed with invalid headerFile");
  }
 catch (  CheckstyleException ex) {
    assertEquals("Invalid exception message","cannot initialize module" + " com.puppycrawl.tools.checkstyle.checks.header.RegexpHeaderCheck" + " - Cannot set property 'headerFile' to '' in"+ " module com.puppycrawl.tools.checkstyle.checks.header.RegexpHeaderCheck",ex.getMessage());
  }
}
