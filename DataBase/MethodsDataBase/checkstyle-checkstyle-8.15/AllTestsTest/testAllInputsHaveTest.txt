@Test public void testAllInputsHaveTest() throws Exception {
  final Map<String,List<String>> allTests=new HashMap<>();
  Files.walk(Paths.get("src/test/java")).forEach(filePath -> {
    grabAllTests(allTests,filePath.toFile());
  }
);
  Assert.assertTrue("found tests",!allTests.keySet().isEmpty());
  Files.walk(Paths.get("src/test/resources/com/puppycrawl")).forEach(filePath -> {
    verifyInputFile(allTests,filePath.toFile());
  }
);
  Files.walk(Paths.get("src/test/resources-noncompilable/com/puppycrawl")).forEach(filePath -> {
    verifyInputFile(allTests,filePath.toFile());
  }
);
}
