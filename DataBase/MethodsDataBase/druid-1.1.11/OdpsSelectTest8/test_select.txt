public void test_select() throws Exception {
  String sql="select * from (select * from abc limit 1) a;";
  Assert.assertEquals("SELECT *" + "\nFROM (" + "\n\tSELECT *"+ "\n\tFROM abc"+ "\n\tLIMIT 1"+ "\n) a;",SQLUtils.formatOdps(sql));
  Assert.assertEquals("select *" + "\nfrom (" + "\n\tselect *"+ "\n\tfrom abc"+ "\n\tlimit 1"+ "\n) a;",SQLUtils.formatOdps(sql,SQLUtils.DEFAULT_LCASE_FORMAT_OPTION));
  List<SQLStatement> statementList=SQLUtils.parseStatements(sql,JdbcConstants.ODPS);
  SQLStatement stmt=statementList.get(0);
  Assert.assertEquals(1,statementList.size());
  SchemaStatVisitor visitor=SQLUtils.createSchemaStatVisitor(JdbcConstants.ODPS);
  stmt.accept(visitor);
  Assert.assertEquals(1,visitor.getTables().size());
  Assert.assertEquals(1,visitor.getColumns().size());
  Assert.assertEquals(0,visitor.getConditions().size());
  Assert.assertTrue(visitor.getColumns().contains(new Column("abc","*")));
}
