@Test public void kTableNamedMaterializedMapValuesShouldPreserveTopologyStructure(){
  final StreamsBuilder builder=new StreamsBuilder();
  final KTable<Object,Object> table=builder.table("input-topic");
  table.mapValues((readOnlyKey,value) -> null,Materialized.<Object,Object,KeyValueStore<Bytes,byte[]>>as("store-name").withKeySerde(null).withValueSerde(null));
  final TopologyDescription describe=builder.build().describe();
  Assert.assertEquals("Topologies:\n" + "   Sub-topology: 0\n" + "    Source: KSTREAM-SOURCE-0000000001 (topics: [input-topic])\n"+ "      --> KTABLE-SOURCE-0000000002\n"+ "    Processor: KTABLE-SOURCE-0000000002 (stores: [input-topic-STATE-STORE-0000000000])\n"+ "      --> KTABLE-MAPVALUES-0000000003\n"+ "      <-- KSTREAM-SOURCE-0000000001\n"+ "    Processor: KTABLE-MAPVALUES-0000000003 (stores: [store-name])\n"+ "      --> none\n"+ "      <-- KTABLE-SOURCE-0000000002\n"+ "\n",describe.toString());
}
