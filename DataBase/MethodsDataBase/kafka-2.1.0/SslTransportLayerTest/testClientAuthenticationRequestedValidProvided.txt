/** 
 * Tests that server accepts connections from a client configured with a valid certificate if client authentication is requested
 */
@Test public void testClientAuthenticationRequestedValidProvided() throws Exception {
  String node="0";
  sslServerConfigs.put(BrokerSecurityConfigs.SSL_CLIENT_AUTH_CONFIG,"requested");
  server=createEchoServer(SecurityProtocol.SSL);
  createSelector(sslClientConfigs);
  InetSocketAddress addr=new InetSocketAddress("localhost",server.port());
  selector.connect(node,addr,BUFFER_SIZE,BUFFER_SIZE);
  NetworkTestUtils.checkClientConnection(selector,node,100,10);
}
