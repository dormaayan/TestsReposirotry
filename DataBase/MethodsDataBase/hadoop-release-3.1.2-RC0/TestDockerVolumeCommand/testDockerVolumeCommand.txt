@Test public void testDockerVolumeCommand(){
  DockerVolumeCommand dockerVolumeCommand=new DockerVolumeCommand("create");
  assertEquals("volume",dockerVolumeCommand.getCommandOption());
  Assert.assertTrue(dockerVolumeCommand.getDockerCommandWithArguments().get("sub-command").contains("create"));
  dockerVolumeCommand.setDriverName("driver1");
  dockerVolumeCommand.setVolumeName("volume1");
  Assert.assertTrue(dockerVolumeCommand.getDockerCommandWithArguments().get("driver").contains("driver1"));
  Assert.assertTrue(dockerVolumeCommand.getDockerCommandWithArguments().get("volume").contains("volume1"));
}
