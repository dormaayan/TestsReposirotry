@Test void intObjectKeyVisitorShouldNotSeeEntriesAfterRequestingBreakOut(){
  PrimitiveIntObjectMap<Integer> map=Primitive.intObjectMap();
  map.put(1,100);
  map.put(2,200);
  map.put(3,300);
  map.put(4,400);
  final AtomicInteger counter=new AtomicInteger();
  map.visitKeys(value -> counter.incrementAndGet() > 2);
  assertThat(counter.get(),is(3));
}
