/** 
 * If there are any public methods annotated with @Rule returning a  {@link MethodRule}then it should also be run. <p>This case has been added with  <a href="https://github.com/junit-team/junit4/issues/589">Issue #589</a> -  Support @Rule for methods works only for TestRule but not for MethodRule
 */
@Test public void runsMethodRuleThatIsReturnedByMethod(){
  ruleWasEvaluated=false;
  assertThat(testResult(HasMethodReturningMethodRule.class),isSuccessful());
  assertTrue(ruleWasEvaluated);
}
