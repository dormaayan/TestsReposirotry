@SuppressWarnings("unchecked") @Test public void noSubsequentSubscriptionIterable(){
  final int[] calls={0};
  Flowable<Integer> source=Flowable.create(new FlowableOnSubscribe<Integer>(){
    @Override public void subscribe(    FlowableEmitter<Integer> s) throws Exception {
      calls[0]++;
      s.onNext(1);
      s.onComplete();
    }
  }
,BackpressureStrategy.MISSING);
  Flowable.concat(Arrays.asList(source,source)).firstElement().test().assertResult(1);
  assertEquals(1,calls[0]);
}
