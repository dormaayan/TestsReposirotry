@Test public void testPredicateThrowsExceptionAndValueInCauseMessage(){
  TestObserver<Boolean> to=new TestObserver<Boolean>();
  final IllegalArgumentException ex=new IllegalArgumentException();
  Flowable.just("Boo!").any(new Predicate<String>(){
    @Override public boolean test(    String v){
      throw ex;
    }
  }
).subscribe(to);
  to.assertTerminated();
  to.assertNoValues();
  to.assertNotComplete();
  to.assertError(ex);
}
