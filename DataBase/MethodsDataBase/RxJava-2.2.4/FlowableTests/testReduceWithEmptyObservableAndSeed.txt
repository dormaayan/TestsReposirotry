/** 
 * A reduce on an empty Observable and a seed should just pass the seed through. This is confirmed at https://github.com/ReactiveX/RxJava/issues/423#issuecomment-27642456
 */
@Test public void testReduceWithEmptyObservableAndSeed(){
  Flowable<Integer> flowable=Flowable.range(1,0);
  int value=flowable.reduce(1,new BiFunction<Integer,Integer,Integer>(){
    @Override public Integer apply(    Integer t1,    Integer t2){
      return t1 + t2;
    }
  }
).blockingGet();
  assertEquals(1,value);
}
