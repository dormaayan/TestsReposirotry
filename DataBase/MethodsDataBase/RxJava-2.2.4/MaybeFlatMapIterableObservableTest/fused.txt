@Test public void fused(){
  TestObserver<Integer> to=ObserverFusion.newTest(QueueFuseable.ANY);
  Maybe.just(1).flattenAsObservable(new Function<Integer,Iterable<Integer>>(){
    @Override public Iterable<Integer> apply(    Integer v) throws Exception {
      return Arrays.asList(v,v + 1);
    }
  }
).subscribe(to);
  to.assertOf(ObserverFusion.<Integer>assertFuseable()).assertOf(ObserverFusion.<Integer>assertFusionMode(QueueFuseable.ASYNC)).assertResult(1,2);
  ;
}
