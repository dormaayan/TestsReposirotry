@Test public void disposerThrows(){
  List<Throwable> errors=TestHelper.trackPluginErrors();
  try {
    Observable.generate(new Callable<Object>(){
      @Override public Object call() throws Exception {
        return 1;
      }
    }
,new BiConsumer<Object,Emitter<Object>>(){
      @Override public void accept(      Object s,      Emitter<Object> e) throws Exception {
        e.onComplete();
      }
    }
,new Consumer<Object>(){
      @Override public void accept(      Object d) throws Exception {
        throw new TestException();
      }
    }
).test().assertResult();
    TestHelper.assertUndeliverable(errors,0,TestException.class);
  }
  finally {
    RxJavaPlugins.reset();
  }
}
