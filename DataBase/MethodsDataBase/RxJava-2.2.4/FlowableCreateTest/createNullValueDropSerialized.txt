    final Throwable[] error={null};
    Flowable.create(new FlowableOnSubscribe<Integer>(){
      @Override public void subscribe(      FlowableEmitter<Integer> e) throws Exception {
        e=e.serialize();
        try {
          e.onNext(null);
          e.onNext(1);
          e.onError(new TestException());
          e.onComplete();
        }
 catch (        Throwable ex) {
          error[0]=ex;
        }
      }
    }
,BackpressureStrategy.DROP).test().assertFailure(NullPointerException.class);
    assertNull(error[0]);

  }
  finally {
    RxJavaPlugins.reset();
  }
