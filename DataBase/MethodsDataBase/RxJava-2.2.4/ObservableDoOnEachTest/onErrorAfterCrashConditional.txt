@Test public void onErrorAfterCrashConditional(){
  List<Throwable> errors=TestHelper.trackPluginErrors();
  try {
    Observable.wrap(new ObservableSource<Object>(){
      @Override public void subscribe(      Observer<? super Object> observer){
        observer.onSubscribe(Disposables.empty());
        observer.onError(new TestException());
      }
    }
).doAfterTerminate(new Action(){
      @Override public void run() throws Exception {
        throw new IOException();
      }
    }
).filter(Functions.alwaysTrue()).test().assertFailure(TestException.class);
    TestHelper.assertUndeliverable(errors,0,IOException.class);
  }
  finally {
    RxJavaPlugins.reset();
  }
}
