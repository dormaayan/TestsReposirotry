@Test public void fused(){
  TestObserver<Integer> to=ObserverFusion.newTest(QueueFuseable.ANY);
  Observable.range(1,10).flatMapCompletable(new Function<Integer,CompletableSource>(){
    @Override public CompletableSource apply(    Integer v) throws Exception {
      return Completable.complete();
    }
  }
).<Integer>toObservable().subscribe(to);
  to.assertOf(ObserverFusion.<Integer>assertFuseable()).assertOf(ObserverFusion.<Integer>assertFusionMode(QueueFuseable.ASYNC)).assertResult();
}
