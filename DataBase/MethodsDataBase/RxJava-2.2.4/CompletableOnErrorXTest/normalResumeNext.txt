@Test public void normalResumeNext(){
  final int[] call={0};
  Completable.complete().onErrorResumeNext(new Function<Throwable,CompletableSource>(){
    @Override public CompletableSource apply(    Throwable e) throws Exception {
      call[0]++;
      return Completable.complete();
    }
  }
).test().assertResult();
  assertEquals(0,call[0]);
}
