@Test public void predicateThrows(){
  List<Throwable> errors=TestHelper.trackPluginErrors();
  try {
    new Flowable<Integer>(){
      @Override protected void subscribeActual(      Subscriber<? super Integer> subscriber){
        subscriber.onSubscribe(new BooleanSubscription());
        subscriber.onNext(1);
        subscriber.onNext(2);
        subscriber.onError(new TestException());
        subscriber.onComplete();
      }
    }
.all(new Predicate<Integer>(){
      @Override public boolean test(      Integer v) throws Exception {
        throw new TestException();
      }
    }
).toFlowable().test().assertFailure(TestException.class);
    TestHelper.assertUndeliverable(errors,0,TestException.class);
  }
  finally {
    RxJavaPlugins.reset();
  }
}
