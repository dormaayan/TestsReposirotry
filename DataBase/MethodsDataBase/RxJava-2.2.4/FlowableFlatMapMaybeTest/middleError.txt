@Test public void middleError(){
  Flowable.fromArray(new String[]{"1","a","2"}).flatMapMaybe(new Function<String,MaybeSource<Integer>>(){
    @Override public MaybeSource<Integer> apply(    final String s) throws NumberFormatException {
      return Maybe.fromCallable(new Callable<Integer>(){
        @Override public Integer call() throws NumberFormatException {
          return Integer.valueOf(s);
        }
      }
);
    }
  }
).test().assertFailure(NumberFormatException.class,1);
}
