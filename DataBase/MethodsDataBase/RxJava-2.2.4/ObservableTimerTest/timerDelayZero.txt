@Test public void timerDelayZero(){
  List<Throwable> errors=TestHelper.trackPluginErrors();
  try {
    for (int i=0; i < 1000; i++) {
      Observable.timer(0,TimeUnit.MILLISECONDS).blockingFirst();
    }
    assertTrue(errors.toString(),errors.isEmpty());
  }
  finally {
    RxJavaPlugins.reset();
  }
}
