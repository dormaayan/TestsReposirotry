static class Base {
  String base;
  String child;
  public Base(  final String base,  final String child){
    this.base=base;
    this.child=child;
  }
  public String getBaseProperty(){
    return base;
  }
  public String getChildProperty(){
    return child;
  }
  public String base(){
    return base;
  }
  public String child(){
    return child;
  }
}
