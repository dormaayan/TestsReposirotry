/** 
 * Unit test for  {@link Text}.
 * @author edgar.espina
 * @since 0.1.0
 */
public class TextTest {
  @Test public void newText(){
    assertEquals("a",new Text(new Handlebars(),"a").text());
  }
  @Test public void newTextSequence(){
    assertEquals("abc",new Text(new Handlebars(),"abc").text());
  }
  @Test(expected=NullPointerException.class) public void newTextFail(){
    new Text(new Handlebars(),null);
  }
}
