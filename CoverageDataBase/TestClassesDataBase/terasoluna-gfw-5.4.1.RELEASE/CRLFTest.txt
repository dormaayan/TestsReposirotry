public class CRLFTest {
  public Class<? extends CodePoints> clazz=CRLF.class;
  @Test public void testIsValid(){
    CodePoints codePoints=CodePoints.of(clazz);
    assertThat(codePoints.containsAll("\n"),is(true));
  }
  @Test public void testIsInValid(){
    CodePoints codePoints=CodePoints.of(clazz);
    assertThat(codePoints.containsAll("„ÅÇ\n"),is(false));
  }
}
