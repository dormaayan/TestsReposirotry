public class JIS_X_0208_LatinLettersTest {
  public Class<? extends CodePoints> clazz=JIS_X_0208_LatinLetters.class;
  @Test public void testIsValid(){
    CodePoints codePoints=CodePoints.of(clazz);
    assertThat(codePoints.containsAll("０１２３４５６７８９"),is(true));
  }
  @Test public void testIsInValid(){
    CodePoints codePoints=CodePoints.of(clazz);
    assertThat(codePoints.containsAll("０１２３4５６７８９"),is(false));
  }
}
