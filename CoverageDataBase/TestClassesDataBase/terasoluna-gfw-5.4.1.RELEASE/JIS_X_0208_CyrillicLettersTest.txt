public class JIS_X_0208_CyrillicLettersTest {
  public Class<? extends CodePoints> clazz=JIS_X_0208_CyrillicLetters.class;
  @Test public void testIsValid(){
    CodePoints codePoints=CodePoints.of(clazz);
    assertThat(codePoints.containsAll("АВС"),is(true));
  }
  @Test public void testIsInValid(){
    CodePoints codePoints=CodePoints.of(clazz);
    assertThat(codePoints.containsAll("ABC"),is(false));
  }
}
