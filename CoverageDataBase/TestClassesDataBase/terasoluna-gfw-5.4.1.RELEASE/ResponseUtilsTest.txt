public class ResponseUtilsTest {
  MockHttpServletResponse response;
  @Before public void setUp() throws Exception {
    response=new MockHttpServletResponse();
  }
  @Test public void testSetPreventionCachingHeaders(){
    ResponseUtils.setPreventionCachingHeaders(response);
    assertThat(response.getHeader("Cache-Control"),is("private,no-store,no-cache,must-revalidate"));
    assertThat(response.getHeader("Pragma"),is("no-cache"));
    assertThat(response.getHeader("Expires"),is("Thu, 01 Jan 1970 00:00:00 GMT"));
  }
  @Test public void testSetPreventionCachingHeadersWithNullResponse(){
    try {
      ResponseUtils.setPreventionCachingHeaders(null);
    }
 catch (    Exception ex) {
      ex.printStackTrace();
      fail();
    }
  }
  @Test public void testPrivateConstructor() throws Exception {
    Constructor<ResponseUtils> c=ResponseUtils.class.getDeclaredConstructor();
    c.setAccessible(true);
    assertNotNull(c.newInstance());
  }
}
