public class JIS_X_0208_GreekLettersTest {
  public Class<? extends CodePoints> clazz=JIS_X_0208_GreekLetters.class;
  @Test public void testIsValid(){
    CodePoints codePoints=CodePoints.of(clazz);
    assertThat(codePoints.containsAll("αβγ"),is(true));
  }
  @Test public void testIsInValid(){
    CodePoints codePoints=CodePoints.of(clazz);
    assertThat(codePoints.containsAll("αBγ"),is(false));
  }
}
