public class JIS_X_0208_KanjiTest {
  public Class<? extends CodePoints> clazz=JIS_X_0208_Kanji.class;
  @Test public void testIsValid(){
    CodePoints codePoints=CodePoints.of(clazz);
    assertThat(codePoints.containsAll("小網代"),is(true));
  }
  @Test public void testIsInValid(){
    CodePoints codePoints=CodePoints.of(clazz);
    assertThat(codePoints.containsAll("\uD867\uDE15網代"),is(false));
  }
}
