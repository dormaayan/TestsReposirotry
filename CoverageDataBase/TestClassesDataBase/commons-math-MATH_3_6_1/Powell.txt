private static class Powell implements MultivariateFunction {
  private int count;
  public Powell(){
    count=0;
  }
  public double value(  double[] x){
    ++count;
    double a=x[0] + 10 * x[1];
    double b=x[2] - x[3];
    double c=x[1] - 2 * x[2];
    double d=x[0] - x[3];
    return a * a + 5 * b * b + c * c * c* c + 10 * d * d* d* d;
  }
  public int getCount(){
    return count;
  }
}
