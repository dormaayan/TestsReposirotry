private static class Cigar implements MultivariateFunction {
  private double factor;
  Cigar(){
    this(1e3);
  }
  Cigar(  double axisratio){
    factor=axisratio * axisratio;
  }
  public double value(  double[] x){
    double f=x[0] * x[0];
    for (int i=1; i < x.length; ++i)     f+=factor * x[i] * x[i];
    return f;
  }
}
