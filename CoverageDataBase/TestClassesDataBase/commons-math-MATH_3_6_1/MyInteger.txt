/** 
 * A mutable integer.
 */
private static class MyInteger {
  private int i;
  public MyInteger(  int i){
    this.i=i;
  }
  public void set(  int i){
    this.i=i;
  }
  @Override public boolean equals(  Object o){
    if (!(o instanceof MyInteger)) {
      return false;
    }
 else {
      return i == ((MyInteger)o).i;
    }
  }
  @Override public int hashCode(){
    return i;
  }
}
