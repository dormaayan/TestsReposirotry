private static class GetVisitor extends DefaultFieldMatrixPreservingVisitor<Fraction> {
  private int count;
  public GetVisitor(){
    super(Fraction.ZERO);
    count=0;
  }
  @Override public void visit(  int i,  int j,  Fraction value){
    ++count;
    Assert.assertEquals(new Fraction(i * 1024 + j,1024),value);
  }
  public int getCount(){
    return count;
  }
}
