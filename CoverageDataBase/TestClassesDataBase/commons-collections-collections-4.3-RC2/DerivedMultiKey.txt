static class DerivedMultiKey<T> extends MultiKey<T> {
  private static final long serialVersionUID=1928896152249821416L;
  public DerivedMultiKey(  final T key1,  final T key2){
    super(key1,key2);
  }
  public T getFirst(){
    return getKey(0);
  }
  public T getSecond(){
    return getKey(1);
  }
}
