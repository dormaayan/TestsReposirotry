/** 
 * Test class for populateMap(MultiMap).
 */
public static class X implements Comparable<X> {
  int key;
  String name;
  public X(  final int key,  final String name){
    this.key=key;
    this.name=name;
  }
  @Override public int compareTo(  final X o){
    return key - o.key | name.compareTo(o.name);
  }
}
