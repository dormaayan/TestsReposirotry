static class CustomBuilder extends HttpClientBuilder {
  public boolean customized;
  public CustomBuilder(  MetricRegistry metricRegistry){
    super(metricRegistry);
    customized=false;
  }
  @Override protected org.apache.http.impl.client.HttpClientBuilder customizeBuilder(  org.apache.http.impl.client.HttpClientBuilder builder){
    customized=true;
    return builder;
  }
}
