public interface FlightDao {
  @SqlUpdate("INSERT INTO flights(flight_id, departure_airport, arrival_airport,departure_time, arrival_time) " + "VALUES (:flight_id, :departure_airport, :arrival_airport, :departure_time, :arrival_time)") int insert(  @Bind("flight_id") String flightId,  @Bind("departure_airport") String departureAirport,  @Bind("arrival_airport") String arrivalAirport,  @Bind("departure_time") DateTime departureTime,  @Bind("arrival_time") DateTime arrivalTime);
  @SqlQuery("SELECT arrival_time FROM flights WHERE flight_id=:flight_id") DateTime getArrivalTime(  @Bind("flight_id") String flightId);
  @SqlQuery("SELECT departure_time FROM flights WHERE flight_id=:flight_id") DateTime getDepartureTime(  @Bind("flight_id") String flightId);
}
