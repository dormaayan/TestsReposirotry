interface TaskDao {
  @SqlUpdate("INSERT INTO tasks(id, assignee, start_date, end_date, comments) " + "VALUES (:id, :assignee, :start_date, :end_date, :comments)") void insert(  @Bind("id") int id,  @Bind("assignee") Optional<String> assignee,  @Bind("start_date") Instant startDate,  @Bind("end_date") Optional<Instant> endDate,  @Bind("comments") Optional<String> comments);
  @SqlQuery("SELECT end_date FROM tasks WHERE id = :id") @SingleValueResult Optional<Instant> findEndDateById(  @Bind("id") int id);
}
