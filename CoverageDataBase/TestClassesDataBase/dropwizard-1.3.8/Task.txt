static class Task {
  private int id;
  @Nullable private Duration duration;
  @JsonProperty int getId(){
    return id;
  }
  @JsonProperty void setId(  int id){
    this.id=id;
  }
  @JsonProperty @Nullable Duration getDuration(){
    return duration;
  }
  @JsonProperty void setDuration(  Duration duration){
    this.duration=duration;
  }
}
