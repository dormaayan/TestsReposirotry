static final class Closing {
  public final int code;
  public final String reason;
  Closing(  int code,  String reason){
    this.code=code;
    this.reason=reason;
  }
  @Override public String toString(){
    return "Closing[" + code + " "+ reason+ "]";
  }
  @Override public int hashCode(){
    return code * 37 + reason.hashCode();
  }
  @Override public boolean equals(  Object other){
    return other instanceof Closing && ((Closing)other).code == code && ((Closing)other).reason.equals(reason);
  }
}
