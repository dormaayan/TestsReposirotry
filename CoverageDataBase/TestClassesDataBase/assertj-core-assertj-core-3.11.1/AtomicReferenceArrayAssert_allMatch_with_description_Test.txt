public class AtomicReferenceArrayAssert_allMatch_with_description_Test extends AtomicReferenceArrayAssertBaseTest {
  private Predicate<Object> predicate;
  @BeforeEach public void beforeOnce(){
    predicate=o -> o != null;
  }
  @Override protected AtomicReferenceArrayAssert<Object> invoke_api_method(){
    return assertions.allMatch(predicate,"custom");
  }
  @Override protected void verify_internal_effects(){
    verify(iterables).assertAllMatch(info(),newArrayList(internalArray()),predicate,new PredicateDescription("custom"));
  }
}
