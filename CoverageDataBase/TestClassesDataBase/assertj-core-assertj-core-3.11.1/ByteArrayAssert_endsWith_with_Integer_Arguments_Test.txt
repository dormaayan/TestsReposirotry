/** 
 * Tests for <code> {@link ByteArrayAssert#endsWith(int)}</code>.
 */
public class ByteArrayAssert_endsWith_with_Integer_Arguments_Test extends ByteArrayAssertBaseTest {
  @Override protected ByteArrayAssert invoke_api_method(){
    return assertions.endsWith(6,8);
  }
  @Override protected void verify_internal_effects(){
    verify(arrays).assertEndsWith(getInfo(assertions),getActual(assertions),IntArrays.arrayOf(6,8));
  }
  @Test public void invoke_api_like_user(){
    assertThat(new byte[]{1,2,3}).endsWith(2,3);
  }
}
