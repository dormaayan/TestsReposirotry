public class AtomicReferenceArrayAssert_usingElementComparator_Test extends AtomicReferenceArrayAssertBaseTest {
  @Mock private Comparator<Object> elementComparator;
  @BeforeEach public void before(){
    initMocks(this);
  }
  @Override protected AtomicReferenceArrayAssert<Object> invoke_api_method(){
    return assertions.usingElementComparator(elementComparator);
  }
  @Override protected void verify_internal_effects(){
    assertThat(elementComparator).isSameAs(getArrays(assertions).getComparator());
  }
}
