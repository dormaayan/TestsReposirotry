public class ShouldHaveReference_create_Test {
  private static final TestDescription TEST_DESCRIPTION=new TestDescription("TEST");
  @Test public void should_create_error_message_for_AtomicMarkableReference(){
    AtomicMarkableReference<String> actual=new AtomicMarkableReference<>("foo",true);
    String message=shouldHaveReference(actual,actual.getReference(),"bar").create(TEST_DESCRIPTION,CONFIGURATION_PROVIDER.representation());
    assertThat(message).isEqualTo(format("[TEST] %n" + "Expecting%n" + "  <AtomicMarkableReference[marked=true, reference=\"foo\"]>%n"+ "to have reference:%n"+ "  <\"bar\">%n"+ "but had:%n"+ "  <\"foo\">"));
  }
  @Test public void should_create_error_message_for_AtomicStampedReference(){
    AtomicStampedReference<String> actual=new AtomicStampedReference<>("foo",123);
    String message=shouldHaveReference(actual,actual.getReference(),"bar").create(TEST_DESCRIPTION,CONFIGURATION_PROVIDER.representation());
    assertThat(message).isEqualTo(format("[TEST] %n" + "Expecting%n" + "  <AtomicStampedReference[stamp=123, reference=\"foo\"]>%n"+ "to have reference:%n"+ "  <\"bar\">%n"+ "but had:%n"+ "  <\"foo\">"));
  }
}
