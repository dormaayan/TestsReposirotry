/** 
 * Tests for <code> {@link ObjectArrayAssert#containsSequence(List)}</code>.
 * @author Chris Arnott
 */
public class ObjectArrayAssert_containsSequence_List_Test extends ObjectArrayAssertBaseTest {
  @Override protected ObjectArrayAssert<Object> invoke_api_method(){
    return assertions.containsSequence(newArrayList((Object)"Luke","Yoda"));
  }
  @Override protected void verify_internal_effects(){
    verify(arrays).assertContainsSequence(getInfo(assertions),getActual(assertions),array("Luke","Yoda"));
  }
  @Test public void should_throw_error_if_sequence_is_null(){
    assertThatNullPointerException().isThrownBy(() -> {
      List<Object> nullList=null;
      assertions.containsSequence(nullList);
    }
).withMessage(nullSequence());
  }
}
