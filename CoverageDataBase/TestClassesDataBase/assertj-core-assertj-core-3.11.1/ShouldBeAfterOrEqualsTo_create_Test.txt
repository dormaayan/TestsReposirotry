/** 
 * Tests for <code> {@link ShouldBeAfterOrEqualsTo#create(Description,org.assertj.core.presentation.Representation)}</code>.
 * @author Joel Costigliola
 */
public class ShouldBeAfterOrEqualsTo_create_Test {
  private ErrorMessageFactory factory;
  @BeforeEach public void setUp(){
    factory=shouldBeAfterOrEqualsTo(parse("2011-01-01"),parse("2012-01-01"));
  }
  @Test public void should_create_error_message(){
    String message=factory.create(new TextDescription("Test"),new StandardRepresentation());
    assertThat(message).isEqualTo(format("[Test] %n" + "Expecting:%n" + "  <2011-01-01T00:00:00.000>%n"+ "to be after or equals to:%n"+ "  <2012-01-01T00:00:00.000>"));
  }
}
