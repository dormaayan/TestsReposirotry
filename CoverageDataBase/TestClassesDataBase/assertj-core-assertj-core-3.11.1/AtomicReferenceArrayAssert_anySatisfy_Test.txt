    restrictions=o -> assertThat(o).isNotNull();

    return assertions.anySatisfy(restrictions);
  }
  @Override protected void verify_internal_effects(){
    verify(iterables).assertAnySatisfy(info(),newArrayList(internalArray()),restrictions);
  }
}
