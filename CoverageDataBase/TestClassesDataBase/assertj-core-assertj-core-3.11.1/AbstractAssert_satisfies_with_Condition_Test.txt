public class AbstractAssert_satisfies_with_Condition_Test extends AbstractAssertBaseTest {
  private static Condition<Object> condition;
  @BeforeClass public static void setUpOnce(){
    condition=new TestCondition<>();
  }
  @Override protected ConcreteAssert invoke_api_method(){
    return assertions.satisfies(condition);
  }
  @Override protected void verify_internal_effects(){
    verify(conditions).assertSatisfies(getInfo(assertions),getActual(assertions),condition);
  }
}
