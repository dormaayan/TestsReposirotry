public class Offset_built_with_BigInteger_Test {
  @Test public void should_throw_error_if_value_is_null(){
    assertThatNullPointerException().isThrownBy(() -> {
      BigInteger value=null;
      offset(value);
    }
);
  }
  @Test public void should_throw_error_if_value_is_negative(){
    assertThatIllegalArgumentException().isThrownBy(() -> offset(new BigInteger("-1"))).withMessage(offsetValueIsNotPositive());
  }
  @Test public void should_throw_error_if_value_is_zero_strict_offset(){
    assertThatIllegalArgumentException().isThrownBy(() -> strictOffset(BigInteger.ZERO)).withMessage(strictOffsetValueIsNotStrictlyPositive());
  }
  @Test public void should_create_Offset(){
    BigInteger value=BigInteger.TEN;
    Offset<BigInteger> offset=offset(value);
    assertThat(offset.value).isSameAs(value);
  }
}
