public class IterableAssert_allMatch_with_description_Test extends IterableAssertBaseTest {
  private Predicate<Object> predicate;
  @BeforeEach public void beforeOnce(){
    predicate=o -> o != null;
  }
  @Override protected ConcreteIterableAssert<Object> invoke_api_method(){
    return assertions.allMatch(predicate,"custom");
  }
  @Override protected void verify_internal_effects(){
    verify(iterables).assertAllMatch(getInfo(assertions),getActual(assertions),predicate,new PredicateDescription("custom"));
  }
}
