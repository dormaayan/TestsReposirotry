/** 
 * Tests for <code> {@link ObjectAssert#usingDefaultComparator()}</code>.
 * @author Joel Costigliola
 * @author Mikhail Mazursky
 */
public class ObjectAssert_usingDefaultComparator_Test extends ObjectAssertBaseTest {
  @Mock private Comparator<Jedi> comparator;
  @BeforeEach public void before(){
    initMocks(this);
    assertions.usingComparator(comparator);
  }
  @Override protected ObjectAssert<Jedi> invoke_api_method(){
    return assertions.usingDefaultComparator();
  }
  @Override protected void verify_internal_effects(){
    assertThat(Objects.instance()).isSameAs(getObjects(assertions));
  }
}
