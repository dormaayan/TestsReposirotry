/** 
 * Tests for <code> {@link FloatArrayAssert#endsWith(float)}</code>.
 * @author Alex Ruiz
 */
public class FloatArrayAssert_endsWith_Test extends FloatArrayAssertBaseTest {
  @Override protected FloatArrayAssert invoke_api_method(){
    return assertions.endsWith(6f,8f);
  }
  @Override protected void verify_internal_effects(){
    verify(arrays).assertEndsWith(getInfo(assertions),getActual(assertions),arrayOf(6f,8f));
  }
  @Test public void should_pass_with_precision_specified_as_last_argument(){
    float[] actual=arrayOf(1.0f,2.0f,3.0f);
    assertThat(actual).endsWith(arrayOf(2.01f,3.0f),withPrecision(0.1f));
  }
  @Test public void should_pass_with_precision_specified_in_comparator(){
    float[] actual=arrayOf(1.0f,2.0f,3.0f);
    assertThat(actual).usingComparatorWithPrecision(0.1f).endsWith(2.01f,3.05f);
  }
}
