/** 
 * Tests for <code> {@link org.assertj.core.api.BooleanArrayAssert#containsExactlyInAnyOrder(boolean)}</code>.
 */
public class BooleanArrayAssert_containsExactlyInAnyOrder_Test extends BooleanArrayAssertBaseTest {
  @Override protected BooleanArrayAssert invoke_api_method(){
    return assertions.containsExactlyInAnyOrder(true,false);
  }
  @Override protected void verify_internal_effects(){
    verify(arrays).assertContainsExactlyInAnyOrder(getInfo(assertions),getActual(assertions),arrayOf(true,false));
  }
  @Test public void invoke_api_like_user(){
    assertThat(new boolean[]{true,false,false}).containsExactlyInAnyOrder(false,false,true);
  }
}
