/** 
 * Tests for <code> {@link BooleanArrays#assertContains(AssertionInfo,boolean[],boolean[])}</code>.
 * @author Alex Ruiz
 * @author Joel Costigliola
 */
public class BooleanArrays_assertContains_Test extends BooleanArraysBaseTest {
  private Arrays internalArrays;
  @BeforeEach @Override public void setUp(){
    super.setUp();
    internalArrays=mock(Arrays.class);
    setArrays(internalArrays);
  }
  @Test public void should_delegate_to_internal_Arrays(){
    arrays.assertContains(someInfo(),actual,arrayOf(true,false));
    verify(internalArrays).assertContains(someInfo(),failures,actual,arrayOf(true,false));
  }
}
