/** 
 * Tests for <code> {@link ArrayWrapperList#wrap(Object)}</code>.
 * @author Alex Ruiz
 */
public class ArrayWrapperList_wrap_Test {
  @Test public void should_create_ArrayWrapperList_if_array_is_not_null(){
    int[] array={6,8};
    ArrayWrapperList list=ArrayWrapperList.wrap(array);
    assertThat(list).isNotNull();
    assertThat(list.array).isSameAs(array);
  }
  @Test public void should_return_null_if_array_is_null(){
    assertThat(ArrayWrapperList.wrap(null)).isNull();
  }
  @Test public void should_throw_error_if_parameter_is_not_array(){
    assertThatIllegalArgumentException().isThrownBy(() -> ArrayWrapperList.wrap("Yoda")).withMessage("The object to wrap should be an array");
  }
}
