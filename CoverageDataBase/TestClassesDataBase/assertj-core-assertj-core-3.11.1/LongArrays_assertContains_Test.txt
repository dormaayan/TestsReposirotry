/** 
 * Tests for <code> {@link LongArrays#assertContains(AssertionInfo,long[],long[])}</code>.
 * @author Alex Ruiz
 * @author Joel Costigliola
 */
public class LongArrays_assertContains_Test extends LongArraysBaseTest {
  private Arrays internalArrays;
  @BeforeEach @Override public void setUp(){
    super.setUp();
    internalArrays=mock(Arrays.class);
    setArrays(internalArrays);
  }
  @Test public void should_delegate_to_internal_Arrays(){
    arrays.assertContains(someInfo(),actual,arrayOf(6L,8L,10L));
    verify(internalArrays).assertContains(someInfo(),failures,actual,arrayOf(6L,8L,10L));
  }
}
