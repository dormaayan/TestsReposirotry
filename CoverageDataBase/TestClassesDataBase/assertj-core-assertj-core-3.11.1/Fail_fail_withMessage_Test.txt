/** 
 * Tests for <code> {@link Fail#fail(String)}</code>.
 * @author Yvonne Wang
 * @author Alex Ruiz
 */
public class Fail_fail_withMessage_Test {
  @Test public void shouldIncludeMessageWhenFailing(){
    String message="Failed :(";
    assertThatExceptionOfType(AssertionError.class).isThrownBy(() -> Fail.fail(message)).withMessage(message);
  }
}
