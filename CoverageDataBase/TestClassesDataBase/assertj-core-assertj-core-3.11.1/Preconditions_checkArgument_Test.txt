public class Preconditions_checkArgument_Test {
  @Test public void should_throw_illegalargumentexception_if_expression_is_false(){
    assertThatIllegalArgumentException().isThrownBy(() -> Preconditions.checkArgument(false,"Invalid parameter %s","foo")).withMessage("Invalid parameter foo");
  }
  @Test public void should_not_throw_if_expression_is_true(){
    Preconditions.checkArgument(true,"Invalid parameter %s","foo");
  }
}
