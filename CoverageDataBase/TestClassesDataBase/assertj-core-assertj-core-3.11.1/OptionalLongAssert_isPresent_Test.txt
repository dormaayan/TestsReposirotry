public class OptionalLongAssert_isPresent_Test extends BaseTest {
  @Test public void should_pass_when_OptionalLong_is_present(){
    assertThat(OptionalLong.of(10L)).isPresent();
  }
  @Test public void should_fail_when_OptionalLong_is_empty(){
    assertThatExceptionOfType(AssertionError.class).isThrownBy(() -> assertThat(OptionalLong.empty()).isPresent()).withMessage(shouldBePresent(OptionalLong.empty()).create());
  }
  @Test public void should_fail_when_OptionalLong_is_null(){
    assertThatExceptionOfType(AssertionError.class).isThrownBy(() -> assertThat((OptionalLong)null).isPresent()).withMessage(actualIsNull());
  }
}
