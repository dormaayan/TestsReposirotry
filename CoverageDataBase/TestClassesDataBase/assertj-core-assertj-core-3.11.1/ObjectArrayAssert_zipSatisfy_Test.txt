public class ObjectArrayAssert_zipSatisfy_Test extends ObjectArrayAssertBaseTest {
  private BiConsumer<Object,String> requirements;
  private String[] other;
  @BeforeEach public void beforeOnce(){
    requirements=(o1,o2) -> assertThat(o1).hasSameHashCodeAs(o2);
  }
  @Override protected ObjectArrayAssert<Object> invoke_api_method(){
    return assertions.zipSatisfy(other,requirements);
  }
  @Override protected void verify_internal_effects(){
    verify(iterables).assertZipSatisfy(getInfo(assertions),newArrayList(getActual(assertions)),newArrayList(other),requirements);
  }
}
