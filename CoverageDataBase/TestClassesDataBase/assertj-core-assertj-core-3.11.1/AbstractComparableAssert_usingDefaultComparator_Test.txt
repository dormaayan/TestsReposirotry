/** 
 * Tests for <code> {@link AbstractComparableAssert#usingDefaultComparator()}</code>.
 * @author Filip Hrisafov
 */
public class AbstractComparableAssert_usingDefaultComparator_Test extends AbstractComparableAssertBaseTest {
  @Mock private Comparator<Integer> comparator;
  @BeforeEach public void before(){
    initMocks(this);
    assertions.usingComparator(comparator);
  }
  @Override protected ConcreteComparableAssert invoke_api_method(){
    return assertions.usingDefaultComparator();
  }
  @Override protected void verify_internal_effects(){
    assertThat(getObjects(assertions)).isSameAs(Objects.instance());
    assertThat(getComparables(assertions)).isEqualTo(new Comparables());
  }
}
