public class FloatArrayAssert_containsExactly_Test extends FloatArrayAssertBaseTest {
  @Override protected FloatArrayAssert invoke_api_method(){
    return assertions.containsExactly(1.0f,2.0f);
  }
  @Override protected void verify_internal_effects(){
    verify(arrays).assertContainsExactly(getInfo(assertions),getActual(assertions),arrayOf(1.0f,2.0f));
  }
  @Test public void should_pass_with_precision_specified_as_last_argument(){
    float[] actual=arrayOf(1.0f,2.0f);
    assertThat(actual).containsExactly(arrayOf(1.01f,2.0f),withPrecision(0.1f));
  }
  @Test public void should_pass_when_multiple_expected_values_are_the_same_according_to_the_given_precision(){
    float[] actual=arrayOf(-1.71f,-1.51f,-1.51f);
    assertThat(actual).containsExactly(arrayOf(-1.7f,-1.6f,-1.4101f),within(0.1f));
  }
  @Test public void should_pass_with_precision_specified_in_comparator(){
    float[] actual=arrayOf(1.0f,2.0f);
    assertThat(actual).usingComparatorWithPrecision(0.1f).containsExactly(1.01f,2.0f);
  }
}
