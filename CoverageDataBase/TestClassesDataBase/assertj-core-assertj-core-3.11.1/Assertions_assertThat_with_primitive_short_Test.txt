/** 
 * Tests for <code> {@link Assertions#assertThat(short)}</code>.
 * @author Alex Ruiz
 */
public class Assertions_assertThat_with_primitive_short_Test {
  @Test public void should_create_Assert(){
    AbstractShortAssert<?> assertions=Assertions.assertThat((short)8);
    assertThat(assertions).isNotNull();
  }
  @Test public void should_pass_actual(){
    AbstractShortAssert<?> assertions=Assertions.assertThat((short)8);
    assertThat(assertions.actual).isEqualTo(new Short((short)8));
  }
}
