public class MapAssert_containsAnyOf_Test extends MapAssertBaseTest {
  final MapEntry<String,String>[] entries=array(entry("key1","value1"),entry("key2","value2"));
  @Override protected MapAssert<Object,Object> invoke_api_method(){
    return assertions.containsAnyOf(entry("key1","value1"),entry("key2","value2"));
  }
  @Override protected void verify_internal_effects(){
    verify(maps).assertContainsAnyOf(getInfo(assertions),getActual(assertions),entries);
  }
  @Test public void invoke_api_like_user(){
    assertThat(map("key1","value1","key2","value2")).containsAnyOf(entry("key2","value2"));
  }
}
