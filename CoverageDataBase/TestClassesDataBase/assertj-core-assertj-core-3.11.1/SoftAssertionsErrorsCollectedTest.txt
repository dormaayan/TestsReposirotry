/** 
 * This tests that classes extended from  {@link AbstractStandardSoftAssertions} will have access to the list ofcollected errors that the various proxies have collected.
 */
public class SoftAssertionsErrorsCollectedTest {
  private final Object objectForTesting=null;
  private final TestCollector softly=new TestCollector();
  @Test public void return_empty_list_of_errors(){
    softly.assertThat(objectForTesting).isNull();
    assertThat(softly.getErrors()).isEmpty();
  }
  @Test public void returns_nonempty_list_of_errors(){
    softly.assertThat(objectForTesting).isNotNull();
    assertThat(softly.getErrors()).hasAtLeastOneElementOfType(Throwable.class);
  }
private class TestCollector extends AbstractStandardSoftAssertions {
    public List<Throwable> getErrors(){
      return errorsCollected();
    }
  }
}
