public class AtomicIntegerArrayAssert_usingDefaultComparator_Test extends AtomicIntegerArrayAssertBaseTest {
  @Mock private Comparator<AtomicIntegerArray> comparator;
  private IntArrays arraysBefore;
  @BeforeEach public void before(){
    initMocks(this);
    assertions.usingComparator(comparator);
    arraysBefore=getArrays(assertions);
  }
  @Override protected AtomicIntegerArrayAssert invoke_api_method(){
    return assertions.usingDefaultComparator();
  }
  @Override protected void verify_internal_effects(){
    assertThat(Objects.instance()).isSameAs(getObjects(assertions));
    assertThat(arraysBefore).isSameAs(getArrays(assertions));
  }
}
