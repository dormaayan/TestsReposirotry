public class Description_mostRelevantDescription_Test {
  @Test public void should_return_exsiting_description_if_not_empty(){
    String existing="bar";
    Description description=new TextDescription(existing);
    String mostRelevantDescription=mostRelevantDescription(description,"foo");
    assertThat(mostRelevantDescription).isEqualTo(existing);
  }
  @Test public void should_return_the_other_value_if_description_value_is_empty(){
    Description description=new TextDescription("");
    String other="foo";
    String mostRelevantDescription=mostRelevantDescription(description,other);
    assertThat(mostRelevantDescription).isEqualTo(other);
  }
  @Test public void should_return_the_other_value_if_description_value_is_null(){
    Description description=new TextDescription(null);
    String other="foo";
    String mostRelevantDescription=mostRelevantDescription(description,other);
    assertThat(mostRelevantDescription).isEqualTo(other);
  }
  @Test public void should_return_the_other_value_if_null(){
    Description description=null;
    String other="foo";
    String mostRelevantDescription=mostRelevantDescription(description,other);
    assertThat(mostRelevantDescription).isEqualTo(other);
  }
}
