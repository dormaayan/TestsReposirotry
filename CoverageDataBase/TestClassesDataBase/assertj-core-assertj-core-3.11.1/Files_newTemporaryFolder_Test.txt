/** 
 * Tests for <code> {@link Files#newTemporaryFile()}</code>.
 * @author Alex Ruiz
 * @author Yvonne Wang
 */
public class Files_newTemporaryFolder_Test extends Files_TestCase {
  @Test public void should_create_new_temporary_folder(){
    File f=null;
    try {
      f=Files.newTemporaryFolder();
      assertThat(f).isDirectory();
    }
  finally {
      if (f != null)       f.delete();
    }
  }
}
