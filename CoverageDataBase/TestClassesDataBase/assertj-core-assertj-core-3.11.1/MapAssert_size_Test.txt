public class MapAssert_size_Test {
  @Test @SuppressWarnings("unchecked") public void should_be_able_to_use_integer_assertions_on_size_the_map_size(){
    Map<String,String> stringToString=mapOf(entry("a","1"),entry("b","2"));
    assertThat(stringToString).size().isGreaterThan(0).isLessThanOrEqualTo(3).returnToMap().contains(entry("a","1"));
  }
  @Test public void should_have_an_helpful_error_message_when_size_is_used_on_a_null_map(){
    Map<String,String> nullMap=null;
    assertThatNullPointerException().isThrownBy(() -> assertThat(nullMap).size().isGreaterThan(1)).withMessage("Can not perform assertions on the size of a null map.");
  }
}
