public class SoftAssertionsErrorDescriptionTest {
  @Test public void should_display_the_error_cause_and_the_cause_first_stack_trace_elements(){
    SoftAssertions softly=new SoftAssertions();
    softly.fail("failure",throwRuntimeException());
    AssertionError error=catchThrowableOfType(() -> softly.assertAll(),AssertionError.class);
    assertThat(error).hasMessageStartingWith(format("Multiple Failures (1 failure)%n" + "\tfailure%n" + "at SoftAssertionsErrorDescriptionTest.should_display_the_error_cause_and_the_cause_first_stack_trace_elements(SoftAssertionsErrorDescriptionTest.java:28)"));
  }
  protected static RuntimeException throwRuntimeException(){
    return new RuntimeException("abc");
  }
}
