/** 
 * Tests for <code> {@link ShortAssert#usingDefaultComparator()}</code>.
 * @author Joel Costigliola
 */
public class ShortAssert_usingDefaultComparator_Test extends ShortAssertBaseTest {
  @Mock private Comparator<Short> comparator;
  @BeforeEach public void before(){
    initMocks(this);
    assertions.usingComparator(comparator);
  }
  @Override protected ShortAssert invoke_api_method(){
    return assertions.usingDefaultComparator();
  }
  @Override protected void verify_internal_effects(){
    assertThat(Objects.instance()).isSameAs(getObjects(assertions));
    assertThat(Shorts.instance()).isSameAs(getShorts(assertions));
  }
}
