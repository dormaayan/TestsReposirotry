/** 
 * Tests for <code> {@link ByteArrayAssert#doesNotContain(int)}</code>.
 */
public class ByteArrayAssert_doesNotContain_with_Integer_Arguments_Test extends ByteArrayAssertBaseTest {
  @Override protected ByteArrayAssert invoke_api_method(){
    return assertions.doesNotContain(6,8);
  }
  @Override protected void verify_internal_effects(){
    verify(arrays).assertDoesNotContain(getInfo(assertions),getActual(assertions),IntArrays.arrayOf(6,8));
  }
  @Test public void invoke_api_like_user(){
    assertThat(new byte[]{1,2}).doesNotContain(3,4);
  }
}
