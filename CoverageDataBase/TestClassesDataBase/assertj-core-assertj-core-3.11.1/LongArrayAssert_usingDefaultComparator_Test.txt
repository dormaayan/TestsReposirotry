/** 
 * Tests for <code> {@link LongArrayAssert#usingDefaultComparator()}</code>.
 * @author Joel Costigliola
 * @author Mikhail Mazursky
 */
public class LongArrayAssert_usingDefaultComparator_Test extends LongArrayAssertBaseTest {
  @Mock private Comparator<long[]> comparator;
  private LongArrays arraysBefore;
  @BeforeEach public void before(){
    initMocks(this);
    assertions.usingComparator(comparator);
    arraysBefore=getArrays(assertions);
  }
  @Override protected LongArrayAssert invoke_api_method(){
    return assertions.usingDefaultComparator();
  }
  @Override protected void verify_internal_effects(){
    assertThat(Objects.instance()).isSameAs(getObjects(assertions));
    assertThat(arraysBefore).isSameAs(getArrays(assertions));
  }
}
