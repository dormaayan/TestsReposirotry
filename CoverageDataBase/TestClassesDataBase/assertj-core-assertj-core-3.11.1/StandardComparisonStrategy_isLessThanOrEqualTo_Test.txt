/** 
 * Tests for  {@link StandardComparisonStrategy#isLessThanOrEqualTo(Object,Object)}.
 * @author Joel Costigliola
 */
public class StandardComparisonStrategy_isLessThanOrEqualTo_Test extends AbstractTest_StandardComparisonStrategy {
  @Test public void should_pass(){
    Employee boss=new Employee(10000,35);
    Employee young=new Employee(10000,25);
    assertThat(standardComparisonStrategy.isLessThanOrEqualTo(young,boss)).isTrue();
    assertThat(standardComparisonStrategy.isLessThanOrEqualTo(boss,young)).isFalse();
    assertThat(standardComparisonStrategy.isLessThanOrEqualTo(boss,boss)).isTrue();
  }
  @Test public void should_fail_if_a_parameter_is_not_comparable(){
    assertThatIllegalArgumentException().isThrownBy(() -> {
      Rectangle r1=new Rectangle(10,20);
      Rectangle r2=new Rectangle(20,10);
      standardComparisonStrategy.isLessThanOrEqualTo(r1,r2);
    }
);
  }
}
