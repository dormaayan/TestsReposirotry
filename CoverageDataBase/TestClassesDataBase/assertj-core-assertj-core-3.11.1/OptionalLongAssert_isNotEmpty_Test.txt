public class OptionalLongAssert_isNotEmpty_Test extends BaseTest {
  @Test public void should_pass_when_OptionalLong_is_present(){
    assertThat(OptionalLong.of(10L)).isNotEmpty();
  }
  @Test public void should_fail_when_OptionalLong_is_empty(){
    assertThatExceptionOfType(AssertionError.class).isThrownBy(() -> assertThat(OptionalLong.empty()).isNotEmpty()).withMessage(shouldBePresent(OptionalLong.empty()).create());
  }
  @Test public void should_fail_when_OptionalLong_is_null(){
    assertThatExceptionOfType(AssertionError.class).isThrownBy(() -> assertThat((OptionalLong)null).isNotEmpty()).withMessage(actualIsNull());
  }
}
