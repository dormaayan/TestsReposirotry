/** 
 * Tests for <code> {@link FloatArrayAssert#containsSequence(float)}</code>.
 * @author Alex Ruiz
 */
public class FloatArrayAssert_containsSequence_Test extends FloatArrayAssertBaseTest {
  @Override protected FloatArrayAssert invoke_api_method(){
    return assertions.containsSequence(6f,8f);
  }
  @Override protected void verify_internal_effects(){
    verify(arrays).assertContainsSequence(getInfo(assertions),getActual(assertions),arrayOf(6f,8f));
  }
  @Test public void should_pass_with_precision_specified_as_last_argument(){
    float[] actual=arrayOf(1.0f,2.0f,3.0f);
    assertThat(actual).containsSequence(arrayOf(1.1f,2.1f),withPrecision(0.5f));
  }
  @Test public void should_pass_with_precision_specified_in_comparator(){
    float[] actual=arrayOf(1.0f,2.0f,3.0f);
    assertThat(actual).usingComparatorWithPrecision(0.5f).containsSequence(1.1f,2.1f);
  }
}
