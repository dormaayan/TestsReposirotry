public class OptionalDoubleAssert_isPresent_Test extends BaseTest {
  @Test public void should_pass_when_optionaldouble_is_present(){
    assertThat(OptionalDouble.of(10.0)).isPresent();
  }
  @Test public void should_fail_when_optionaldouble_is_empty(){
    assertThatExceptionOfType(AssertionError.class).isThrownBy(() -> assertThat(OptionalDouble.empty()).isPresent()).withMessage(shouldBePresent(OptionalDouble.empty()).create());
  }
  @Test public void should_fail_when_optionaldouble_is_null(){
    assertThatExceptionOfType(AssertionError.class).isThrownBy(() -> assertThat((OptionalDouble)null).isPresent()).withMessage(actualIsNull());
  }
}
