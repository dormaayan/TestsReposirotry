public class ObjectArrayAssert_allMatch_with_description_Test extends ObjectArrayAssertBaseTest {
  private Predicate<Object> predicate;
  @BeforeEach public void beforeOnce(){
    predicate=o -> o != null;
  }
  @Override protected ObjectArrayAssert<Object> invoke_api_method(){
    return assertions.allMatch(predicate,"custom");
  }
  @Override protected void verify_internal_effects(){
    verify(iterables).assertAllMatch(getInfo(assertions),newArrayList(getActual(assertions)),predicate,new PredicateDescription("custom"));
  }
}
