public class CaseSensitiveNamingTest {
  InternalNamingStrategy strategy=new CaseSensitiveNaming();
  @Test public void should_return_blank_name() throws Exception {
    String name="     ";
    final String actual=strategy.apply(name);
    assertThat(actual).isEmpty();
  }
  @Test public void should_return_blank_name_on_null() throws Exception {
    String name=null;
    final String actual=strategy.apply(name);
    assertThat(actual).isEmpty();
  }
  @Test public void should_return_case_sensitive() throws Exception {
    String name="dcSdIfk_sdf";
    final String actual=strategy.apply(name);
    assertThat(actual).isEqualTo("\"dcSdIfk_sdf\"");
  }
}
