public class LowerCaseNamingTest {
  private InternalNamingStrategy strategy=new LowerCaseNaming();
  @Test public void should_return_blank_name() throws Exception {
    String name="     ";
    final String actual=strategy.apply(name);
    assertThat(actual).isEmpty();
  }
  @Test public void should_return_blank_name_on_null() throws Exception {
    String name=null;
    final String actual=strategy.apply(name);
    assertThat(actual).isEmpty();
  }
  @Test public void should_return_lower_case() throws Exception {
    String name="ZfrEr_rfR";
    final String actual=strategy.apply(name);
    assertThat(actual).isEqualTo("zfrer_rfr");
  }
}
