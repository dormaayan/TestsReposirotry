@RunWith(DefaultServer.class) @AjpIgnore public class LongURLTestCase {
  private static final String MESSAGE="HelloUrl";
  private static final int COUNT=10000;
  @BeforeClass public static void setup(){
    final BlockingHandler blockingHandler=new BlockingHandler();
    DefaultServer.setRootHandler(blockingHandler);
    blockingHandler.setRootHandler(new HttpHandler(){
      @Override public void handleRequest(      final HttpServerExchange exchange){
        exchange.getResponseSender().send(exchange.getRelativePath());
      }
    }
);
  }
  @Test public void testLargeURL() throws IOException {
    StringBuilder sb=new StringBuilder();
    for (int i=0; i < COUNT; ++i) {
      sb.append(MESSAGE);
    }
    String message=sb.toString();
    TestHttpClient client=new TestHttpClient();
    try {
      HttpGet get=new HttpGet(DefaultServer.getDefaultServerURL() + "/" + message);
      HttpResponse result=client.execute(get);
      Assert.assertEquals("/" + message,HttpClientUtils.readResponse(result));
    }
  finally {
      client.getConnectionManager().shutdown();
    }
  }
}
