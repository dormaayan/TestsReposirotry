@Entity @Table(name="MyCustomCollectionBean") class CustomConvertedCollectionBean {
  @Column(name="mySet",converter=ObjectConverter.class) private Set<Integer> mySet=new HashSet<Integer>();
  public Set<Integer> getMySet(){
    return mySet;
  }
  public void setMySet(  Set<Integer> mySet){
    this.mySet=mySet;
  }
  public CustomConvertedCollectionBean addItem(  Integer myInt){
    mySet.add(myInt);
    return this;
  }
}
