public class BooleanSerializerTest {
  @Test public void testConversions(){
    test(true);
    test(false);
    test(null);
  }
  private void test(  Boolean bool){
    BooleanSerializer ext=BooleanSerializer.get();
    assertEquals(bool,ext.fromByteBuffer(ext.toByteBuffer(bool)));
  }
}
