@Entity @Table(name="MyCollectionBean") class CollectionBean {
  @Column(name="mySet") private Set<Integer> mySet=new HashSet<Integer>();
  public Set<Integer> getMySet(){
    return mySet;
  }
  public void setMySet(  Set<Integer> mySet){
    this.mySet=mySet;
  }
  public CollectionBean addItem(  Integer myInt){
    mySet.add(myInt);
    return this;
  }
}
