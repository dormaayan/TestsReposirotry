static class MockPoolEntry extends PoolEntry<String,HttpConnection> {
  public MockPoolEntry(  final String route,  final long timeToLive,  final TimeUnit timeUnit){
    super(null,route,Mockito.mock(HttpConnection.class),timeToLive,timeUnit);
  }
  public MockPoolEntry(  final String route,  final HttpConnection conn,  final long timeToLive,  final TimeUnit timeUnit){
    super(null,route,conn,timeToLive,timeUnit);
  }
  @Override public void close(){
    try {
      getConnection().close();
    }
 catch (    final IOException ignore) {
    }
  }
  @Override public boolean isClosed(){
    return !getConnection().isOpen();
  }
}
