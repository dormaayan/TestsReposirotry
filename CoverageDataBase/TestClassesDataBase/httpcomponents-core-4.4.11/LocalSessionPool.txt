static class LocalSessionPool extends AbstractNIOConnPool<String,IOSession,LocalPoolEntry> {
  public LocalSessionPool(  final ConnectingIOReactor ioReactor,  final int defaultMaxPerRoute,  final int maxTotal){
    super(ioReactor,new LocalConnFactory(),new LocalAddressResolver(),defaultMaxPerRoute,maxTotal);
  }
  public LocalSessionPool(  final ConnectingIOReactor ioReactor,  final SocketAddressResolver<String> addressResolver,  final int defaultMaxPerRoute,  final int maxTotal){
    super(ioReactor,new LocalConnFactory(),addressResolver,defaultMaxPerRoute,maxTotal);
  }
  @Override protected LocalPoolEntry createEntry(  final String route,  final IOSession session){
    return new LocalPoolEntry(route,session);
  }
}
