private static final class HashtableFactory extends SizedMapFactory implements Function0<Hashtable<Integer,String>> {
  private HashtableFactory(  int size,  float loadFactor){
    super(size,loadFactor);
  }
  @Override @SuppressWarnings("UseOfObsoleteCollectionType") public Hashtable<Integer,String> value(){
    int defaultSize=11;
    Hashtable<Integer,String> map=new Hashtable<>(defaultSize,this.loadFactor);
    return this.fill(map);
  }
}
