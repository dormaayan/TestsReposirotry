private static final class THashMapFactory extends SizedMapFactory implements Function0<THashMap<Integer,String>> {
  private THashMapFactory(  int size,  float loadFactor){
    super(size,loadFactor);
  }
  @Override public THashMap<Integer,String> value(){
    return this.fill(new THashMap<Integer,String>(Constants.DEFAULT_CAPACITY,this.loadFactor));
  }
}
