@RunWith(Java8Runner.class) public class CompositeFastListTest implements MutableListTestCase {
  @SafeVarargs @Override public final <T>MutableList<T> newWith(  T... elements){
    RichIterable<RichIterable<T>> chunks=FastList.newListWith(elements).chunk(3);
    CompositeFastList<T> result=new CompositeFastList<>();
    for (    RichIterable<T> chunk : chunks) {
      result.addAll(FastList.newList(chunk));
    }
    return result;
  }
}
