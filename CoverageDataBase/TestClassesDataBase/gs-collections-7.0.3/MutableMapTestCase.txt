public interface MutableMapTestCase extends UnsortedMapIterableTestCase, MutableUnorderedIterableTestCase {
  @Override <T>MutableMap<Object,T> newWith(  T... elements);
  @Override default void Iterable_remove(){
    MutableMap<Object,Integer> iterable=this.newWith(3,3,3,2,2,1);
    Iterator<Integer> iterator=iterable.iterator();
    iterator.next();
    iterator.remove();
    assertEquals(this.allowsDuplicates() ? 5 : 2,Iterate.sizeOf(iterable));
    assertThat(iterable.toBag(),isOneOf(this.getExpectedFiltered(3,3,3,2,2),this.getExpectedFiltered(3,3,3,2,1),this.getExpectedFiltered(3,3,2,2,1)));
  }
}
