public abstract static class SizedListFactory {
  protected final int size;
  protected SizedListFactory(  int size){
    this.size=size;
  }
  protected <R extends List<String>>R fill(  R list){
    if (this.size > 0) {
      list.addAll(Collections.nCopies(this.size,"dummy"));
    }
    return list;
  }
}
