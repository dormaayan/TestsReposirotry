private static final class UnifiedMapFactory extends SizedMapFactory implements Function0<UnifiedMap<Integer,String>> {
  private UnifiedMapFactory(  int size,  float loadFactor){
    super(size,loadFactor);
  }
  @Override public UnifiedMap<Integer,String> value(){
    int defaultSize=8;
    UnifiedMap<Integer,String> map=new UnifiedMap<Integer,String>(defaultSize,this.loadFactor);
    return this.fill(map);
  }
}
