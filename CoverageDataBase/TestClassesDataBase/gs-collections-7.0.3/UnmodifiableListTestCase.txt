public interface UnmodifiableListTestCase extends UnmodifiableCollectionTestCase, ListTestCase {
  @Override @Test(expected=UnsupportedOperationException.class) default void List_set(){
    List<Integer> list=this.newWith(1,2,3);
    list.set(1,4);
  }
  @Override @Test(expected=UnsupportedOperationException.class) default void List_set_negative(){
    List<Integer> list=this.newWith(1,2,3);
    list.set(-1,4);
  }
  @Override @Test(expected=UnsupportedOperationException.class) default void List_set_out_of_bounds(){
    List<Integer> list=this.newWith(1,2,3);
    list.set(4,4);
  }
}
