class Bar {
  private final int value;
  Bar(  int value){
    this.value=value;
  }
  public int getValue(){
    return this.value;
  }
  @Override public boolean equals(  Object o){
    if (this == o) {
      return true;
    }
    if (o == null || this.getClass() != o.getClass()) {
      return false;
    }
    Bar bar=(Bar)o;
    return this.value == bar.value;
  }
  @Override public int hashCode(){
    return this.value;
  }
}
