public class FastListSerializationTest {
  public static final String FAST_LIST_WITH_ONE_NULL="rO0ABXNyAC1jb20uZ3MuY29sbGVjdGlvbnMuaW1wbC5saXN0Lm11dGFibGUuRmFzdExpc3QAAAAA\n" + "AAAAAQwAAHhwdwQAAAABcHg=";
  public static final String FAST_LIST_EMPTY="rO0ABXNyAC1jb20uZ3MuY29sbGVjdGlvbnMuaW1wbC5saXN0Lm11dGFibGUuRmFzdExpc3QAAAAA\n" + "AAAAAQwAAHhwdwQAAAAAeA==";
  @Test public void serializedForm(){
    Verify.assertSerializedForm(1L,FAST_LIST_EMPTY,FastList.newList());
  }
  @Test public void listWithOneNull(){
    Verify.assertSerializedForm(1L,FAST_LIST_WITH_ONE_NULL,FastList.newListWith((Object)null));
  }
  @Test public void subList(){
    Verify.assertSerializedForm(FAST_LIST_WITH_ONE_NULL,FastList.newListWith(null,null).subList(0,1));
  }
}
