private static final class PresizedTHashMapFactory extends SizedMapFactory implements Function0<THashMap<Integer,String>> {
  private PresizedTHashMapFactory(  int size,  float loadFactor){
    super(size,loadFactor);
  }
  @Override public THashMap<Integer,String> value(){
    return this.fill(new THashMap<Integer,String>(this.size,this.loadFactor));
  }
}
