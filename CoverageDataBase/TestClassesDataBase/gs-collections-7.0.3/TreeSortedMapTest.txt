@RunWith(Java8Runner.class) public class TreeSortedMapTest implements MutableSortedMapIterableTestCase {
  @Override public <T>MutableSortedMap<Object,T> newWith(  T... elements){
    int i=elements.length;
    MutableSortedMap<Object,T> result=new TreeSortedMap<>(Comparators.reverseNaturalOrder());
    for (    T each : elements) {
      assertNull(result.put(i,each));
      i--;
    }
    return result;
  }
}
