public interface UnmodifiableCollectionTestCase extends CollectionTestCase, UnmodifiableIterableTestCase {
  @Override @Test(expected=UnsupportedOperationException.class) default void Collection_add(){
    this.newWith(1,2,3).add(4);
  }
  @Override @Test(expected=UnsupportedOperationException.class) default void Collection_clear(){
    this.newWith(1,2,3).clear();
  }
  @Override @Test default void Iterable_remove(){
    UnmodifiableIterableTestCase.super.Iterable_remove();
  }
}
