@Test public void test_IO_246(){
  assertMatch("aaa","aa*?",true);
  assertMatch("","?*",false);
  assertMatch("a","a?*",false);
  assertMatch("aa","aa?*",false);
  assertMatch("a","?*",true);
  assertMatch("aa","?*",true);
  assertMatch("aaa","?*",true);
  assertMatch("","?",false);
  assertMatch("a","a?",false);
  assertMatch("aa","aa?",false);
  assertMatch("aab","aa?",true);
  assertMatch("aaa","*a",true);
}
