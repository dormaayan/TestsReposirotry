@Test public void testNormalizeNoEndSeparatorUnixWin() throws Exception {
  assertEquals("/a/c",FilenameUtils.normalizeNoEndSeparator("/a/b/../c/",true));
  assertEquals("/a/c",FilenameUtils.normalizeNoEndSeparator("\\a\\b\\..\\c\\",true));
  assertEquals("\\a\\c",FilenameUtils.normalizeNoEndSeparator("/a/b/../c/",false));
  assertEquals("\\a\\c",FilenameUtils.normalizeNoEndSeparator("\\a\\b\\..\\c\\",false));
}
