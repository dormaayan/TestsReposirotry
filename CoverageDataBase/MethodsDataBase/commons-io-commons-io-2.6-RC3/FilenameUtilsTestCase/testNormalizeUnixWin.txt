@Test public void testNormalizeUnixWin() throws Exception {
  assertEquals("/a/c/",FilenameUtils.normalize("/a/b/../c/",true));
  assertEquals("/a/c/",FilenameUtils.normalize("\\a\\b\\..\\c\\",true));
  assertEquals("\\a\\c\\",FilenameUtils.normalize("/a/b/../c/",false));
  assertEquals("\\a\\c\\",FilenameUtils.normalize("\\a\\b\\..\\c\\",false));
}
