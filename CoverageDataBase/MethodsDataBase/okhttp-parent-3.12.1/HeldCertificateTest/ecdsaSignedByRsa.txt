@Test public void ecdsaSignedByRsa(){
  HeldCertificate root=new HeldCertificate.Builder().certificateAuthority(0).rsa2048().build();
  HeldCertificate leaf=new HeldCertificate.Builder().certificateAuthority(0).ecdsa256().signedBy(root).build();
  assertEquals("SHA256WITHRSA",root.certificate().getSigAlgName());
  assertEquals("SHA256WITHRSA",leaf.certificate().getSigAlgName());
}
