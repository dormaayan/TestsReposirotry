@Test public void multipleChallengesInOneHeader(){
  Headers headers=new Headers.Builder().add("WWW-Authenticate","Basic realm = \"myrealm\",Digest").build();
  assertEquals(Arrays.asList(new Challenge("Basic",singletonMap("realm","myrealm")),new Challenge("Digest",Collections.<String,String>emptyMap())),HttpHeaders.parseChallenges(headers,"WWW-Authenticate"));
}
