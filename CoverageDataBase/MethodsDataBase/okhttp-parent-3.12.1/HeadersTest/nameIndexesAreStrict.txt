@Test public void nameIndexesAreStrict(){
  Headers headers=Headers.of("a","b","c","d");
  try {
    headers.name(-1);
    fail();
  }
 catch (  IndexOutOfBoundsException expected) {
  }
  assertEquals("a",headers.name(0));
  assertEquals("c",headers.name(1));
  try {
    headers.name(2);
    fail();
  }
 catch (  IndexOutOfBoundsException expected) {
  }
}
