@Test public void escapedCharactersInQuotedString(){
  Headers headers=new Headers.Builder().add("WWW-Authenticate","Digest,,,, Basic ,,,realm=\"my\\\\\\\"r\\ealm\"").build();
  assertEquals(Arrays.asList(new Challenge("Digest",Collections.<String,String>emptyMap()),new Challenge("Basic",singletonMap("realm","my\\\"realm"))),HttpHeaders.parseChallenges(headers,"WWW-Authenticate"));
}
