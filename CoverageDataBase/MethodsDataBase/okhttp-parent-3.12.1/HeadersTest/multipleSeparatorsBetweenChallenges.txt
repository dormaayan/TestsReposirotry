@Test public void multipleSeparatorsBetweenChallenges(){

  assertEquals(Arrays.asList(new Challenge("Digest",Collections.<String,String>emptyMap()),new Challenge("Basic",singletonMap("realm","myrealm"))),HttpHeaders.parseChallenges(headers,"WWW-Authenticate"));
}
