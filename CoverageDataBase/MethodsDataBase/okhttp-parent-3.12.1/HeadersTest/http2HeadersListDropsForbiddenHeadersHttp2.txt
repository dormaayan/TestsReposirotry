@Test public void http2HeadersListDropsForbiddenHeadersHttp2(){
  Request request=new Request.Builder().url("http://square.com/").header("Connection","upgrade").header("Upgrade","websocket").header("Host","square.com").build();
  List<Header> expected=headerEntries(":method","GET",":path","/",":authority","square.com",":scheme","http");
  assertEquals(expected,Http2Codec.http2HeadersList(request));
}
