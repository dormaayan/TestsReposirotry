@Test public void multipleChallengesWithSameSchemeButDifferentRealmInOneHeader(){
  Headers headers=new Headers.Builder().add("WWW-Authenticate","Basic realm = \"myrealm\",Basic realm=myotherrealm").build();
  assertEquals(Arrays.asList(new Challenge("Basic",singletonMap("realm","myrealm")),new Challenge("Basic",singletonMap("realm","myotherrealm"))),HttpHeaders.parseChallenges(headers,"WWW-Authenticate"));
}
