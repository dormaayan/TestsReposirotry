@Test public void valueIndexesAreStrict(){
  Headers headers=Headers.of("a","b","c","d");
  try {
    headers.value(-1);
    fail();
  }
 catch (  IndexOutOfBoundsException expected) {
  }
  assertEquals("b",headers.value(0));
  assertEquals("d",headers.value(1));
  try {
    headers.value(2);
    fail();
  }
 catch (  IndexOutOfBoundsException expected) {
  }
}
