@Test public void escapedChars(){
  assertCn(",","cn=\\,");
  assertCn("#","cn=\\#");
  assertCn("+","cn=\\+");
  assertCn("\"","cn=\\\"");
  assertCn("\\","cn=\\\\");
  assertCn("<","cn=\\<");
  assertCn(">","cn=\\>");
  assertCn(";","cn=\\;");
  assertCn("+","cn=\\+");
  assertCn("+","cn=\"\\+\"");
  assertCn(",","cn=\"\\,\"");
  assertCn("a =","cn= a =");
  assertCn("=","cn==");
}
