@Test public void applicationInterceptorsCanShortCircuitResponses() throws Exception {
  server.shutdown();
  Request request=new Request.Builder().url("https://localhost:1/").build();
  final Response interceptorResponse=new Response.Builder().request(request).protocol(Protocol.HTTP_1_1).code(200).message("Intercepted!").body(ResponseBody.create(MediaType.get("text/plain; charset=utf-8"),"abc")).build();
  client=client.newBuilder().addInterceptor(new Interceptor(){
    @Override public Response intercept(    Chain chain) throws IOException {
      return interceptorResponse;
    }
  }
).build();
  Response response=client.newCall(request).execute();
  assertSame(interceptorResponse,response);
}
