@Test public void resolveSchemeLikePath() throws Exception {
  HttpUrl base=parse("http://a/");
  assertEquals(parse("http://a/http//b/"),base.resolve("http//b/"));
  assertEquals(parse("http://a/ht+tp//b/"),base.resolve("ht+tp//b/"));
  assertEquals(parse("http://a/ht-tp//b/"),base.resolve("ht-tp//b/"));
  assertEquals(parse("http://a/ht.tp//b/"),base.resolve("ht.tp//b/"));
}
