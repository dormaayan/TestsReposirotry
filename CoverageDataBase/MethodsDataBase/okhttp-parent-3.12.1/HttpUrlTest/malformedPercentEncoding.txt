@Test public void malformedPercentEncoding(){
  assertEquals(Arrays.asList("a%f","b"),parse("http://host/a%f/b").pathSegments());
  assertEquals(Arrays.asList("%","b"),parse("http://host/%/b").pathSegments());
  assertEquals(Arrays.asList("%"),parse("http://host/%").pathSegments());
  assertEquals(Arrays.asList("%00"),parse("http://github.com/%%30%30").pathSegments());
}
