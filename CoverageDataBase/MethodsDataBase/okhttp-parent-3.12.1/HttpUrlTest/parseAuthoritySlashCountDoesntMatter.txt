@Test public void parseAuthoritySlashCountDoesntMatter() throws Exception {
  assertEquals(parse("http://host/path"),parse("http:host/path"));
  assertEquals(parse("http://host/path"),parse("http:/host/path"));
  assertEquals(parse("http://host/path"),parse("http:\\host/path"));
  assertEquals(parse("http://host/path"),parse("http://host/path"));
  assertEquals(parse("http://host/path"),parse("http:\\/host/path"));
  assertEquals(parse("http://host/path"),parse("http:/\\host/path"));
  assertEquals(parse("http://host/path"),parse("http:\\\\host/path"));
  assertEquals(parse("http://host/path"),parse("http:///host/path"));
  assertEquals(parse("http://host/path"),parse("http:\\//host/path"));
  assertEquals(parse("http://host/path"),parse("http:/\\/host/path"));
  assertEquals(parse("http://host/path"),parse("http://\\host/path"));
  assertEquals(parse("http://host/path"),parse("http:\\\\/host/path"));
  assertEquals(parse("http://host/path"),parse("http:/\\\\host/path"));
  assertEquals(parse("http://host/path"),parse("http:\\\\\\host/path"));
  assertEquals(parse("http://host/path"),parse("http:////host/path"));
}
