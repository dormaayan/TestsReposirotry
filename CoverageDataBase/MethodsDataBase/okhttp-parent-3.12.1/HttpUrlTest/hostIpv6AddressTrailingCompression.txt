@Test public void hostIpv6AddressTrailingCompression() throws Exception {
  assertEquals("1::",parse("http://[0001:0000::]").host());
  assertEquals("1::",parse("http://[0001::0000]").host());
  assertEquals("1::",parse("http://[0001::]").host());
  assertEquals("1::",parse("http://[1::]").host());
}
