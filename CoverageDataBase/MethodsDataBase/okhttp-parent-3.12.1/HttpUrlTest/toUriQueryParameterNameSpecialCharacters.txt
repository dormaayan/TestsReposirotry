@Test public void toUriQueryParameterNameSpecialCharacters() throws Exception {
  HttpUrl url=new HttpUrl.Builder().scheme("http").host("host").addQueryParameter("=[]:;\"~|?#@^/$%*","a").build();
  assertEquals("http://host/?%3D%5B%5D%3A%3B%22%7E%7C%3F%23%40%5E%2F%24%25*=a",url.toString());
  assertEquals("http://host/?%3D%5B%5D%3A%3B%22%7E%7C%3F%23%40%5E%2F%24%25*=a",url.uri().toString());
  assertEquals("a",url.queryParameter("=[]:;\"~|?#@^/$%*"));
}
