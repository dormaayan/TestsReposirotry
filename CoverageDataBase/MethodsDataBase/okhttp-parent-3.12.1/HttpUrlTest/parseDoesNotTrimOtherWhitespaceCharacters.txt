@Test public void parseDoesNotTrimOtherWhitespaceCharacters() throws Exception {
  assertEquals("/%0B",parse("http://h/\u000b").encodedPath());
  assertEquals("/%1C",parse("http://h/\u001c").encodedPath());
  assertEquals("/%1D",parse("http://h/\u001d").encodedPath());
  assertEquals("/%1E",parse("http://h/\u001e").encodedPath());
  assertEquals("/%1F",parse("http://h/\u001f").encodedPath());
  assertEquals("/%C2%85",parse("http://h/\u0085").encodedPath());
  assertEquals("/%C2%A0",parse("http://h/\u00a0").encodedPath());
  assertEquals("/%E1%9A%80",parse("http://h/\u1680").encodedPath());
  assertEquals("/%E1%A0%8E",parse("http://h/\u180e").encodedPath());
  assertEquals("/%E2%80%80",parse("http://h/\u2000").encodedPath());
  assertEquals("/%E2%80%81",parse("http://h/\u2001").encodedPath());
  assertEquals("/%E2%80%82",parse("http://h/\u2002").encodedPath());
  assertEquals("/%E2%80%83",parse("http://h/\u2003").encodedPath());
  assertEquals("/%E2%80%84",parse("http://h/\u2004").encodedPath());
  assertEquals("/%E2%80%85",parse("http://h/\u2005").encodedPath());
  assertEquals("/%E2%80%86",parse("http://h/\u2006").encodedPath());
  assertEquals("/%E2%80%87",parse("http://h/\u2007").encodedPath());
  assertEquals("/%E2%80%88",parse("http://h/\u2008").encodedPath());
  assertEquals("/%E2%80%89",parse("http://h/\u2009").encodedPath());
  assertEquals("/%E2%80%8A",parse("http://h/\u200a").encodedPath());
  assertEquals("/%E2%80%8B",parse("http://h/\u200b").encodedPath());
  assertEquals("/%E2%80%8C",parse("http://h/\u200c").encodedPath());
  assertEquals("/%E2%80%8D",parse("http://h/\u200d").encodedPath());
  assertEquals("/%E2%80%8E",parse("http://h/\u200e").encodedPath());
  assertEquals("/%E2%80%8F",parse("http://h/\u200f").encodedPath());
  assertEquals("/%E2%80%A8",parse("http://h/\u2028").encodedPath());
  assertEquals("/%E2%80%A9",parse("http://h/\u2029").encodedPath());
  assertEquals("/%E2%80%AF",parse("http://h/\u202f").encodedPath());
  assertEquals("/%E2%81%9F",parse("http://h/\u205f").encodedPath());
  assertEquals("/%E3%80%80",parse("http://h/\u3000").encodedPath());
}
