/** 
 * If a cookie incorrectly defines multiple 'Expires' attributes, the last one defined wins. 
 */
@Test public void lastExpiresAtWins() throws Exception {
  assertEquals(3000L,Cookie.parse(0L,url,"a=b; " + "Expires=Thu, 01 Jan 1970 00:00:02 GMT; " + "Expires=Thu, 01 Jan 1970 00:00:04 GMT; "+ "Expires=Thu, 01 Jan 1970 00:00:01 GMT; "+ "Expires=Thu, 01 Jan 1970 00:00:03 GMT").expiresAt());
}
