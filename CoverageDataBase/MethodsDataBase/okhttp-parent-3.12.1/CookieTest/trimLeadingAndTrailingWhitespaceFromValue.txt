@Test public void trimLeadingAndTrailingWhitespaceFromValue() throws Exception {
  assertEquals("",Cookie.parse(url,"a= ").value());
  assertEquals("b",Cookie.parse(url,"a= b").value());
  assertEquals("b",Cookie.parse(url,"a=b ").value());
