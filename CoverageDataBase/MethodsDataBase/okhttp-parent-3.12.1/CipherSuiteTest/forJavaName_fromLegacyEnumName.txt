/** 
 * On the Oracle JVM some older cipher suites have the "SSL_" prefix and others have the "TLS_" prefix. On the IBM JVM all cipher suites have the "SSL_" prefix. <p>Prior to OkHttp 3.3.1 we accepted either form and consider them equivalent. And since OkHttp 3.7.0 this is also true. But OkHttp 3.3.1 through 3.6.0 treated these as different.
 */
@Test public void forJavaName_fromLegacyEnumName(){
  assertEquals(forJavaName("TLS_RSA_EXPORT_WITH_RC4_40_MD5"),forJavaName("SSL_RSA_EXPORT_WITH_RC4_40_MD5"));
  assertEquals(forJavaName("TLS_DH_RSA_EXPORT_WITH_DES40_CBC_SHA"),forJavaName("SSL_DH_RSA_EXPORT_WITH_DES40_CBC_SHA"));
  assertEquals(forJavaName("TLS_FAKE_NEW_CIPHER"),forJavaName("SSL_FAKE_NEW_CIPHER"));
}
