/** 
 * Tests that interned CipherSuite instances remain the case across garbage collections, even if the String used to construct them is no longer strongly referenced outside of the CipherSuite.
 */
@SuppressWarnings("RedundantStringConstructorCall") @Test public void instancesAreInterned_survivesGarbageCollection(){
  CipherSuite cs=forJavaName(new String("FakeCipherSuite_instancesAreInterned"));
  System.gc();
  assertSame(cs,forJavaName(new String(cs.javaName())));
}
