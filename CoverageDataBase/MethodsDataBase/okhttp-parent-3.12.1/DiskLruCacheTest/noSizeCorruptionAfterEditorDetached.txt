@Test public void noSizeCorruptionAfterEditorDetached() throws Exception {
  set("k1","a","a");
  DiskLruCache.Editor editor=cache.edit("k1");
  setString(editor,0,"bb");
  setString(editor,1,"bb");
  cache.evictAll();
  set("k1","ccc","ccc");
  assertEquals(6,cache.size());
  editor.commit();
  assertEquals(6,cache.size());
  assertValue("k1","ccc","ccc");
}
