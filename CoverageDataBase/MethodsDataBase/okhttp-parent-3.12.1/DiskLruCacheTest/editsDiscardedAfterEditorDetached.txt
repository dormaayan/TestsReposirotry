@Test public void editsDiscardedAfterEditorDetached() throws Exception {
  set("k1","a","a");
  DiskLruCache.Editor editor=cache.edit("k1");
  BufferedSink sink=Okio.buffer(editor.newSink(0));
  cache.evictAll();
  set("k1","ccc","ccc");
  sink.writeUtf8("bb");
  sink.close();
  assertValue("k1","ccc","ccc");
}
