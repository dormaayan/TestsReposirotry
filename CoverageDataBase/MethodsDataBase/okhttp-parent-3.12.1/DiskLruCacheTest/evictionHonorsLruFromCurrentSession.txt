@Test public void evictionHonorsLruFromCurrentSession() throws Exception {
  cache.close();
  createNewCacheWithSize(10);
  set("a","a","a");
  set("b","b","b");
  set("c","c","c");
  set("d","d","d");
  set("e","e","e");
  cache.get("b").close();
  set("f","f","f");
  set("g","g","g");
  cache.flush();
  assertEquals(10,cache.size());
  assertAbsent("a");
  assertValue("b","b","b");
  assertAbsent("c");
  assertValue("d","d","d");
  assertValue("e","e","e");
  assertValue("f","f","f");
}
