@Test public void evictOnUpdate() throws Exception {
  cache.close();
  createNewCacheWithSize(10);
  set("a","a","aa");
  set("b","b","bb");
  set("c","c","cc");
  assertEquals(9,cache.size());
  set("b","b","bbbb");
  cache.flush();
  assertEquals(8,cache.size());
  assertAbsent("a");
  assertValue("b","b","bbbb");
  assertValue("c","c","cc");
}
