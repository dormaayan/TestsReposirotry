@Test public void noSizeCorruptionAfterCreatorDetached() throws Exception {
  DiskLruCache.Editor editor=cache.edit("k1");
  setString(editor,0,"a");
  setString(editor,1,"a");
  cache.evictAll();
  set("k1","bb","bb");
  assertEquals(4,cache.size());
  editor.commit();
  assertEquals(4,cache.size());
  assertValue("k1","bb","bb");
}
