@Test public void expect100ContinueNonEmptyRequestBody() throws Exception {
  server.enqueue(new MockResponse().setSocketPolicy(SocketPolicy.EXPECT_CONTINUE));
  Request request=new Request.Builder().url(server.url("/")).header("Expect","100-continue").post(RequestBody.create(MediaType.get("text/plain"),"abc")).build();
  executeSynchronously(request).assertCode(200).assertSuccessful();
  assertEquals("abc",server.takeRequest().getBody().readUtf8());
}
