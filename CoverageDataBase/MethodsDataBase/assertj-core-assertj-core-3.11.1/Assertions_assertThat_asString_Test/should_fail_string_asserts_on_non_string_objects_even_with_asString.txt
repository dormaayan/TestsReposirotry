@Test public void should_fail_string_asserts_on_non_string_objects_even_with_asString(){
  Object nonString=new Object();
  assertThatExceptionOfType(AssertionError.class).isThrownBy(() -> assertThat(nonString).asString().contains("hello")).withMessageContaining(format("an instance of:%n  <java.lang.String>%nbut was instance of:%n  <java.lang.Object>"));
}
