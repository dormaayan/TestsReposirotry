@Test public void test_isBefore_assertion_error_message(){
  assertThatExceptionOfType(AssertionError.class).isThrownBy(() -> assertThat(ZonedDateTime.of(2000,1,5,3,0,0,0,UTC)).isBefore(ZonedDateTime.of(1998,1,1,3,3,0,0,UTC))).withMessage(format("%nExpecting:%n  <2000-01-05T03:00Z>%nto be strictly before:%n  <1998-01-01T03:03Z>"));
}
