@SuppressWarnings("unchecked") @Test public void class_soft_assertions_should_report_errors_on_final_methods(){
  Class<AnnotatedClass> actual=AnnotatedClass.class;
  softly.then(actual).hasAnnotations(MyAnnotation.class,AnotherAnnotation.class).hasAnnotations(SafeVarargs.class,VisibleForTesting.class);
  List<Throwable> errorsCollected=softly.errorsCollected();
  assertThat(errorsCollected).hasSize(1);
  assertThat(errorsCollected.get(0)).hasMessageContaining("SafeVarargs").hasMessageContaining("VisibleForTesting");
}
