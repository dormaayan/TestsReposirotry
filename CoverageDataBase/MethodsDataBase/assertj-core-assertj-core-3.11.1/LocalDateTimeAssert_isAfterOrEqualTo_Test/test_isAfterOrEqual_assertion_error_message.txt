@Test public void test_isAfterOrEqual_assertion_error_message(){
  assertThatExceptionOfType(AssertionError.class).isThrownBy(() -> assertThat(LocalDateTime.of(2000,1,5,3,0,5)).isAfterOrEqualTo(LocalDateTime.of(2012,1,1,3,3,3))).withMessage(format("%n" + "Expecting:%n" + "  <2000-01-05T03:00:05>%n"+ "to be after or equals to:%n"+ "  <2012-01-01T03:03:03>"));
}
