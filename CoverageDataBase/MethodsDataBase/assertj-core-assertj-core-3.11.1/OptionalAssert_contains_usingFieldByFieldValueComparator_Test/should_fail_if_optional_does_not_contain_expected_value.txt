@Test public void should_fail_if_optional_does_not_contain_expected_value(){
  Optional<Foo> actual=Optional.of(new Foo("something"));
  Foo expectedValue=new Foo("something else");
  assertThatExceptionOfType(AssertionError.class).isThrownBy(() -> assertThat(actual).usingFieldByFieldValueComparator().contains(expectedValue)).withMessage(shouldContain(actual,expectedValue).create());
}
