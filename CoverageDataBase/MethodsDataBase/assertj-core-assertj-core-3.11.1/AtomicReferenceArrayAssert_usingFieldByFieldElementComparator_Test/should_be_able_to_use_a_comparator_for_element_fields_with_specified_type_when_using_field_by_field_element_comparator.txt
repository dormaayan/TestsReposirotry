@Test public void should_be_able_to_use_a_comparator_for_element_fields_with_specified_type_when_using_field_by_field_element_comparator(){
  Jedi actual=new Jedi("Yoda","green");
  Jedi other=new Jedi("Luke","blue");
  assertThat(atomicArrayOf(actual)).usingComparatorForElementFieldsWithType(ALWAY_EQUALS_STRING,String.class).usingFieldByFieldElementComparator().contains(other);
}
