@Test public void should_fail_if_completable_future_is_not_completed_exceptionally(){
  CompletableFuture<String> future=new CompletableFuture<>();
  assertThatThrownBy(() -> assertThat(future).isCompletedExceptionally()).isInstanceOf(AssertionError.class).hasMessage(shouldHaveCompletedExceptionally(future).create());
}
