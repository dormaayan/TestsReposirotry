@Test public void should_allow_assertions_on_extractor_assertions_extracted_from_given_array_compatibility_runtimeexception(){
  assertThatExceptionOfType(RuntimeException.class).isThrownBy(() -> assertThat(jedis).extracting(new Extractor<Employee,String>(){
    @Override public String extract(    Employee input){
      if (input.getAge() > 100) {
        throw new RuntimeException("age > 100");
      }
      return input.getName().getFirst();
    }
  }
));
}
