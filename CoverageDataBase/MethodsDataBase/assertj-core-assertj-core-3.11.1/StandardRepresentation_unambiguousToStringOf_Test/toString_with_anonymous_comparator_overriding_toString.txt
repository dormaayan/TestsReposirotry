@Test public void toString_with_anonymous_comparator_overriding_toString(){
  Comparator<String> anonymousComparator=new Comparator<String>(){
    @Override public int compare(    String s1,    String s2){
      return s1.length() - s2.length();
    }
    @Override public String toString(){
      return "foo";
    }
  }
;
  assertThat(unambiguousToStringOf(anonymousComparator)).isEqualTo(format("foo (%s@%s)",anonymousComparator.getClass().getName(),toHexString(System.identityHashCode(anonymousComparator))));
}
