@Test public void should_be_able_to_use_a_comparator_for_specified_fields_of_elements_when_using_element_comparator_on_fields(){
  Jedi actual=new Jedi("Yoda","green");
  Jedi other=new Jedi("Luke","green");
  assertThat(singletonList(actual)).usingComparatorForElementFieldsWithNames(ALWAY_EQUALS_STRING,"name").usingElementComparatorOnFields("name","lightSaberColor").contains(other);
}
