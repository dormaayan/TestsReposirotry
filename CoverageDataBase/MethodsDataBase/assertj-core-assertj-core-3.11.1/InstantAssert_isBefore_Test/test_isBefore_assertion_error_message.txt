@Test public void test_isBefore_assertion_error_message(){
  Instant instantReference=Instant.parse("2007-12-03T10:15:30.00Z");
  Instant instantAfter=Instant.parse("2007-12-03T10:15:35.00Z");
  assertThatExceptionOfType(AssertionError.class).isThrownBy(() -> assertThat(instantAfter).isBefore(instantReference)).withMessage(format("%nExpecting:%n  <2007-12-03T10:15:35Z>%nto be strictly before:%n  <2007-12-03T10:15:30Z>"));
}
