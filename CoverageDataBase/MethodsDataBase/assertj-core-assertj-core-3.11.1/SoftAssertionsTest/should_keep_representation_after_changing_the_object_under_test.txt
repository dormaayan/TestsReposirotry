@Test public void should_keep_representation_after_changing_the_object_under_test(){
  List<Name> names=asList(name("John","Doe"),name("Jane","Doe"));
  softly.assertThat(names).as("unicode").withRepresentation(UNICODE_REPRESENTATION).extracting(throwingFirstNameExtractor).contains("รณ");
  List<Throwable> errorsCollected=softly.errorsCollected();
  assertThat(errorsCollected).hasSize(1);
  assertThat(errorsCollected.get(0)).hasMessageContaining("unicode").hasMessageContaining("\\u00f3");
}
