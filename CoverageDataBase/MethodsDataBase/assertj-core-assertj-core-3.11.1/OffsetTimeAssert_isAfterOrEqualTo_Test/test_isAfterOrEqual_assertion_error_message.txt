@Test public void test_isAfterOrEqual_assertion_error_message(){
  assertThatExceptionOfType(AssertionError.class).isThrownBy(() -> assertThat(OffsetTime.of(3,0,5,0,ZoneOffset.UTC)).isAfterOrEqualTo(OffsetTime.of(3,3,3,0,ZoneOffset.UTC))).withMessage(format("%n" + "Expecting:%n" + "  <03:00:05Z>%n"+ "to be after or equals to:%n"+ "  <03:03:03Z>"));
}
