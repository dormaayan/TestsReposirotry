@Test public void test_isAfter_assertion_error_message(){
  Instant instantReference=Instant.parse("2007-12-03T10:15:30.00Z");
  Instant instantAfter=Instant.parse("2007-12-03T10:15:35.00Z");
  assertThatExceptionOfType(AssertionError.class).isThrownBy(() -> assertThat(instantReference).isAfter(instantAfter)).withMessage(format("%n" + "Expecting:%n" + "  <2007-12-03T10:15:30Z>%n"+ "to be strictly after:%n"+ "  <2007-12-03T10:15:35Z>"));
}
