@Test public void should_fail_if_optional_contains_other_type_than_required(){
  Optional<ParentClass> actual=Optional.of(new ParentClass());
  Throwable thrown=catchThrowable(() -> {
    assertThat(actual).containsInstanceOf(OtherClass.class);
  }
);
  assertThat(thrown).isInstanceOf(AssertionError.class).hasMessage(shouldContainInstanceOf(actual,OtherClass.class).create());
}
