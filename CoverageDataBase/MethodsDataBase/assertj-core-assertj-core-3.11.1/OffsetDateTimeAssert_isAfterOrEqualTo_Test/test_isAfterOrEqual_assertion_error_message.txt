@Test public void test_isAfterOrEqual_assertion_error_message(){
  assertThatExceptionOfType(AssertionError.class).isThrownBy(() -> assertThat(of(2000,1,5,3,0,5,0,UTC)).isAfterOrEqualTo(of(2012,1,1,3,3,3,0,UTC))).withMessage(format("%n" + "Expecting:%n" + "  <2000-01-05T03:00:05Z>%n"+ "to be after or equals to:%n"+ "  <2012-01-01T03:03:03Z>"));
}
