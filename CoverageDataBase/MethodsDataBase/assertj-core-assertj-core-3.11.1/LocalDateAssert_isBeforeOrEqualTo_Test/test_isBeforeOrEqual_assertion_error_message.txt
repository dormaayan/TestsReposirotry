@Test public void test_isBeforeOrEqual_assertion_error_message(){
  assertThatExceptionOfType(AssertionError.class).isThrownBy(() -> assertThat(LocalDate.of(2000,1,5)).isBeforeOrEqualTo(LocalDate.of(1998,1,1))).withMessage(format("%nExpecting:%n  <2000-01-05>%nto be before or equals to:%n  <1998-01-01>"));
}
