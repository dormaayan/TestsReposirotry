@Test public void should_fail_if_completable_future_is_not_cancelled(){
  CompletableFuture<String> future=new CompletableFuture<>();
  future.cancel(true);
  assertThatThrownBy(() -> assertThat(future).isNotCancelled()).isInstanceOf(AssertionError.class).hasMessage(shouldNotBeCancelled(future).create());
}
