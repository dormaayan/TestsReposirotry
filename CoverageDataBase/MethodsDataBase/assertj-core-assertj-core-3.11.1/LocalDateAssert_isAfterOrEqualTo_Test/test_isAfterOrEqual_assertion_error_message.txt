@Test public void test_isAfterOrEqual_assertion_error_message(){
  assertThatExceptionOfType(AssertionError.class).isThrownBy(() -> assertThat(LocalDate.of(2000,1,5)).isAfterOrEqualTo(LocalDate.of(2012,1,1))).withMessage(format("%n" + "Expecting:%n" + "  <2000-01-05>%n"+ "to be after or equals to:%n"+ "  <2012-01-01>"));
}
