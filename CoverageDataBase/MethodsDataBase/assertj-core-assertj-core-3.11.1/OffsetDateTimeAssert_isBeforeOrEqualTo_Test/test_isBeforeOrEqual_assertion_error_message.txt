@Test public void test_isBeforeOrEqual_assertion_error_message(){
  assertThatExceptionOfType(AssertionError.class).isThrownBy(() -> assertThat(of(2000,1,5,3,0,5,0,UTC)).isBeforeOrEqualTo(of(1998,1,1,3,3,3,0,UTC))).withMessage(format("%nExpecting:%n" + "  <2000-01-05T03:00:05Z>%n" + "to be before or equals to:%n"+ "  <1998-01-01T03:03:03Z>"));
}
