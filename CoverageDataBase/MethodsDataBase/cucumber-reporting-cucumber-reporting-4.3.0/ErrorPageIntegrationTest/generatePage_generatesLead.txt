@Test public void generatePage_generatesLead(){
  setUpWithJson(SAMPLE_JSON);
  configuration.setBuildNumber("12");
  page=new ErrorPage(reportResult,configuration,cause,jsonReports);
  page.generatePage();
  DocumentAssertion document=documentFrom(page.getWebPage());
  LeadAssertion lead=document.getLead();
  assertThat(lead.getHeader()).isEqualTo("Error");
  assertThat(lead.getDescription()).isEqualTo(String.format("Something went wrong with project %s, build %s",configuration.getProjectName(),configuration.getBuildNumber()));
}
