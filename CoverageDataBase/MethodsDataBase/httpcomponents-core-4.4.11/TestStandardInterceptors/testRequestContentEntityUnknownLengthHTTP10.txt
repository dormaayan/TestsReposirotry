@Test public void testRequestContentEntityUnknownLengthHTTP10() throws Exception {
  final HttpContext context=new BasicHttpContext(null);
  final BasicHttpEntityEnclosingRequest request=new BasicHttpEntityEnclosingRequest("POST","/",HttpVersion.HTTP_1_0);
  final BasicHttpEntity entity=new BasicHttpEntity();
  entity.setContentLength(-1);
  entity.setChunked(false);
  request.setEntity(entity);
  final RequestContent interceptor=new RequestContent();
  try {
    interceptor.process(request,context);
    Assert.fail("ProtocolException should have been thrown");
  }
 catch (  final ProtocolException ex) {
  }
}
