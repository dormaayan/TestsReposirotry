@Test public void testStaleWhenIOError() throws Exception {

  Mockito.when(socket.getInputStream()).thenReturn(inStream);
  Mockito.when(inStream.read(Matchers.<byte[]>any(),Matchers.anyInt(),Matchers.anyInt())).thenThrow(new SocketException());

  conn.ensureOpen();
