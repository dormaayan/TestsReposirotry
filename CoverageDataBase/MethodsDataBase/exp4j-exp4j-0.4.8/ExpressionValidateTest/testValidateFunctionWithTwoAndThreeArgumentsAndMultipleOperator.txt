@Test public void testValidateFunctionWithTwoAndThreeArgumentsAndMultipleOperator() throws Exception {
  Expression exp=new ExpressionBuilder("gamma(x * 2 / 4, beta(y, h + 1 + 2), z + 1 + 2 + 3 + 4)").variables("x","y","z","h").functions(gamma,beta).build();
  ValidationResult result=exp.validate(false);
  Assert.assertTrue(result.isValid());
}
