@Test public void testIsEmpty(){
  ArrayStack stack=new ArrayStack(5);
  assertTrue(stack.isEmpty());
  stack.push(4);
  assertFalse(stack.isEmpty());
  stack.push(4);
  assertFalse(stack.isEmpty());
  stack.push(4);
  assertFalse(stack.isEmpty());
  stack.pop();
  stack.pop();
  stack.pop();
  assertTrue(stack.isEmpty());
  stack.push(4);
  assertFalse(stack.isEmpty());
  stack.peek();
  assertFalse(stack.isEmpty());
  stack.pop();
  assertTrue(stack.isEmpty());
}
