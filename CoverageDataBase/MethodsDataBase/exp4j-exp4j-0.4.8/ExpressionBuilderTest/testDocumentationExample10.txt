@Test(expected=ArithmeticException.class) public void testDocumentationExample10() throws Exception {
  Operator reciprocal=new Operator("$",1,true,Operator.PRECEDENCE_DIVISION){
    @Override public double apply(    final double... args){
      if (args[0] == 0d) {
        throw new ArithmeticException("Division by zero!");
      }
      return 1d / args[0];
    }
  }
;
  Expression e=new ExpressionBuilder("0$").operator(reciprocal).build();
  e.evaluate();
}
