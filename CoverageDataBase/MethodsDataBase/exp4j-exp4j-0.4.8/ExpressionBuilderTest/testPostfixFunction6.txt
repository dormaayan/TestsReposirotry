@Test public void testPostfixFunction6() throws Exception {
  String expr;
  double expected;
  double x=1.565d;
  double y=2.1323d;
  expr="ceil(x) + 1 / y * abs(1.4)";
  expected=Math.ceil(x) + 1 / y * Math.abs(1.4);
  Expression e=new ExpressionBuilder(expr).variables("x","y").build();
  assertTrue(expected == e.setVariable("x",x).setVariable("y",y).evaluate());
}
