@Test public void testFunction11() throws Exception {
  Function custom1=new Function("power",2){
    @Override public double apply(    double... values){
      return Math.pow(values[0],values[1]);
    }
  }
;
  Expression e=new ExpressionBuilder("power(x,y)").variables("x","y").function(custom1).build().setVariable("x",2d).setVariable("y",4d);
  assertTrue(Math.pow(2,4) == e.evaluate());
}
