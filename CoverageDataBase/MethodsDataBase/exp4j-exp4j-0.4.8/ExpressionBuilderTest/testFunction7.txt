@Test public void testFunction7() throws Exception {
  Function custom1=new Function("half"){
    @Override public double apply(    double... values){
      return values[0] / 2;
    }
  }
;
  Expression e=new ExpressionBuilder("half(x)").variables("x").function(custom1).build().setVariable("x",1d);
  assertTrue(0.5d == e.evaluate());
}
