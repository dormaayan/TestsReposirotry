@Test public void testPostfixExponentation4() throws Exception {
  String expr;
  double expected;
  double x=4.334d;
  expr="(24 + 4) * 2^log(x)";
  expected=(24 + 4) * Math.pow(2,Math.log(x));
  Expression e=new ExpressionBuilder(expr).variables("x").build().setVariable("x",x);
  assertTrue(expected == e.evaluate());
}
