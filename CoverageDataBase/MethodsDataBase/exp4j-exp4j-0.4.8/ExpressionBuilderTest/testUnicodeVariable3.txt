@Test public void testUnicodeVariable3() throws Exception {
  Function log=new Function("λωγ",1){
    @Override public double apply(    double... args){
      return log(args[0]);
    }
  }
;
  Expression e=new ExpressionBuilder("λωγ(π)").variable("π").function(log).build().setVariable("π",PI);
  assertEquals(log(PI),e.evaluate(),0d);
}
