@Test public void testFunction14() throws Exception {
  Function custom1=new Function("multiply",2){
    @Override public double apply(    double... values){
      return values[0] * values[1];
    }
  }
;
  double varX=1;
  Expression e=new ExpressionBuilder("multiply(sin(x),x+1)").variables("x").function(custom1).build().setVariable("x",varX);
  double expected=Math.sin(varX) * (varX + 1);
  double actual=e.evaluate();
  assertTrue(expected == actual);
}
