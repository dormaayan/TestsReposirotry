@Test public void testFunction2() throws Exception {
  Function custom=new Function("loglog"){
    @Override public double apply(    double... values){
      return Math.log(Math.log(values[0]));
    }
  }
;
  Expression e=new ExpressionBuilder("loglog(x)").variables("x").function(custom).build().setVariable("x",1);
  double result=e.evaluate();
  assertTrue(result == Math.log(Math.log(1)));
}
