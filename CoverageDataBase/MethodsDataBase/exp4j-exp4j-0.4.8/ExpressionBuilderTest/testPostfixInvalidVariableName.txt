@Test(expected=IllegalArgumentException.class) public void testPostfixInvalidVariableName() throws Exception {
  String expr;
  double expected;
  double x=4.5334332d;
  double log=Math.PI;
  expr="x * pi";
  expected=x * log;
  Expression e=new ExpressionBuilder(expr).variables("x","pi").build();
  assertTrue(expected == e.setVariable("x",x).setVariable("log",log).evaluate());
}
