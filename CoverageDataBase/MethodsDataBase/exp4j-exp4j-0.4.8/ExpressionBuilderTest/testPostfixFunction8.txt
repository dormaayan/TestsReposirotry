@Test public void testPostfixFunction8() throws Exception {
  String expr;
  double expected;
  double varE=Math.E;
  expr="2^3.4223232 + tan(e)";
  expected=Math.pow(2,3.4223232d) + Math.tan(Math.E);
  Expression e=new ExpressionBuilder(expr).variables("e").build();
  assertTrue(expected == e.setVariable("e",varE).evaluate());
}
