@Test public void testFunction6() throws Exception {
  Function custom1=new Function("foo"){
    @Override public double apply(    double... values){
      return values[0] * Math.E;
    }
  }
;
  Function custom2=new Function("bar"){
    @Override public double apply(    double... values){
      return values[0] * Math.PI;
    }
  }
;
  double varX=32.24979131d;
  Expression e=new ExpressionBuilder("bar(foo(log(x)))").variables("x").functions(custom1,custom2).build().setVariable("x",varX);
  double result=e.evaluate();
  assertTrue(result == Math.log(varX) * Math.E * Math.PI);
}
