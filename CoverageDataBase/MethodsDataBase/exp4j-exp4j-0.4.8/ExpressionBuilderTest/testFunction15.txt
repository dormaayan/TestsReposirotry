@Test public void testFunction15() throws Exception {
  Function custom1=new Function("timesPi"){
    @Override public double apply(    double... values){
      return values[0] * Math.PI;
    }
  }
;
  double varX=1;
  Expression e=new ExpressionBuilder("timesPi(x^2)").variables("x").function(custom1).build().setVariable("x",varX);
  double expected=varX * Math.PI;
  double actual=e.evaluate();
  assertTrue(expected == actual);
}
