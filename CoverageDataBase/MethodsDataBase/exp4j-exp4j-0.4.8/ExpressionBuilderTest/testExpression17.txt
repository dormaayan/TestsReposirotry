@Test public void testExpression17() throws Exception {
  String expr="x^2 - 2 * y";
  double x=Math.E;
  double y=Math.PI;
  double expected=x * x - 2 * y;
  Expression e=new ExpressionBuilder(expr).variables("x","y").build().setVariable("x",x).setVariable("y",y);
  assertTrue(expected == e.evaluate());
}
