@Test public void testExpressionBuilder06() throws Exception {
  double varX=1.3d;
  double varY=4.22d;
  double varZ=4.22d;
  Expression e=new ExpressionBuilder("x * y * z").variables("x","y","z").build();
  e.setVariable("x",varX);
  e.setVariable("y",varY);
  e.setVariable("z",varZ);
  double result=e.evaluate();
  assertTrue(result == varX * varY * varZ);
}
