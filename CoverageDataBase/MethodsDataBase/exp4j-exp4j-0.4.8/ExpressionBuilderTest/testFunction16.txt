@Test public void testFunction16() throws Exception {
  Function custom1=new Function("multiply",3){
    @Override public double apply(    double... values){
      return values[0] * values[1] * values[2];
    }
  }
;
  double varX=1;
  Expression e=new ExpressionBuilder("multiply(sin(x),x+1^(-2),log(x))").variables("x").function(custom1).build().setVariable("x",varX);
  double expected=Math.sin(varX) * Math.pow((varX + 1),-2) * Math.log(varX);
  assertTrue(expected == e.evaluate());
}
