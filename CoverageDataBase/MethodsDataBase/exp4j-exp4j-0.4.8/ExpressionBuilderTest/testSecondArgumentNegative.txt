@Test public void testSecondArgumentNegative() throws Exception {
  Function round=new Function("MULTIPLY",2){
    @Override public double apply(    double... args){
      return Math.round(args[0] * args[1]);
    }
  }
;
  double result=new ExpressionBuilder("MULTIPLY(2,-1)").function(round).build().evaluate();
  assertEquals(-2d,result,0d);
}
