@Test public void testUnicodeVariable4() throws Exception {
  Function log=new Function("λ_ωγ",1){
    @Override public double apply(    double... args){
      return log(args[0]);
    }
  }
;
  Expression e=new ExpressionBuilder("3λ_ωγ(πε6)").variables("π","ε").function(log).build().setVariable("π",PI).setVariable("ε",E);
  assertEquals(3 * log(PI * E * 6),e.evaluate(),0d);
}
