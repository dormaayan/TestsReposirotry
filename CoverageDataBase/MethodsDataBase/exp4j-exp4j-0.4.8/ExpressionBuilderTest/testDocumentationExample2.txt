@Test public void testDocumentationExample2() throws Exception {
  ExecutorService exec=Executors.newFixedThreadPool(1);
  Expression e=new ExpressionBuilder("3log(y)/(x+1)").variables("x","y").build().setVariable("x",2.3).setVariable("y",3.14);
  Future<Double> result=e.evaluateAsync(exec);
  double expected=3 * Math.log(3.14d) / (3.3);
  assertEquals(expected,result.get(),0d);
}
