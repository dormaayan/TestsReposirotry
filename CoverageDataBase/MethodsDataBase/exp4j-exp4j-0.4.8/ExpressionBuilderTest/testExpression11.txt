@Test public void testExpression11() throws Exception {
  double x=1d;
  double y=2d;
  String expr="log(x) ^ sin(y)";
  double expected=Math.pow(Math.log(x),Math.sin(y));
  Expression e=new ExpressionBuilder(expr).variables("x","y").build().setVariable("x",x).setVariable("y",y);
  assertTrue(expected == e.evaluate());
}
