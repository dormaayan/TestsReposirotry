@Test public void testFunction17() throws Exception {
  Function custom1=new Function("timesPi"){
    @Override public double apply(    double... values){
      return values[0] * Math.PI;
    }
  }
;
  double varX=Math.E;
  Expression e=new ExpressionBuilder("timesPi(log(x^(2+1)))").variables("x").function(custom1).build().setVariable("x",varX);
  double expected=Math.log(Math.pow(varX,3)) * Math.PI;
  assertTrue(expected == e.evaluate());
}
