@Test public void testPostfixVariables() throws Exception {
  String expr;
  double expected;
  double x=4.5334332d;
  double pi=Math.PI;
  expr="x * pi";
  expected=x * pi;
  Expression e=new ExpressionBuilder(expr).variables("x","pi").build();
  assertTrue(expected == e.setVariable("x",x).setVariable("pi",pi).evaluate());
}
