@Test public void testPostfixFunction11() throws Exception {
  String expr;
  double expected;
  expr="cos(x) - (1/cbrt(x))";
  Expression e=new ExpressionBuilder(expr).variables("x").build();
  for (double x=-10; x < 10; x=x + 0.5d) {
    if (x == 0d)     continue;
    expected=Math.cos(x) - (1 / Math.cbrt(x));
    assertTrue(expected == e.setVariable("x",x).evaluate());
  }
}
