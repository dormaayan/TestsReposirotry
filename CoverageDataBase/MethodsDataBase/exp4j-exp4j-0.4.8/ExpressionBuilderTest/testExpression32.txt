@Test public void testExpression32() throws Exception {
  String expr="-2 *33.34/log(y_2x)^-2 + 14 *6";
  double x=1.334d;
  double expected=-2 * 33.34 / Math.pow(Math.log(x),-2) + 14 * 6;
  Expression e=new ExpressionBuilder(expr).variables("y_2x").build().setVariable("y_2x",x);
  assertTrue(expected == e.evaluate());
}
