@Test public void testExpressionBuilder4() throws Exception {
  Function log2=new Function("log2",1){
    @Override public double apply(    double... args){
      return Math.log(args[0]) / Math.log(2);
    }
  }
;
  double result=new ExpressionBuilder("log2(4)").function(log2).build().evaluate();
  double expected=2;
  assertEquals(expected,result,0d);
}
