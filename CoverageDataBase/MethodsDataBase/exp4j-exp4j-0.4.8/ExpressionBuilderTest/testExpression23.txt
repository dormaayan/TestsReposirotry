@Test public void testExpression23() throws Exception {
  String expr="-2 *33.34/(log(foo)^-2 + 14 *6) - sin(foo)";
  double x=1.334d;
  double expected=-2 * 33.34 / (Math.pow(Math.log(x),-2) + 14 * 6) - Math.sin(x);
  Expression e=new ExpressionBuilder(expr).variables("foo").build().setVariable("foo",x);
  assertTrue(expected == e.evaluate());
}
