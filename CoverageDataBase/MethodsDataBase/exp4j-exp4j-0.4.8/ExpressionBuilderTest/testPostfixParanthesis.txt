@Test public void testPostfixParanthesis() throws Exception {
  String expr;
  double expected;
  expr="(3 + 3 * 14) * (2 * (24-17) - 14)/((34) -2)";
  expected=(3 + 3 * 14) * (2 * (24 - 17) - 14) / ((34) - 2);
  Expression e=new ExpressionBuilder(expr).build();
  assertTrue(expected == e.evaluate());
}
