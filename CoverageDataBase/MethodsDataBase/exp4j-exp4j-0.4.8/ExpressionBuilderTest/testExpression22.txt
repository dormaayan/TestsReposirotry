@Test public void testExpression22() throws Exception {
  String expr="-2 *33.34/log(x)^-2 + 14 *6";
  double x=1.334d;
  double expected=-2 * 33.34 / Math.pow(Math.log(x),-2) + 14 * 6;
  Expression e=new ExpressionBuilder(expr).variables("x").build().setVariable("x",x);
  assertTrue(expected == e.evaluate());
}
