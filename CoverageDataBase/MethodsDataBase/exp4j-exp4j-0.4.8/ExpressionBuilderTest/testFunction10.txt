@Test public void testFunction10() throws Exception {
  Function custom1=new Function("max",2){
    @Override public double apply(    double... values){
      return values[0] < values[1] ? values[1] : values[0];
    }
  }
;
  Expression e=new ExpressionBuilder("max(x,y)").variables("x","y").function(custom1).build().setVariable("x",1d).setVariable("y",2d);
  assertTrue(2 == e.evaluate());
}
