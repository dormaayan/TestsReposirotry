@Test public void metricPrefix() throws IOException {
  assertEquals("200",handlebars.compileInline("{{metricPrefix this}}").apply(200));
  assertEquals("1k",handlebars.compileInline("{{metricPrefix this}}").apply(1000));
  assertEquals("3.5M",handlebars.compileInline("{{metricPrefix this}}").apply(3500000));
}
