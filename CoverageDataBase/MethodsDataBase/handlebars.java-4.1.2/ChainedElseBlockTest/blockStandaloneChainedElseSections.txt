@Test public void blockStandaloneChainedElseSections() throws IOException {
  shouldCompileTo("{{#people}}{{name}}{{else if none}}{{none}}{{/people}}",$("none","No people"),"No people");
  shouldCompileTo("{{#people}}{{name}}{{else if none}}{{none}}{{^}}{{/people}}",$("none","No people"),"No people");
}
