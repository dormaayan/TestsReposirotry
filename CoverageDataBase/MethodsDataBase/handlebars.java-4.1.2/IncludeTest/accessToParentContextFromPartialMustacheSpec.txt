@Test public void accessToParentContextFromPartialMustacheSpec() throws IOException {
  String string="{{#each hobbies}}{{> hobby}} {{/each}}";
  Object hash=$("name","Dennis","town","berlin","hobbies",new Object[]{$("hobbyname","swimming"),$("hobbyname","dancing"),$("hobbyname","movies")});
  Hash partials=$("hobby","{{name}} has hobby {{hobbyname}} and lives in {{town}}");
  shouldCompileToWithPartials(string,hash,partials,"Dennis has hobby swimming and lives in berlin " + "Dennis has hobby dancing and lives in berlin " + "Dennis has hobby movies and lives in berlin ");
}
