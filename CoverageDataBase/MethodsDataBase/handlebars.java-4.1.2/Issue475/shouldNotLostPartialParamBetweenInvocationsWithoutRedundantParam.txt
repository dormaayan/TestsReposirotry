@Test public void shouldNotLostPartialParamBetweenInvocationsWithoutRedundantParam() throws IOException {
  final String template="{{> q name=\"firstName\" suffix=suffix}}\n" + "{{> q name=\"lastName\" suffix=suffix}}\n" + "{{> q name=\"streetName\" suffix=suffix}}";
  shouldCompileTo("{{> p suffix=\"foo\"}}",$("partials",$("p",template,"q","{{name}} {{suffix}}")),"firstName foo\n" + "lastName foo\n" + "streetName foo");
}
