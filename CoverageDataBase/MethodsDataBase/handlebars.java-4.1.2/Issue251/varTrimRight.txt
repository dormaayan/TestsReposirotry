@Test public void varTrimRight() throws IOException {
  shouldCompileTo("{{this ~}}   ","trim-right","trim-right");
  shouldCompileTo("{{this ~}}  x ","trim-right","trim-rightx ");
  shouldCompileTo("{{this ~}}   x","trim-right","trim-rightx");
  shouldCompileTo("{{this ~}}x   ","trim-right","trim-rightx   ");
}
