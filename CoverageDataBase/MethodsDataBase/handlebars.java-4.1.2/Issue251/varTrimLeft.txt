@Test public void varTrimLeft() throws IOException {
  shouldCompileTo("    {{~this}}","trim-left","trim-left");
  shouldCompileTo("  x  {{~this}}","trim-left","  xtrim-left");
  shouldCompileTo("x  {{~this}}","trim-left","xtrim-left");
  shouldCompileTo("   x{{~this}}","trim-left","   xtrim-left");
}
