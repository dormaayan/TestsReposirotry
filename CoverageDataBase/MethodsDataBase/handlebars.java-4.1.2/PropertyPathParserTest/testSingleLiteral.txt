@Test public void testSingleLiteral(){
  eq(array("foo"),PathCompiler.compile("[foo]"));
  eq(array("foo.bar.baz"),PathCompiler.compile("[foo.bar.baz]"));
  eq(array("foo/bar/baz"),PathCompiler.compile("[foo/bar/baz]"));
  eq(array("foo/bar.baz"),PathCompiler.compile("[foo/bar.baz]"));
  eq(array("/foo/bar.baz"),PathCompiler.compile("[/foo/bar.baz]"));
  eq(array("../foo/bar.baz"),PathCompiler.compile("[../foo/bar.baz]"));
  eq(array("./foo/bar.baz"),PathCompiler.compile("[./foo/bar.baz]"));
  eq(array("../"),PathCompiler.compile("[../]"));
  eq(array("./"),PathCompiler.compile("[./]"));
}
