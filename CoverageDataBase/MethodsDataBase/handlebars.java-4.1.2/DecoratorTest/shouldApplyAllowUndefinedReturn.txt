@Test public void shouldApplyAllowUndefinedReturn() throws IOException {
  shouldCompileTo("{{#helper}}{{*decorator}}suc{{/helper}}",$("helpers",$("helper",new Helper<Object>(){
    @Override public Object apply(    final Object context,    final Options options) throws IOException {
      return options.fn().toString() + options.data("run");
    }
  }
),"decorators",$("decorator",new Decorator(){
    @Override public void apply(    final Template fn,    final Options options) throws IOException {
      options.data("run","cess");
    }
  }
)),"success");
}
