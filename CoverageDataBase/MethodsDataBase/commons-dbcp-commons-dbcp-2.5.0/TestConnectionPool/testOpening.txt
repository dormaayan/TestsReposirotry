@Test public void testOpening() throws Exception {
  final Connection[] c=new Connection[getMaxTotal()];
  for (int i=0; i < c.length; i++) {
    c[i]=newConnection();
    assertTrue(c[i] != null);
    for (int j=0; j <= i; j++) {
      assertTrue(!c[j].isClosed());
    }
  }
  for (  final Connection element : c) {
    element.close();
  }
}
