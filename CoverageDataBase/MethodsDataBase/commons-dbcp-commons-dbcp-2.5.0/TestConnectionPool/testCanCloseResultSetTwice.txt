@Test public void testCanCloseResultSetTwice() throws Exception {
  final Connection conn=newConnection();
  assertNotNull(conn);
  assertTrue(!conn.isClosed());
  for (int i=0; i < 2; i++) {
    final PreparedStatement stmt=conn.prepareStatement("select * from dual");
    assertNotNull(stmt);
    final ResultSet rset=stmt.executeQuery();
    assertNotNull(rset);
    assertFalse(isClosed(rset));
    rset.close();
    assertTrue(isClosed(rset));
    rset.close();
    assertTrue(isClosed(rset));
    rset.close();
    assertTrue(isClosed(rset));
  }
  conn.close();
}
