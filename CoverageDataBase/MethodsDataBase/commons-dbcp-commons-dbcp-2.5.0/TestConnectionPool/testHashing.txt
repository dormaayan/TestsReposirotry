/** 
 * DBCP-128: BasicDataSource.getConnection() Connections don't work as hashtable keys
 */
@Test public void testHashing() throws Exception {
  final Connection con=getConnection();
  final Hashtable<Connection,String> hash=new Hashtable<>();
  hash.put(con,"test");
  assertEquals("test",hash.get(con));
  assertTrue(hash.containsKey(con));
  assertTrue(hash.contains("test"));
  hash.clear();
  con.close();
}
