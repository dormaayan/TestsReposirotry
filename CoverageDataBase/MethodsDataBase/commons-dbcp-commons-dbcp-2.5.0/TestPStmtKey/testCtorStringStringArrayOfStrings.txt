/** 
 * Tests  {@link org.apache.commons.dbcp2.PStmtKey#PStmtKey(String,String,String,String[])}. See https://issues.apache.org/jira/browse/DBCP-494
 */
@Test public void testCtorStringStringArrayOfStrings(){
  final String[] input={"A","B"};
  final PStmtKey pStmtKey=new PStmtKey("","","",input);
  Assert.assertArrayEquals(input,pStmtKey.getColumnNames());
  input[0]="C";
  input[1]="D";
  Assert.assertFalse(Arrays.equals(input,pStmtKey.getColumnNames()));
}
