/** 
 * Tests  {@link org.apache.commons.dbcp2.PStmtKey#PStmtKey(String,String,String,int[])}. See https://issues.apache.org/jira/browse/DBCP-494
 */
@Test public void testCtorStringStringArrayOfInts(){
  final int[] input={0,0};
  final PStmtKey pStmtKey=new PStmtKey("","","",input);
  Assert.assertArrayEquals(input,pStmtKey.getColumnIndexes());
  input[0]=1;
  input[1]=1;
  Assert.assertFalse(Arrays.equals(input,pStmtKey.getColumnIndexes()));
}
