@Test public void failureRestrictsRange(){
  try {
    behavior.setFailurePercent(-13);
    fail();
  }
 catch (  IllegalArgumentException e) {
    assertThat(e).hasMessage("Failure percentage must be between 0 and 100.");
  }
  try {
    behavior.setFailurePercent(174);
    fail();
  }
 catch (  IllegalArgumentException e) {
    assertThat(e).hasMessage("Failure percentage must be between 0 and 100.");
  }
}
