@Test public void missingConverterThrowsOnNonRequestBody() throws IOException {
  Retrofit retrofit=new Retrofit.Builder().baseUrl(server.url("/")).build();
  CallMethod example=retrofit.create(CallMethod.class);
  try {
    example.disallowed("Hi!");
    fail();
  }
 catch (  IllegalArgumentException e) {
    assertThat(e).hasMessage("Unable to create @Body converter for class java.lang.String (parameter #1)\n" + "    for method CallMethod.disallowed");
    assertThat(e.getCause()).hasMessage("Could not locate RequestBody converter for class java.lang.String. Tried:\n" + " * retrofit2.BuiltInConverters");
  }
}
