@Test public void requestConverterFactoryNoMatchThrows(){
  Type type=String.class;
  Annotation[] annotations=new Annotation[0];
  Converter.Factory factory1=spy(new Converter.Factory(){
    @Override public Converter<?,RequestBody> requestBodyConverter(    Type returnType,    Annotation[] annotations,    Retrofit retrofit){
      return null;
    }
  }
);
  Retrofit retrofit=new Retrofit.Builder().baseUrl("http://example.com/").addConverterFactory(factory1).build();
  try {
    retrofit.requestBodyConverter(type,annotations);
  }
 catch (  IllegalArgumentException e) {
    assertThat(e).hasMessageStartingWith("Could not locate RequestBody converter for class java.lang.String. Tried:");
  }
  verify(factory1).requestBodyConverter(type,annotations,retrofit);
  verifyNoMoreInteractions(factory1);
}
