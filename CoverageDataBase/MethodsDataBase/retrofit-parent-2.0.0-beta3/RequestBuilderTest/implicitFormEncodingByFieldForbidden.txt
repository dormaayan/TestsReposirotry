@Test public void implicitFormEncodingByFieldForbidden(){
class Example {
    @POST("/") Call<ResponseBody> method(    @Field("a") int a){
      return null;
    }
  }
  try {
    buildRequest(Example.class);
    fail();
  }
 catch (  IllegalArgumentException e) {
    assertThat(e).hasMessage("@Field parameters can only be used with form encoding. (parameter #1)\n    for method Example.method");
  }
}
