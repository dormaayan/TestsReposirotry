@Test public void multipleParameterAnnotationsNotAllowed() throws Exception {
class Example {
    @GET("/") Call<ResponseBody> method(    @Body @Query("nope") String o){
      return null;
    }
  }
  try {
    buildRequest(Example.class);
    fail();
  }
 catch (  IllegalArgumentException e) {
    assertThat(e).hasMessage("Multiple Retrofit annotations found, only one allowed. (parameter #1)\n    for method Example.method");
  }
}
