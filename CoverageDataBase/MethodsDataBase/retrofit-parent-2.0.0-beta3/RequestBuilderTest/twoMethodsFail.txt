@Test public void twoMethodsFail(){
class Example {
    @PATCH("/foo") @POST("/foo") Call<ResponseBody> method(){
      return null;
    }
  }
  try {
    buildRequest(Example.class);
    fail();
  }
 catch (  IllegalArgumentException e) {
    assertThat(e).hasMessage("Only one HTTP method is allowed. Found: PATCH and POST.\n    for method Example.method");
  }
}
