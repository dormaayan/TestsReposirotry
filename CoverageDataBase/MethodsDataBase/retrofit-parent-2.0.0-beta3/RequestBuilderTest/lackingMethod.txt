@Test public void lackingMethod(){
class Example {
    Call<ResponseBody> method(){
      return null;
    }
  }
  try {
    buildRequest(Example.class);
    fail();
  }
 catch (  IllegalArgumentException e) {
    assertThat(e).hasMessage("HTTP method annotation is required (e.g., @GET, @POST, etc.).\n    for method Example.method");
  }
}
