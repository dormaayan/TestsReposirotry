@Test public void nonBodyHttpMethodWithSingleEntity(){

    @GET("/") Call<ResponseBody> method(    @Body String o){
      return null;
    }
  }
  try {
    buildRequest(Example.class);
    fail();
  }
 catch (  IllegalArgumentException e) {
    assertThat(e).hasMessage("Non-body HTTP method cannot contain @Body.\n    for method Example.method");
  }
}
