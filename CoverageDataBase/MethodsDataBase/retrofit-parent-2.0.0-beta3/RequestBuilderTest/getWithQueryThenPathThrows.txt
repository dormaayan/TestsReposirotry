@Test public void getWithQueryThenPathThrows(){
class Example {
    @GET("/foo/bar/{ping}/") Call<ResponseBody> method(    @Query("kit") String kit,    @Path("ping") String ping){
      return null;
    }
  }
  try {
    buildRequest(Example.class,"kat","pong");
    fail();
  }
 catch (  IllegalArgumentException e) {
    assertThat(e).hasMessage("A @Path parameter must not come after a @Query. (parameter #2)\n" + "    for method Example.method");
  }
}
