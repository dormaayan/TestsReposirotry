@Test public void multipartFailsWithNoParts(){
class Example {
    @Multipart @POST("/") Call<ResponseBody> method(){
      return null;
    }
  }
  try {
    buildRequest(Example.class);
    fail();
  }
 catch (  IllegalArgumentException e) {
    assertThat(e).hasMessage("Multipart method must contain at least one @Part.\n    for method Example.method");
  }
}
