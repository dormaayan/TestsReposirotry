@Test public void getWithPathThenUrlThrows(){
class Example {
    @GET Call<ResponseBody> method(    @Path("hey") String hey,    @Url Object url){
      return null;
    }
  }
  try {
    buildRequest(Example.class,"foo/bar");
    fail();
  }
 catch (  IllegalArgumentException e) {
    assertThat(e).hasMessage("@Path can only be used with relative url on @GET (parameter #1)\n" + "    for method Example.method");
  }
}
