@Test public void implicitFormEncodingByFieldMapForbidden(){
class Example {
    @POST("/") Call<ResponseBody> method(    @FieldMap Map<String,String> a){
      return null;
    }
  }
  try {
    buildRequest(Example.class);
    fail();
  }
 catch (  IllegalArgumentException e) {
    assertThat(e).hasMessage("@FieldMap parameters can only be used with form encoding. (parameter #1)\n    for method Example.method");
  }
}
