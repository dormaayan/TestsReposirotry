@Test public void fieldMapMustBeAMap(){
class Example {
    @FormUrlEncoded @POST("/") Call<ResponseBody> method(    @FieldMap List<String> a){
      return null;
    }
  }
  try {
    buildRequest(Example.class);
    fail();
  }
 catch (  IllegalArgumentException e) {
    assertThat(e).hasMessage("@FieldMap parameter type must be Map. (parameter #1)\n    for method Example.method");
  }
}
