@Test public void getUrlAndUrlParamThrows(){
class Example {
    @GET("foo/bar") Call<ResponseBody> method(    @Url Object url){
      return null;
    }
  }
  try {
    buildRequest(Example.class,"foo/bar");
    fail();
  }
 catch (  IllegalArgumentException e) {
    assertThat(e).hasMessage("@Url cannot be used with @GET URL (parameter #1)\n" + "    for method Example.method");
  }
}
