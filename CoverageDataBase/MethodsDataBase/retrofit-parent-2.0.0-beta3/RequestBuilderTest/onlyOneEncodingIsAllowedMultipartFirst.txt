@Test public void onlyOneEncodingIsAllowedMultipartFirst(){
class Example {
    @Multipart @FormUrlEncoded @POST("/") Call<ResponseBody> method(){
      return null;
    }
  }
  try {
    buildRequest(Example.class);
    fail();
  }
 catch (  IllegalArgumentException e) {
    assertThat(e).hasMessage("Only one encoding annotation is allowed.\n    for method Example.method");
  }
}
