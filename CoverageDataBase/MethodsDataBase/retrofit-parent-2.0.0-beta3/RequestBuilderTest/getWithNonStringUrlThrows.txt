@Test public void getWithNonStringUrlThrows(){
class Example {
    @GET Call<ResponseBody> method(    @Url Object url){
      return null;
    }
  }
  try {
    buildRequest(Example.class,"foo/bar");
    fail();
  }
 catch (  IllegalArgumentException e) {
    assertThat(e).hasMessage("@Url must be String, java.net.URI, or android.net.Uri type. (parameter #1)\n" + "    for method Example.method");
  }
}
