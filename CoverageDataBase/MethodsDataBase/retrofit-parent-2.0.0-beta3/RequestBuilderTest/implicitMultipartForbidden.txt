@Test public void implicitMultipartForbidden(){
class Example {
    @POST("/") Call<ResponseBody> method(    @Part("a") int a){
      return null;
    }
  }
  try {
    buildRequest(Example.class);
    fail();
  }
 catch (  IllegalArgumentException e) {
    assertThat(e).hasMessage("@Part parameters can only be used with multipart encoding. (parameter #1)\n    for method Example.method");
  }
}
