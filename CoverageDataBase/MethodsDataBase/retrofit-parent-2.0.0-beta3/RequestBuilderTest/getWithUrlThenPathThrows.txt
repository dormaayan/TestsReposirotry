@Test public void getWithUrlThenPathThrows(){
class Example {
    @GET Call<ResponseBody> method(    @Url String url,    @Path("hey") String hey){
      return null;
    }
  }
  try {
    buildRequest(Example.class,"foo/bar");
    fail();
  }
 catch (  IllegalArgumentException e) {
    assertThat(e).hasMessage("@Path parameters may not be used with @Url. (parameter #2)\n" + "    for method Example.method");
  }
}
