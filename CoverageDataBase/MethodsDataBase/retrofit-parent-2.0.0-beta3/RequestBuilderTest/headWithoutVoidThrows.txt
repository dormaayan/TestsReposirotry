@Test public void headWithoutVoidThrows(){
class Example {
    @HEAD("/foo/bar/") Call<ResponseBody> method(){
      return null;
    }
  }
  try {
    buildRequest(Example.class);
  }
 catch (  IllegalArgumentException e) {
    assertThat(e).hasMessage("HEAD method must use Void as response type.\n    for method Example.method");
  }
}
