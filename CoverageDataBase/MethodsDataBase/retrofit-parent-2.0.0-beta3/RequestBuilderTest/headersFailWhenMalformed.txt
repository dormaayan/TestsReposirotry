@Test public void headersFailWhenMalformed(){
class Example {
    @GET("/") @Headers("Malformed") Call<ResponseBody> method(){
      return null;
    }
  }
  try {
    buildRequest(Example.class);
    fail();
  }
 catch (  IllegalArgumentException e) {
    assertThat(e).hasMessage("@Headers value must be in the form \"Name: Value\". Found: \"Malformed\"\n    for method Example.method");
  }
}
