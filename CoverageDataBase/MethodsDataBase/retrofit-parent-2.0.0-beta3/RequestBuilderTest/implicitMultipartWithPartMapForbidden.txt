@Test public void implicitMultipartWithPartMapForbidden(){
class Example {
    @POST("/") Call<ResponseBody> method(    @PartMap Map<String,String> params){
      return null;
    }
  }
  try {
    buildRequest(Example.class);
    fail();
  }
 catch (  IllegalArgumentException e) {
    assertThat(e).hasMessage("@PartMap parameters can only be used with multipart encoding. (parameter #1)\n    for method Example.method");
  }
}
