@Test public void getWithQueryThenUrlThrows(){
class Example {
    @GET("foo/bar") Call<ResponseBody> method(    @Query("hey") String hey,    @Url Object url){
      return null;
    }
  }
  try {
    buildRequest(Example.class,"hey","foo/bar/");
    fail();
  }
 catch (  IllegalArgumentException e) {
    assertThat(e).hasMessage("A @Url parameter must not come after a @Query (parameter #2)\n" + "    for method Example.method");
  }
}
