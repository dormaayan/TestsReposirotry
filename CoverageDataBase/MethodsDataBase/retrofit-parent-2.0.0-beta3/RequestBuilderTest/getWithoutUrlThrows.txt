@Test public void getWithoutUrlThrows(){
class Example {
    @GET Call<ResponseBody> method(){
      return null;
    }
  }
  try {
    buildRequest(Example.class);
    fail();
  }
 catch (  IllegalArgumentException e) {
    assertThat(e).hasMessage("Missing either @GET URL or @Url parameter.\n" + "    for method Example.method");
  }
}
