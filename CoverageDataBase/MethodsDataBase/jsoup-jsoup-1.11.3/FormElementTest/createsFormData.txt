@Test public void createsFormData(){
  String html="<form><input name='one' value='two'><select name='three'><option value='not'>" + "<option value='four' selected><option value='five' selected><textarea name=six>seven</textarea>" + "<input name='seven' type='radio' value='on' checked><input name='seven' type='radio' value='off'>"+ "<input name='eight' type='checkbox' checked><input name='nine' type='checkbox' value='unset'>"+ "<input name='ten' value='text' disabled>"+ "</form>";
  Document doc=Jsoup.parse(html);
  FormElement form=(FormElement)doc.select("form").first();
  List<Connection.KeyVal> data=form.formData();
  assertEquals(6,data.size());
  assertEquals("one=two",data.get(0).toString());
  assertEquals("three=four",data.get(1).toString());
  assertEquals("three=five",data.get(2).toString());
  assertEquals("six=seven",data.get(3).toString());
  assertEquals("seven=on",data.get(4).toString());
  assertEquals("eight=on",data.get(5).toString());
}
