@Test public void removeFormElement(){
  String html="<html>\n" + "  <body> \n" + "      <form action=\"/hello.php\" method=\"post\">\n"+ "      User:<input type=\"text\" name=\"user\" />\n"+ "      Password:<input type=\"password\" name=\"pass\" />\n"+ "      <input type=\"submit\" name=\"login\" value=\"login\" />\n"+ "   </form>\n"+ "  </body>\n"+ "</html>  ";
  Document doc=Jsoup.parse(html);
  FormElement form=(FormElement)doc.selectFirst("form");
  Element pass=form.selectFirst("input[name=pass]");
  pass.remove();
  List<Connection.KeyVal> data=form.formData();
  assertEquals(2,data.size());
  assertEquals("user",data.get(0).key());
  assertEquals("login",data.get(1).key());
  assertEquals(null,doc.selectFirst("input[name=pass]"));
}
