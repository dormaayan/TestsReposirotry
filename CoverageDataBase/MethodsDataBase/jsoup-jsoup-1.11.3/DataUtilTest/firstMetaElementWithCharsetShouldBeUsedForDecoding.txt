@Test public void firstMetaElementWithCharsetShouldBeUsedForDecoding() throws Exception {
  String html="<html><head>" + "<meta http-equiv=\"Content-Type\" content=\"text/html; charset=iso-8859-1\">" + "<meta http-equiv=\"Content-Type\" content=\"text/html; charset=koi8-u\">"+ "</head><body>Übergrößenträger</body></html>";
  Document doc=DataUtil.parseInputStream(stream(html,"iso-8859-1"),null,"http://example.com",Parser.htmlParser());
  assertEquals("Übergrößenträger",doc.body().text());
}
