@Test public void testMetaCharsetUpdateCleanup(){
  final Document doc=createHtmlDocument("dontTouch");
  doc.updateMetaCharsetElement(true);
  doc.charset(Charset.forName(charsetUtf8));
  final String htmlCharsetUTF8="<html>\n" + " <head>\n" + "  <meta charset=\"" + charsetUtf8 + "\">\n"+ " </head>\n"+ " <body></body>\n"+ "</html>";
  assertEquals(htmlCharsetUTF8,doc.toString());
}
