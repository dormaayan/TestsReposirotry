@Test public void testHtmlAndXmlSyntax(){
  String h="<!DOCTYPE html><body><img async checked='checked' src='&<>\"'>&lt;&gt;&amp;&quot;<foo />bar";
  Document doc=Jsoup.parse(h);
  doc.outputSettings().syntax(Syntax.html);
  assertEquals("<!doctype html>\n" + "<html>\n" + " <head></head>\n"+ " <body>\n"+ "  <img async checked src=\"&amp;<>&quot;\">&lt;&gt;&amp;\"\n"+ "  <foo />bar\n"+ " </body>\n"+ "</html>",doc.html());
  doc.outputSettings().syntax(Document.OutputSettings.Syntax.xml);
  assertEquals("<!DOCTYPE html>\n" + "<html>\n" + " <head></head>\n"+ " <body>\n"+ "  <img async=\"\" checked=\"checked\" src=\"&amp;<>&quot;\" />&lt;&gt;&amp;\"\n"+ "  <foo />bar\n"+ " </body>\n"+ "</html>",doc.html());
}
