@Test public void matchesAny(){
  char[] scan={' ','\n','\t'};
  CharacterReader r=new CharacterReader("One\nTwo\tThree");
  assertFalse(r.matchesAny(scan));
  assertEquals("One",r.consumeToAny(scan));
  assertTrue(r.matchesAny(scan));
  assertEquals('\n',r.consume());
  assertFalse(r.matchesAny(scan));
}
