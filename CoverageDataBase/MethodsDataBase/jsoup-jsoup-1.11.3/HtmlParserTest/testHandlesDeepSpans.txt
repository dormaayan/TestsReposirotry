@Test public void testHandlesDeepSpans(){
  StringBuilder sb=new StringBuilder();
  for (int i=0; i < 200; i++) {
    sb.append("<span>");
  }
  sb.append("<p>One</p>");
  Document doc=Jsoup.parse(sb.toString());
  assertEquals(200,doc.select("span").size());
  assertEquals(1,doc.select("p").size());
}
