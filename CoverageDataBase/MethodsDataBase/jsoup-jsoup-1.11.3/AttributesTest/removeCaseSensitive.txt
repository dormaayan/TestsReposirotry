@Test public void removeCaseSensitive(){
  Attributes a=new Attributes();
  a.put("Tot","a&p");
  a.put("tot","one");
  a.put("Hello","There");
  a.put("hello","There");
  a.put("data-name","Jsoup");
  assertEquals(5,a.size());
  a.remove("Tot");
  a.remove("Hello");
  assertEquals(3,a.size());
  assertTrue(a.hasKey("tot"));
  assertFalse(a.hasKey("Tot"));
}
