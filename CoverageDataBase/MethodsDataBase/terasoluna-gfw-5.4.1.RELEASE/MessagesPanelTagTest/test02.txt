/** 
 * Set default messages attribute name & ResultMessages set property key.<br> check property key to textMessasge in messagesPanelTag.
 */

  assertThat(getOutput(),is(expected));
  assertThat(ret,is(TagSupport.EVAL_BODY_INCLUDE));
}
