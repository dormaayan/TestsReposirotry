/** 
 * Set default messages attribute name & ResultMessages.fromText().<br> check textMessasge in messagesPanelTag.
 */
@Test public void test01() throws Exception {
  request.setAttribute(ResultMessages.DEFAULT_MESSAGES_ATTRIBUTE_NAME,ResultMessages.error().add(ResultMessage.fromText("hello world!")));
  int ret=tag.doStartTag();
  String expected="<div class=\"alert alert-error\"><ul><li>hello world!</li></ul></div>";
  assertThat(getOutput(),is(expected));
  assertThat(ret,is(TagSupport.EVAL_BODY_INCLUDE));
}
