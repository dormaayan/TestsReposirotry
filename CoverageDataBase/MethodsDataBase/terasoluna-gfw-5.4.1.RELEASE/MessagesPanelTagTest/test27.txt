/** 
 * Set default messages attribute name & Use Array & set PanelClassName and PanelTypeClassPrefix is empty & MessagesType is "error".<br> check CSS class is error in messagesPanelTag.
 */
@Test public void test27() throws Exception {
  request.setAttribute(ResultMessages.DEFAULT_MESSAGES_ATTRIBUTE_NAME,Arrays.asList("foo","bar"));
  tag.setPanelClassName("");
  tag.setPanelTypeClassPrefix("");
  tag.setMessagesType("error");
  int ret=tag.doStartTag();
  String expected="<div class=\"error\"><ul><li>foo</li><li>bar</li></ul></div>";
  assertThat(getOutput(),is(expected));
  assertThat(ret,is(TagSupport.EVAL_BODY_INCLUDE));
}
