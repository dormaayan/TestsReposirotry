/** 
 * Set default messages attribute name & Use ResultMessages & change OuterElement and InnerElement is empty.<br> check no OuterElement and PanelElement in messagesPanelTag.
 */
@Test public void test17() throws Exception {
  request.setAttribute(ResultMessages.DEFAULT_MESSAGES_ATTRIBUTE_NAME,ResultMessages.info().add(ResultMessage.fromText("foo")));
  tag.setOuterElement("");
  tag.setInnerElement("");
  int ret=tag.doStartTag();
  String expected="<div class=\"alert alert-info\">foo</div>";
  assertThat(getOutput(),is(expected));
  assertThat(ret,is(TagSupport.EVAL_BODY_INCLUDE));
}
