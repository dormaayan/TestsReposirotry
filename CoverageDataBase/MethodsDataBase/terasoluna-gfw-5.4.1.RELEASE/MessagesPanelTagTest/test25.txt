/** 
 * Set default messages attribute name & Use arrays & set MessagesType is empty.<br> check PanelTypeClassPrefix + MessagesType is empty and array messages in messagesPanelTag.
 */
@Test public void test25() throws Exception {
  request.setAttribute(ResultMessages.DEFAULT_MESSAGES_ATTRIBUTE_NAME,Arrays.asList("foo","bar"));
  tag.setMessagesType("");
  int ret=tag.doStartTag();
  String expected="<div class=\"alert\"><ul><li>foo</li><li>bar</li></ul></div>";
  assertThat(getOutput(),is(expected));
  assertThat(ret,is(TagSupport.EVAL_BODY_INCLUDE));
}
