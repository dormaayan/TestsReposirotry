/** 
 * Set default messages attribute name & Arrays.<br> check two textMessages in messagesPanelTag.
 */
@Test public void test08() throws Exception {
  request.setAttribute(ResultMessages.DEFAULT_MESSAGES_ATTRIBUTE_NAME,Arrays.asList("hello","world"));
  int ret=tag.doStartTag();
  String expected="<div class=\"alert\"><ul><li>hello</li><li>world</li></ul></div>";
  assertThat(getOutput(),is(expected));
  assertThat(ret,is(TagSupport.EVAL_BODY_INCLUDE));
}
