/** 
 * Set default messages attribute name & ResultMessage.fromCode("property key").<br> check property key to textMessasge in messagesPanelTag.
 */
@Test public void test05() throws Exception {
  request.setAttribute(ResultMessages.DEFAULT_MESSAGES_ATTRIBUTE_NAME,ResultMessage.fromCode("hello.world"));
  int ret=tag.doStartTag();
  String expected="<div class=\"alert\"><ul><li>hello world!</li></ul></div>";
  assertThat(getOutput(),is(expected));
  assertThat(ret,is(TagSupport.EVAL_BODY_INCLUDE));
}
