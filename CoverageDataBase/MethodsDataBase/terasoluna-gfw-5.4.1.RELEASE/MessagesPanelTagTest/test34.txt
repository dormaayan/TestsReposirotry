/** 
 * Set default messages attribute name & Empty Array.<br> check that start tags are not self-closing in messagesPanelTag.
 */

  int ret=tag.doStartTag();

  assertThat(ret,is(TagSupport.EVAL_BODY_INCLUDE));
}
