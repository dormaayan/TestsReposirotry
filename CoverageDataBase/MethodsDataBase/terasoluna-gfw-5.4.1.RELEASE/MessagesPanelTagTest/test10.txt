/** 
 * Set default messages attribute name & Add ResultMessages twice by property key.<br> check two messages property key to textMessage in messagesPanelTag.
 */
@Test public void test10() throws Exception {
  request.setAttribute(ResultMessages.DEFAULT_MESSAGES_ATTRIBUTE_NAME,ResultMessages.success().add("hello.world").add("foo.bar",1,2));
  int ret=tag.doStartTag();
  String expected="<div class=\"alert alert-success\"><ul><li>hello world!</li><li>foo1 and bar2</li></ul></div>";
  assertThat(getOutput(),is(expected));
  assertThat(ret,is(TagSupport.EVAL_BODY_INCLUDE));
}
