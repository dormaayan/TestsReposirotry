/** 
 * Set attribute name & Use ResultMessages.<br> check messagesPanelTag can get set Attribute name.
 */
@Test public void test19() throws Exception {
  request.setAttribute("result",ResultMessages.error().add(ResultMessage.fromText("hello world!")));
  tag.setMessagesAttributeName("result");
  int ret=tag.doStartTag();
  String expected="<div class=\"alert alert-error\"><ul><li>hello world!</li></ul></div>";
  assertThat(getOutput(),is(expected));
  assertThat(ret,is(TagSupport.EVAL_BODY_INCLUDE));
}
