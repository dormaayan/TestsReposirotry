/** 
 * Set default messages attribute name & Use ResultMessages & Change PanelClassName, PanelTypeClassPrefix and MessagesType is empty.<br> check no CSS class in messagesPanelTag.
 */
@Test public void test21() throws Exception {
  request.setAttribute(ResultMessages.DEFAULT_MESSAGES_ATTRIBUTE_NAME,ResultMessages.info().add(ResultMessage.fromText("foo")).add(ResultMessage.fromText("bar")));
  tag.setPanelClassName("");
  tag.setPanelTypeClassPrefix("");
  tag.setMessagesType("");
  int ret=tag.doStartTag();
  String expected="<div><ul><li>foo</li><li>bar</li></ul></div>";
  assertThat(getOutput(),is(expected));
  assertThat(ret,is(TagSupport.EVAL_BODY_INCLUDE));
}
