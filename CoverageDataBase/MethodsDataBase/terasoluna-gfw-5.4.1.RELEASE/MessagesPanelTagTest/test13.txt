/** 
 * Set default messages attribute name & Use Arrays & change PanelClassName is empty.<br> check no CSS class setting in messagesPanelTag.
 */
@Test public void test13() throws Exception {
  request.setAttribute(ResultMessages.DEFAULT_MESSAGES_ATTRIBUTE_NAME,Arrays.asList("foo","bar"));
  tag.setPanelClassName("");
  int ret=tag.doStartTag();
  String expected="<div><ul><li>foo</li><li>bar</li></ul></div>";
  assertThat(getOutput(),is(expected));
  assertThat(ret,is(TagSupport.EVAL_BODY_INCLUDE));
}
