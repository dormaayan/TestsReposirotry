/** 
 * specify left and right. expected valid if input left value is equal right value.
 * @throws Throwable
 */
@Test public void testSpecifyLeftAndRight() throws Throwable {
{
    form.setLeft(100);
    form.setRight(99);
    violations=validator.validate(form);
    assertThat(violations.size(),is(1));
    assertThat(violations.iterator().next().getMessage(),is(String.format(MESSAGE_VALIDATION_ERROR,"left","right")));
  }
{
    form.setLeft(100);
    form.setRight(100);
    violations=validator.validate(form);
    assertThat(violations.size(),is(0));
  }
}
