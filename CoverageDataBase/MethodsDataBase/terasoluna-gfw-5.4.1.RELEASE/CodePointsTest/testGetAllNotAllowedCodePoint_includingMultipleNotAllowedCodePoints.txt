@Test public void testGetAllNotAllowedCodePoint_includingMultipleNotAllowedCodePoints(){
  String testStr=SURROGATE_PAIR_CHAR_2000B + "あいうきかくか" + SURROGATE_PARE_CHAR_20B9F;
  Set<Integer> result=new CodePoints("あいうえお").allExcludedCodePoints(testStr);
  assertThat(result.size(),is(5));
  Iterator<Integer> it=result.iterator();
  assertThat(it.next().intValue(),is(0x2000B));
  assertThat(it.next().intValue(),is("き".codePointAt(0)));
  assertThat(it.next().intValue(),is("か".codePointAt(0)));
  assertThat(it.next().intValue(),is("く".codePointAt(0)));
  assertThat(it.next().intValue(),is(0x20B9F));
}
