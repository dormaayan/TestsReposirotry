/** 
 * [afterPropertiesSet] Case of log level is higher than debug. <p> [Expected Result] <ol> <li>target is empty.(don't occur error.)</li> </ol> </p>
 * @throws Exception
 */
@Test public void testAfterPropertiesSet_isDebugEnabled_false() throws Exception {
  Logger logger=(Logger)LoggerFactory.getLogger(CodeListInterceptor.class);
  LogLevelChangeUtil.setLogLevel(LogLevelChangeUtil.LogLevel.INFO);
  testTarget.setApplicationContext(new StaticApplicationContext());
  testTarget.afterPropertiesSet();
  assertThat(testTarget.getCodeLists().isEmpty(),is(true));
  assertThat(logger.isDebugEnabled(),is(false));
  LogLevelChangeUtil.resetLogLevel();
}
