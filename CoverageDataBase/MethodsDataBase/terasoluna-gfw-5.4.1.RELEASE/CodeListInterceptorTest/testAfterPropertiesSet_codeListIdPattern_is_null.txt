/** 
 * [afterPropertiesSet] Case of codeListIdPattern is null. <p> [Expected Result] <ol> <li>target is all bean of CodeList type.(don't filter by bean id.)</li> <li>target list is </ol> </p>
 * @throws Exception
 */
@Test public void testAfterPropertiesSet_codeListIdPattern_is_null() throws Exception {
  testTarget.setCodeListIdPattern(null);
  testTarget.afterPropertiesSet();
  Map<String,CodeList> expectedCodeListMap=new HashMap<String,CodeList>(getApplicationContext().getBeansOfType(CodeList.class));
  assertThat(testTarget.getCodeLists().toString(),is(expectedCodeListMap.values().toString()));
}
