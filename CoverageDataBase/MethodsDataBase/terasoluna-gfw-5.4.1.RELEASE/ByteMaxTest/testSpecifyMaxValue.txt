/** 
 * specify max value. expected valid if input value encoded in UTF-8 is grater than or equal max value.
 * @throws Throwable
 */
@Test public void testSpecifyMaxValue() throws Throwable {
{
    form.setStringProperty("ああa");
    violations=validator.validate(form);
    assertThat(violations.size(),is(1));
    assertThat(violations.iterator().next().getMessage(),is(String.format(MESSAGE_VALIDATION_ERROR,6)));
  }
{
    form.setStringProperty("ああ");
    violations=validator.validate(form);
    assertThat(violations.size(),is(0));
  }
}
