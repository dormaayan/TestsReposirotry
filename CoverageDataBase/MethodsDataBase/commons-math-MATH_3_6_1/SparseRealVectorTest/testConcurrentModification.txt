@Test public void testConcurrentModification(){
  final RealVector u=new OpenMapRealVector(3,1e-6);
  u.setEntry(0,1);
  u.setEntry(1,0);
  u.setEntry(2,2);
  final RealVector v1=new OpenMapRealVector(3,1e-6);
  v1.setEntry(0,0);
  v1.setEntry(1,3);
  v1.setEntry(2,0);
  u.ebeMultiply(v1);
  u.ebeDivide(v1);
}
