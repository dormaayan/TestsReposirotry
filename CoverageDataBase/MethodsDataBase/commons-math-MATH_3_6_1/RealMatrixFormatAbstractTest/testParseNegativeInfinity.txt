@Test public void testParseNegativeInfinity(){
  String source="{{(-Infinity), (-Infinity), (-Infinity)}}";
  RealMatrix actual=realMatrixFormat.parse(source);
  RealMatrix expected=MatrixUtils.createRealMatrix(new double[][]{{Double.NEGATIVE_INFINITY,Double.NEGATIVE_INFINITY,Double.NEGATIVE_INFINITY}});
  Assert.assertEquals(expected,actual);
}
