@Test public void testDefaultFormat(){
  Locale defaultLocale=Locale.getDefault();
  Locale.setDefault(getLocale());
  RealMatrix m=MatrixUtils.createRealMatrix(new double[][]{{232.2222222222,-342.33333333333,432.44444444444}});
  String expected="{{232" + getDecimalCharacter() + "2222222222,-342"+ getDecimalCharacter()+ "3333333333,432"+ getDecimalCharacter()+ "4444444444}}";
  String actual=(new RealMatrixFormat()).format(m);
  Assert.assertEquals(expected,actual);
  Locale.setDefault(defaultLocale);
}
