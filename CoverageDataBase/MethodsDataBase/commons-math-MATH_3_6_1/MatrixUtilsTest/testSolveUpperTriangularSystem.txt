@Test public void testSolveUpperTriangularSystem(){
  RealMatrix rm=new Array2DRowRealMatrix(new double[][]{{1,2,3},{0,1,1},{0,0,2}},false);
  RealVector b=new ArrayRealVector(new double[]{8,4,2},false);
  MatrixUtils.solveUpperTriangularSystem(rm,b);

}
