/** 
 * Make sure Double.NaN is returned iff n < 4
 */
@Test public void testNaN(){
  Kurtosis kurt=new Kurtosis();
  Assert.assertTrue(Double.isNaN(kurt.getResult()));
  kurt.increment(1d);
  Assert.assertTrue(Double.isNaN(kurt.getResult()));
  kurt.increment(1d);
  Assert.assertTrue(Double.isNaN(kurt.getResult()));
  kurt.increment(1d);
  Assert.assertTrue(Double.isNaN(kurt.getResult()));
  kurt.increment(1d);
  Assert.assertFalse(Double.isNaN(kurt.getResult()));
}
