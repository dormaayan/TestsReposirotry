@Test public void testRootN(){
  for (double x=-0.9; x < 0.9; x+=0.05) {
    for (int n=1; n < 5; ++n) {
      if (x < 0) {
        if (n % 2 == 1) {
          checkRelative(-FastMath.pow(-x,1.0 / n),build(x).rootN(n));
        }
      }
 else {
        checkRelative(FastMath.pow(x,1.0 / n),build(x).rootN(n));
      }
    }
  }
}
