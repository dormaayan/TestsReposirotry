@Test public void testLinearCombinationFF3(){
  RandomGenerator r=new Well1024a(0xff3l);
  for (int i=0; i < 50; ++i) {
    double[] aD=generateDouble(r,3);
    double[] bD=generateDouble(r,3);
    T[] aF=toFieldArray(aD);
    T[] bF=toFieldArray(bD);
    checkRelative(MathArrays.linearCombination(aD[0],bD[0],aD[1],bD[1],aD[2],bD[2]),aF[0].linearCombination(aF[0],bF[0],aF[1],bF[1],aF[2],bF[2]));
  }
}
