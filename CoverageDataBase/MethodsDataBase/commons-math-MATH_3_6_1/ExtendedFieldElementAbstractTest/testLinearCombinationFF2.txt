@Test public void testLinearCombinationFF2(){
  RandomGenerator r=new Well1024a(0xff2l);
  for (int i=0; i < 50; ++i) {
    double[] aD=generateDouble(r,2);
    double[] bD=generateDouble(r,2);
    T[] aF=toFieldArray(aD);
    T[] bF=toFieldArray(bD);
    checkRelative(MathArrays.linearCombination(aD[0],bD[0],aD[1],bD[1]),aF[0].linearCombination(aF[0],bF[0],aF[1],bF[1]));
  }
}
