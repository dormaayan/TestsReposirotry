@Test public void testGetReducedFraction(){
  Fraction threeFourths=new Fraction(3,4);
  Assert.assertTrue(threeFourths.equals(Fraction.getReducedFraction(6,8)));
  Assert.assertTrue(Fraction.ZERO.equals(Fraction.getReducedFraction(0,-1)));
  try {
    Fraction.getReducedFraction(1,0);
    Assert.fail("expecting MathArithmeticException");
  }
 catch (  MathArithmeticException ex) {
  }
  Assert.assertEquals(Fraction.getReducedFraction(2,Integer.MIN_VALUE).getNumerator(),-1);
  Assert.assertEquals(Fraction.getReducedFraction(1,-1).getNumerator(),-1);
}
