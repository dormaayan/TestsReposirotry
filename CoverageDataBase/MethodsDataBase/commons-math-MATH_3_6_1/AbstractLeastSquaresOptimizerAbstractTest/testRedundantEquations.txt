@Test public void testRedundantEquations(){
  LinearProblem problem=new LinearProblem(new double[][]{{1,1},{1,-1},{1,3}},new double[]{3,1,5});
  AbstractLeastSquaresOptimizer optimizer=createOptimizer();
  PointVectorValuePair optimum=optimizer.optimize(new MaxEval(100),problem.getModelFunction(),problem.getModelFunctionJacobian(),problem.getTarget(),new Weight(new double[]{1,1,1}),new InitialGuess(new double[]{1,1}));
  Assert.assertEquals(0,optimizer.getRMS(),1e-10);
  Assert.assertEquals(2,optimum.getPointRef()[0],1e-10);
  Assert.assertEquals(1,optimum.getPointRef()[1],1e-10);
}
