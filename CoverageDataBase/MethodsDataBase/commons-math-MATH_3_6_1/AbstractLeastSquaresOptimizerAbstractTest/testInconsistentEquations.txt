@Test public void testInconsistentEquations(){
  LinearProblem problem=new LinearProblem(new double[][]{{1,1},{1,-1},{1,3}},new double[]{3,1,4});
  AbstractLeastSquaresOptimizer optimizer=createOptimizer();
  optimizer.optimize(new MaxEval(100),problem.getModelFunction(),problem.getModelFunctionJacobian(),problem.getTarget(),new Weight(new double[]{1,1,1}),new InitialGuess(new double[]{1,1}));
  Assert.assertTrue(optimizer.getRMS() > 0.1);
}
