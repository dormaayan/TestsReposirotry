@Test(expected=DimensionMismatchException.class) public void testInconsistentSizes2(){
  LinearProblem problem=new LinearProblem(new double[][]{{1,0},{0,1}},new double[]{-1,1});
  AbstractLeastSquaresOptimizer optimizer=createOptimizer();
  PointVectorValuePair optimum=optimizer.optimize(new MaxEval(100),problem.getModelFunction(),problem.getModelFunctionJacobian(),problem.getTarget(),new Weight(new double[]{1,1}),new InitialGuess(new double[]{0,0}));
  Assert.assertEquals(0,optimizer.getRMS(),1e-10);
  Assert.assertEquals(-1,optimum.getPoint()[0],1e-10);
  Assert.assertEquals(1,optimum.getPoint()[1],1e-10);
  optimizer.optimize(new MaxEval(100),problem.getModelFunction(),problem.getModelFunctionJacobian(),new Target(new double[]{1}),new Weight(new double[]{1}),new InitialGuess(new double[]{0,0}));
}
