@Test public void testPointCopy(){
  LinearProblem problem=new LinearProblem(new double[][]{{1,0,0},{-1,1,0},{0,-1,1}},new double[]{1,1,1});
  final boolean[] checked={false};
  final LeastSquaresBuilder builder=problem.getBuilder().checker(new ConvergenceChecker<Evaluation>(){
    public boolean converged(    int iteration,    Evaluation previous,    Evaluation current){
      Assert.assertThat(previous.getPoint(),not(sameInstance(current.getPoint())));
      Assert.assertArrayEquals(new double[3],previous.getPoint().toArray(),0);
      Assert.assertArrayEquals(new double[]{1,2,3},current.getPoint().toArray(),TOl);
      checked[0]=true;
      return true;
    }
  }
);
  optimizer.optimize(builder.build());
  Assert.assertThat(checked[0],is(true));
}
