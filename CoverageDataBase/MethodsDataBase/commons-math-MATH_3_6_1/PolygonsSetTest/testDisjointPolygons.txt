@Test public void testDisjointPolygons(){
  Vector2D[][] vertices=new Vector2D[][]{new Vector2D[]{new Vector2D(0.0,1.0),new Vector2D(2.0,1.0),new Vector2D(1.0,2.0)},new Vector2D[]{new Vector2D(4.0,0.0),new Vector2D(5.0,1.0),new Vector2D(3.0,1.0)}};
  PolygonsSet set=buildSet(vertices);
  Assert.assertEquals(Region.Location.INSIDE,set.checkPoint(new Vector2D(1.0,1.5)));
  checkPoints(Region.Location.INSIDE,set,new Vector2D[]{new Vector2D(1.0,1.5),new Vector2D(4.5,0.8)});
  checkPoints(Region.Location.OUTSIDE,set,new Vector2D[]{new Vector2D(1.0,0.0),new Vector2D(3.5,1.2),new Vector2D(2.5,1.0),new Vector2D(3.0,4.0)});
  checkPoints(Region.Location.BOUNDARY,set,new Vector2D[]{new Vector2D(1.0,1.0),new Vector2D(3.5,0.5),new Vector2D(0.0,1.0)});
  checkVertices(set.getVertices(),vertices);
}
