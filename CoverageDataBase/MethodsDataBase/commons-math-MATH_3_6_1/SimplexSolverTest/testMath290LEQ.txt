@Test(expected=NoFeasibleSolutionException.class) public void testMath290LEQ(){
  LinearObjectiveFunction f=new LinearObjectiveFunction(new double[]{1,5},0);
  Collection<LinearConstraint> constraints=new ArrayList<LinearConstraint>();
  constraints.add(new LinearConstraint(new double[]{2,0},Relationship.LEQ,-1.0));
  SimplexSolver solver=new SimplexSolver();
  solver.optimize(DEFAULT_MAX_ITER,f,new LinearConstraintSet(constraints),GoalType.MINIMIZE,new NonNegativeConstraint(true));
}
