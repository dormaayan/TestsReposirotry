@Test(expected=MaxCountExceededException.class) public void testIncrementorCountExceeded(){
  final int start=1;
  final int max=7;
  final int step=2;
  final IntegerSequence.Incrementor inc=IntegerSequence.Incrementor.create().withStart(start).withMaximalCount(max).withIncrement(step);
  Assert.assertTrue(inc.canIncrement(2));
  Assert.assertFalse(inc.canIncrement(3));
  while (true) {
    inc.increment();
  }
}
