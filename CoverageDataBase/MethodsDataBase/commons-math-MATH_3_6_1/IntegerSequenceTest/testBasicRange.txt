@Test public void testBasicRange(){
  final int start=-2;
  final int end=4;
  final List<Integer> seq=new ArrayList<Integer>();
  for (  Integer i : IntegerSequence.range(start,end)) {
    seq.add(i);
  }
  for (int i=start; i <= end; i++) {
    Assert.assertEquals(i,seq.get(i - start).intValue());
  }
}
