@Test(expected=MathUnsupportedOperationException.class) public void testBoundsUnsupported(){
  SimplexOptimizer optimizer=new SimplexOptimizer(1e-10,1e-30);
  final FourExtrema fourExtrema=new FourExtrema();
  optimizer.optimize(new MaxEval(100),new ObjectiveFunction(fourExtrema),GoalType.MINIMIZE,new InitialGuess(new double[]{-3,0}),new NelderMeadSimplex(new double[]{0.2,0.2}),new SimpleBounds(new double[]{-5,-1},new double[]{5,1}));
}
