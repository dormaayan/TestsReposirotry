/** 
 * Verify that isSupportLowerBoundInclusvie returns true iff the lower bound is finite and density is non-NaN, non-infinite there.
 */
@SuppressWarnings("deprecation") @Test public void testIsSupportLowerBoundInclusive(){
  final double lowerBound=distribution.getSupportLowerBound();
  double result=Double.NaN;
  result=distribution.density(lowerBound);
  Assert.assertEquals(!Double.isInfinite(lowerBound) && !Double.isNaN(result) && !Double.isInfinite(result),distribution.isSupportLowerBoundInclusive());
}
