@Test public void testNISTExample(){
  double[] d=new double[]{95.1772,95.1567,95.1937,95.1959,95.1442,95.0610,95.1591,95.1195,95.1772,95.0925,95.1990,95.1682};

  Assert.assertEquals(95.061,new PSquarePercentile(0d).evaluate(d),0);
  Assert.assertEquals(95.1990,new PSquarePercentile(100d).evaluate(d,0,d.length),0);
}
