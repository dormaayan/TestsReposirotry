@Test public void testAccurateCrossProduct(){
  final Vector3D u1=new Vector3D(-1321008684645961.0 / 268435456.0,-5774608829631843.0 / 268435456.0,-7645843051051357.0 / 8589934592.0);
  final Vector3D u2=new Vector3D(1796571811118507.0 / 2147483648.0,7853468008299307.0 / 2147483648.0,2599586637357461.0 / 17179869184.0);
  final Vector3D u3=new Vector3D(12753243807587107.0 / 18446744073709551616.0,-2313766922703915.0 / 18446744073709551616.0,-227970081415313.0 / 288230376151711744.0);
  Vector3D cNaive=new Vector3D(u1.getY() * u2.getZ() - u1.getZ() * u2.getY(),u1.getZ() * u2.getX() - u1.getX() * u2.getZ(),u1.getX() * u2.getY() - u1.getY() * u2.getX());
  Vector3D cAccurate=u1.crossProduct(u2);
  Assert.assertTrue(u3.distance(cNaive) > 2.9 * u3.getNorm());
  Assert.assertEquals(0.0,u3.distance(cAccurate),1.0e-30 * cAccurate.getNorm());
}
