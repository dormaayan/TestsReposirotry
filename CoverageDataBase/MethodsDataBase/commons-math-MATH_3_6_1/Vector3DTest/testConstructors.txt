@Test public void testConstructors() throws DimensionMismatchException {
  double r=FastMath.sqrt(2) / 2;
  checkVector(new Vector3D(2,new Vector3D(FastMath.PI / 3,-FastMath.PI / 4)),r,r * FastMath.sqrt(3),-2 * r);
  checkVector(new Vector3D(2,Vector3D.PLUS_I,-3,Vector3D.MINUS_K),2,0,3);
  checkVector(new Vector3D(2,Vector3D.PLUS_I,5,Vector3D.PLUS_J,-3,Vector3D.MINUS_K),2,5,3);
  checkVector(new Vector3D(2,Vector3D.PLUS_I,5,Vector3D.PLUS_J,5,Vector3D.MINUS_J,-3,Vector3D.MINUS_K),2,0,3);
  checkVector(new Vector3D(new double[]{2,5,-3}),2,5,-3);
}
