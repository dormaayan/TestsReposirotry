@Test public void testDimension3(){
  RealMatrix matrix=MatrixUtils.createRealMatrix(new double[][]{{39632.0,-4824.0,-16560.0},{-4824.0,8693.0,7920.0},{-16560.0,7920.0,17300.0}});

  Assert.assertEquals(12500.0,ed.getRealEigenvalue(1),3.0e-11);
  Assert.assertEquals(3125.0,ed.getRealEigenvalue(2),3.0e-11);
}
