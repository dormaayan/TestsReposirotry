@Test public void testErf3291(){
  double x=3.291 / FastMath.sqrt(2.0);
  double actual=Erf.erf(x);
  double expected=0.999;
  Assert.assertEquals(expected,actual,1.0e-5);
  Assert.assertEquals(1 - expected,Erf.erfc(x),1.0e-5);
  actual=Erf.erf(-x);
  expected=-expected;
  Assert.assertEquals(expected,actual,1.0e-5);
  Assert.assertEquals(1 - expected,Erf.erfc(-x),1.0e-5);
}
