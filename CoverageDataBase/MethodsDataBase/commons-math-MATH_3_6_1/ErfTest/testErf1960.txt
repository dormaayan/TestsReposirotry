@Test public void testErf1960(){
  double x=1.960 / FastMath.sqrt(2.0);
  double actual=Erf.erf(x);
  double expected=0.95;
  Assert.assertEquals(expected,actual,1.0e-5);
  Assert.assertEquals(1 - actual,Erf.erfc(x),1.0e-15);
  actual=Erf.erf(-x);
  expected=-expected;
  Assert.assertEquals(expected,actual,1.0e-5);
  Assert.assertEquals(1 - actual,Erf.erfc(-x),1.0e-15);
}
