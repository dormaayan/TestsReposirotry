@Test public void testErf2807(){
  double x=2.807 / FastMath.sqrt(2.0);
  double actual=Erf.erf(x);
  double expected=0.995;
  Assert.assertEquals(expected,actual,1.0e-5);
  Assert.assertEquals(1 - actual,Erf.erfc(x),1.0e-15);
  actual=Erf.erf(-x);
  expected=-expected;
  Assert.assertEquals(expected,actual,1.0e-5);
  Assert.assertEquals(1 - actual,Erf.erfc(-x),1.0e-15);
}
