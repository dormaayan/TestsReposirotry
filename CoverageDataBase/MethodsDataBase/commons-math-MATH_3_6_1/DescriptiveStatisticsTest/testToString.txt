@Test public void testToString(){
  DescriptiveStatistics stats=createDescriptiveStatistics();
  stats.addValue(1);
  stats.addValue(2);
  stats.addValue(3);
  Locale d=Locale.getDefault();
  Locale.setDefault(Locale.US);
  Assert.assertEquals("DescriptiveStatistics:\n" + "n: 3\n" + "min: 1.0\n"+ "max: 3.0\n"+ "mean: 2.0\n"+ "std dev: 1.0\n"+ "median: 2.0\n"+ "skewness: 0.0\n"+ "kurtosis: NaN\n",stats.toString());
  Locale.setDefault(d);
}
