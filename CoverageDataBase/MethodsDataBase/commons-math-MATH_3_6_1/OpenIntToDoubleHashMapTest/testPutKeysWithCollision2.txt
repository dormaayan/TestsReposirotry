/** 
 * Similar to testPutKeysWithCollisions() but exercises the codepaths in a slightly different manner.
 */
@Test public void testPutKeysWithCollision2(){
  OpenIntToDoubleHashMap map=new OpenIntToDoubleHashMap();
  int key1=837989881;
  double value1=1.0;
  map.put(key1,value1);
  int key2=476463321;
  map.put(key2,value1);
  Assert.assertEquals(2,map.size());
  Assert.assertTrue(Precision.equals(value1,map.get(key2),1));
  map.remove(key1);
  double value2=2.0;
  map.put(key2,value2);
  Assert.assertEquals(1,map.size());
  Assert.assertTrue(Precision.equals(value2,map.get(key2),1));
}
