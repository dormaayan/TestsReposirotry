@Test public void testDefaultFormatVector3D(){
  Locale defaultLocal=Locale.getDefault();
  Locale.setDefault(getLocale());
  Vector3D c=new Vector3D(232.22222222222,-342.3333333333,432.44444444444);
  String expected="{232" + getDecimalCharacter() + "2222222222; -342"+ getDecimalCharacter()+ "3333333333; 432"+ getDecimalCharacter()+ "4444444444}";
  String actual=(new Vector3DFormat()).format(c);
  Assert.assertEquals(expected,actual);
  Locale.setDefault(defaultLocal);
}
