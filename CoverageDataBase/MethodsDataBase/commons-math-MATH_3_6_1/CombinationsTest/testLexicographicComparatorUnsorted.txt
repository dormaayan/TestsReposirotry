/** 
 * Check that iterates can be passed unsorted.
 */
@Test public void testLexicographicComparatorUnsorted(){
  final int n=5;
  final int k=3;
  final Comparator<int[]> comp=new Combinations(n,k).comparator();
  Assert.assertEquals(1,comp.compare(new int[]{1,4,2},new int[]{1,3,2}));
  Assert.assertEquals(-1,comp.compare(new int[]{0,4,1},new int[]{0,4,2}));
  Assert.assertEquals(0,comp.compare(new int[]{1,4,3},new int[]{1,3,4}));
}
