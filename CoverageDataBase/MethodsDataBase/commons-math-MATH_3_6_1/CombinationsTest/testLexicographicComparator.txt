@Test public void testLexicographicComparator(){
  final int n=5;
  final int k=3;
  final Comparator<int[]> comp=new Combinations(n,k).comparator();
  Assert.assertEquals(1,comp.compare(new int[]{1,2,4},new int[]{1,2,3}));
  Assert.assertEquals(-1,comp.compare(new int[]{0,1,4},new int[]{0,2,4}));
  Assert.assertEquals(0,comp.compare(new int[]{1,3,4},new int[]{1,3,4}));
}
