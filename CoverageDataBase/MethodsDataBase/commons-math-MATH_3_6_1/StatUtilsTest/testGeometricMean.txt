@Test public void testGeometricMean(){
  double[] test=null;
  try {
    StatUtils.geometricMean(test);
    Assert.fail("Expecting MathIllegalArgumentException");
  }
 catch (  MathIllegalArgumentException ex) {
  }
  test=new double[]{2,4,6,8};
  Assert.assertEquals(FastMath.exp(0.25d * StatUtils.sumLog(test)),StatUtils.geometricMean(test),Double.MIN_VALUE);
  Assert.assertEquals(FastMath.exp(0.5 * StatUtils.sumLog(test,0,2)),StatUtils.geometricMean(test,0,2),Double.MIN_VALUE);
}
