@Test public void testSymmetry(){
  for (int i=0; i < 6; ++i) {
    for (int j=0; j < 6; ++j) {
      DSCompiler c=DSCompiler.getCompiler(i,j);
      for (int k=0; k < c.getSize(); ++k) {
        Assert.assertEquals(k,c.getPartialDerivativeIndex(c.getPartialDerivativeOrders(k)));
      }
    }
  }
}
