@Test public void testFit07(){
  GaussianFitter fitter=new GaussianFitter(new LevenbergMarquardtOptimizer());

  double[] parameters=fitter.fit();

  Assert.assertEquals(0.015029412832160017,parameters[2],1e-4);
}
