@Test public void testCopyOfDouble2(){
  final double[] source={Double.NEGATIVE_INFINITY,-Double.MAX_VALUE,-1,0,Double.MIN_VALUE,FastMath.ulp(1d),1,3,113,4769,Double.MAX_VALUE,Double.POSITIVE_INFINITY};
  final int offset=3;
  final double[] dest=MathArrays.copyOf(source,source.length - offset);
  Assert.assertEquals(dest.length,source.length - offset);
  for (int i=0; i < source.length - offset; i++) {
    Assert.assertEquals(source[i],dest[i],0);
  }
}
