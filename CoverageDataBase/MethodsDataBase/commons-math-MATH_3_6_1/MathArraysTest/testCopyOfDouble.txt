@Test public void testCopyOfDouble(){
  final double[] source={Double.NEGATIVE_INFINITY,-Double.MAX_VALUE,-1,0,Double.MIN_VALUE,FastMath.ulp(1d),1,3,113,4769,Double.MAX_VALUE,Double.POSITIVE_INFINITY};
  final double[] dest=MathArrays.copyOf(source);
  Assert.assertEquals(dest.length,source.length);
  for (int i=0; i < source.length; i++) {
    Assert.assertEquals(source[i],dest[i],0);
  }
}
