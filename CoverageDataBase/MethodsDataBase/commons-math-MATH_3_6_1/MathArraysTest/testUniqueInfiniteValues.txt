@Test public void testUniqueInfiniteValues(){
  final double[] x={0,Double.NEGATIVE_INFINITY,3,Double.NEGATIVE_INFINITY,3,Double.POSITIVE_INFINITY,Double.POSITIVE_INFINITY};
  final double[] u={Double.POSITIVE_INFINITY,3,0,Double.NEGATIVE_INFINITY};
  Assert.assertArrayEquals(u,MathArrays.unique(x),0);
}
