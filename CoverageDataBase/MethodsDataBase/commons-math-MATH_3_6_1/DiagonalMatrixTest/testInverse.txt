@Test public void testInverse(){
  final double[] data={1,2,3};
  final DiagonalMatrix m=new DiagonalMatrix(data);
  final DiagonalMatrix inverse=m.inverse();
  final DiagonalMatrix result=m.multiply(inverse);
  TestUtils.assertEquals("DiagonalMatrix.inverse() returns wrong result",MatrixUtils.createRealIdentityMatrix(data.length),result,Math.ulp(1d));
}
