@Test public void testUnstableDerivative() throws DimensionMismatchException, NumberIsTooSmallException, MaxCountExceededException, NoBracketingException {
  final StepProblem stepProblem=new StepProblem(0.0,1.0,2.0);
  FirstOrderIntegrator integ=new DormandPrince853Integrator(0.1,10,1.0e-12,0.0);
  integ.addEventHandler(stepProblem,1.0,1.0e-12,1000);
  double[] y={Double.NaN};
  integ.integrate(stepProblem,0.0,new double[]{0.0},10.0,y);
  Assert.assertEquals(8.0,y[0],1.0e-12);
}
