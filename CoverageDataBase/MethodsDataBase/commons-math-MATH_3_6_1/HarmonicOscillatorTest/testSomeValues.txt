@Test public void testSomeValues(){
  final double a=-1.2;
  final double w=0.34;
  final double p=5.6;
  final UnivariateFunction f=new HarmonicOscillator(a,w,p);
  final double d=0.12345;
  for (int i=0; i < 10; i++) {
    final double v=i * d;
    Assert.assertEquals(a * FastMath.cos(w * v + p),f.value(v),0);
  }
}
