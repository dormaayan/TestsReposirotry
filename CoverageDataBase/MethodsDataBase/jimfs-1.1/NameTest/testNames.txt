@Test public void testNames(){
  assertThat(Name.create("foo","foo")).isEqualTo(Name.create("foo","foo"));
  assertThat(Name.create("FOO","foo")).isEqualTo(Name.create("foo","foo"));
  assertThat(Name.create("FOO","foo")).isNotEqualTo(Name.create("FOO","FOO"));
  assertThat(Name.create("a","b").toString()).isEqualTo("a");
}
