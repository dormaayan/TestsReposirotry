    windows().parsePath("foo<bar");
    fail();
  }
 catch (  InvalidPathException expected) {
  }
  try {
    windows().parsePath("foo?");

  }
 catch (  InvalidPathException expected) {
  }

    windows().parsePath("foo ");
    fail();
  }
 catch (  InvalidPathException expected) {
  }
  try {
    windows().parsePath("foo \\bar");
    fail();
  }
 catch (  InvalidPathException expected) {
  }
}
