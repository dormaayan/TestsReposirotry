@Test public void testResolve_withThisAndParentDirNames(){
  Path path=pathService.parsePath("/foo");
  assertResolvedPathEquals("/foo/bar/../baz",path,"bar/../baz");
  assertResolvedPathEquals("/foo/bar/../baz",path,"bar","..","baz");
  assertResolvedPathEquals("/foo/./bar/baz",path,"./bar/baz");
  assertResolvedPathEquals("/foo/./bar/baz",path,".","bar/baz");
}
