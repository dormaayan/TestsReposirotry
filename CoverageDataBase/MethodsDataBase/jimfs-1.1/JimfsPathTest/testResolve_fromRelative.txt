@Test public void testResolve_fromRelative(){
  Path path=pathService.parsePath("foo");
  assertResolvedPathEquals("foo/bar",path,"bar");
  assertResolvedPathEquals("foo/bar/baz/test",path,"bar/baz/test");
  assertResolvedPathEquals("foo/bar/baz/test",path,"bar","baz","test");
  assertResolvedPathEquals("foo/bar/baz/test",path,"bar/baz","test");
}
