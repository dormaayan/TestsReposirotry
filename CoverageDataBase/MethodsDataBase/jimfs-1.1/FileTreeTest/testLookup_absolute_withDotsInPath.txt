@Test public void testLookup_absolute_withDotsInPath() throws IOException {
  assertExists(lookup("/."),"/","/");
  assertExists(lookup("/./././."),"/","/");
  assertExists(lookup("/work/./one/./././two/three"),"two","three");
  assertExists(lookup("/work/./one/./././two/././three"),"two","three");
  assertExists(lookup("/work/./one/./././two/three/././."),"two","three");
}
