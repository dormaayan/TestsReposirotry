@Test public void testLookup_absolute_notExists() throws IOException {
  try {
    lookup("/a/b");
    fail();
  }
 catch (  NoSuchFileException expected) {
  }
  try {
    lookup("/work/one/foo/bar");
    fail();
  }
 catch (  NoSuchFileException expected) {
  }
  try {
    lookup("$c/d");
    fail();
  }
 catch (  NoSuchFileException expected) {
  }
  try {
    lookup("$a/b/c/d/e");
    fail();
  }
 catch (  NoSuchFileException expected) {
  }
}
