@Test public void testLookup_relative_withDotsInPath() throws IOException {
  assertExists(lookup("."),"/","work");
  assertExists(lookup("././."),"/","work");
  assertExists(lookup("./one/./././two/three"),"two","three");
  assertExists(lookup("./one/./././two/././three"),"two","three");
  assertExists(lookup("./one/./././two/three/././."),"two","three");
}
