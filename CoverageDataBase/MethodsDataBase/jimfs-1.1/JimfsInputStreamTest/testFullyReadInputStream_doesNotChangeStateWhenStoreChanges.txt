@SuppressWarnings("GuardedByChecker") @Test public void testFullyReadInputStream_doesNotChangeStateWhenStoreChanges() throws IOException {
  JimfsInputStream in=newInputStream(1,2,3,4,5);
  assertThat(in.read(new byte[5])).isEqualTo(5);
  assertEmpty(in);
  in.file.write(5,new byte[10],0,10);
  assertEmpty(in);
}
