@Test public void testObjectConstructor(){
  LongRange range=new LongRange(new Long(0),new Long(5));
  assertEquals("[0, 1, 2, 3, 4]",range.toCollection().toString());
  range=new LongRange(new Integer(0),new Long(5),new Long(1));
  assertEquals("[0, 1, 2, 3, 4]",range.toCollection().toString());
}
