@Test public void testEquals() throws Exception {
  ConditionalUnaryFunction<Object,Object> f=new ConditionalUnaryFunction<Object,Object>(Identity.INSTANCE,Constant.of("left"),Constant.of("right"));
  assertEquals(f,f);
  assertObjectsAreEqual(f,new ConditionalUnaryFunction<Object,Object>(Identity.INSTANCE,Constant.of("left"),Constant.of("right")));
  assertObjectsAreNotEqual(f,new ConditionalUnaryFunction<Object,Object>(Identity.INSTANCE,Constant.of(null),Constant.of("right")));
  assertObjectsAreNotEqual(f,new ConditionalUnaryFunction<Object,Object>(Constant.TRUE,Constant.of("left"),Constant.of("right")));
}
