@Test public void testDuplicateAdd() throws Exception {
  BinaryPredicate<Object,Object> p=Constant.TRUE;
  BinaryOr<Object,Object> q=new BinaryOr<Object,Object>(p,p);
  assertTrue(q.test("xyzzy",new Integer(3)));
  for (int i=0; i < 10; i++) {
    q.or(p);
    assertTrue(q.test("xyzzy",new Integer(3)));
  }
}
