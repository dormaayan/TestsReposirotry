@Test public void testStringConcatMap(){
  Map map=makeStringConcatMap();
  map.put("key","value 1");
  assertEquals("value 1",map.get("key"));
  map.put("key","value 2");
  assertEquals("value 1, value 2",map.get("key"));
  map.put("key","value 3");
  assertEquals("value 1, value 2, value 3",map.get("key"));
}
