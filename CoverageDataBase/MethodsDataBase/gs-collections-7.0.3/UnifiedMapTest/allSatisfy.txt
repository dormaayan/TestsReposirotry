@Override @Test public void allSatisfy(){
  super.allSatisfy();
  UnifiedMap<Integer,String> collisions=UnifiedMap.<Integer,String>newMap().withKeysValues(COLLISION_1,"one",COLLISION_2,"two",COLLISION_3,"three");
