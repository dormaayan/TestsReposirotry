@Override @Test public void select(){
  super.select();
  ImmutableByteSet set=this.newWith((byte)0,(byte)1,(byte)31,(byte)127,(byte)-1,(byte)-31,(byte)-64,(byte)-65,(byte)-128);
  Verify.assertSize(8,set.select(BytePredicates.lessThan((byte)32)));
