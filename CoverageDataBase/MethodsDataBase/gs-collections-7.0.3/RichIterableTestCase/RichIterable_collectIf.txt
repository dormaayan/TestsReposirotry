@Test default void RichIterable_collectIf(){
  RichIterable<Integer> iterable=this.newWith(13,13,12,12,11,11,3,3,2,2,1,1);
  assertEquals(this.getExpectedTransformed(3,3,1,1,3,3,1,1),iterable.collectIf(i -> i % 2 != 0,i -> i % 10));
  assertEquals(this.newMutableForTransform(3,3,1,1,3,3,1,1),iterable.collectIf(i -> i % 2 != 0,i -> i % 10,this.newMutableForTransform()));
}
