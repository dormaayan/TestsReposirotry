@Override @Test public void isEmpty(){
  super.isEmpty();
  MutableSet<String> set=this.newWith();
  this.assertIsEmpty(true,set);
  set.add("stuff");
  this.assertIsEmpty(false,set);
  set.remove("stuff");
  this.assertIsEmpty(true,set);
  set.add("Bon");
  set.add("Jovi");
  this.assertIsEmpty(false,set);
  set.remove("Jovi");
  this.assertIsEmpty(false,set);
  set.clear();
  this.assertIsEmpty(true,set);
}
