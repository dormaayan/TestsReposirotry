@Test public void parallelGroupByIntoConcurrentHashMap(){
  MutableMap<Integer,MutableBag<Integer>> actual=ConcurrentHashMap.newMap();
  ParallelIterate.forEach(Interval.oneTo(1000000),each -> actual.getIfAbsentPut(each % 100000,() -> HashBag.<Integer>newBag().asSynchronized()).add(each),10,this.executor);
  Verify.assertEqualsAndHashCode(Interval.oneTo(1000000).groupBy(each -> each % 100000).toMap(HashBag::new),actual);
}
