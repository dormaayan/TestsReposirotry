@Test public void partitionPeopleByCatOwnersAndNonCatOwnersUsingStreams(){
  Map<Boolean,List<Person>> catsAndNoCatsStream=this.people.stream().collect(Collectors.partitioningBy(person -> person.hasPet(PetType.CAT)));
  Verify.assertSize(2,catsAndNoCatsStream.get(true));
  Verify.assertSize(5,catsAndNoCatsStream.get(false));
}
