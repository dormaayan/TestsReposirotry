@Test public void groupPeopleByLastName(){
  MutableListMultimap<String,Person> byLastNameEager=this.people.groupBy(Person::getLastName);
  Verify.assertIterableSize(3,byLastNameEager.get("Smith"));
  MutableBagMultimap<String,Person> byLastNameEagerTargetBag=this.people.groupBy(Person::getLastName,Multimaps.mutable.bag.empty());
  Verify.assertIterableSize(3,byLastNameEagerTargetBag.get("Smith"));
  Multimap<String,Person> byLastNameLazy=this.people.asLazy().groupBy(Person::getLastName);
  Verify.assertIterableSize(3,byLastNameLazy.get("Smith"));
  MutableBagMultimap<String,Person> byLastNameLazyTargetBag=this.people.asLazy().groupBy(Person::getLastName,Multimaps.mutable.bag.empty());
  Verify.assertIterableSize(3,byLastNameLazyTargetBag.get("Smith"));
}
