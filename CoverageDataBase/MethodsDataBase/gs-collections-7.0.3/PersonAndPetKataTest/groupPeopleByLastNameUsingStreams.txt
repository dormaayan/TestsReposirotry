@Test public void groupPeopleByLastNameUsingStreams(){
  Map<String,List<Person>> byLastNameStream=this.people.stream().collect(Collectors.groupingBy(Person::getLastName));
  Verify.assertIterableSize(3,byLastNameStream.get("Smith"));
  Map<String,MutableBag<Person>> byLastNameStreamTargetBag=this.people.stream().collect(Collectors.groupingBy(Person::getLastName,Collectors.toCollection(Bags.mutable::empty)));
  Verify.assertIterableSize(3,byLastNameStreamTargetBag.get("Smith"));
}
