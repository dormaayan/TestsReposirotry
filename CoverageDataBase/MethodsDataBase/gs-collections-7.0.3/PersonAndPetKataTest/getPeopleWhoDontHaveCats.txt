@Test public void getPeopleWhoDontHaveCats(){
  MutableList<Person> peopleWithNoCatsEager=this.people.reject(person -> person.hasPet(PetType.CAT));
  Verify.assertSize(5,peopleWithNoCatsEager);
  MutableList<Person> peopleWithNoCatsEagerMR=this.people.rejectWith(Person::hasPet,PetType.CAT);
  Verify.assertSize(5,peopleWithNoCatsEagerMR);
  MutableList<Person> peopleWithNoCatsLazy=this.people.reject(person -> person.hasPet(PetType.CAT));
  Verify.assertSize(5,peopleWithNoCatsLazy);
  MutableList<Person> peopleWithNoCatsLazyMR=this.people.rejectWith(Person::hasPet,PetType.CAT);
  Verify.assertSize(5,peopleWithNoCatsLazyMR);
}
