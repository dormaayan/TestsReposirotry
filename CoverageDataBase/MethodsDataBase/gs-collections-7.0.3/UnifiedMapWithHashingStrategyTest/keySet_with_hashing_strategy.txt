@Test public void keySet_with_hashing_strategy(){
  Set<Person> people=UnifiedMapWithHashingStrategy.newWithKeysValues(LAST_NAME_HASHING_STRATEGY,JOHNDOE,1,JANEDOE,2,JOHNSMITH,3,JANESMITH,4).keySet();

}
