@Override @Test public void symmetricDifference(){
  Assert.assertEquals(UnifiedSet.newSetWith("not present"),SortedSets.immutable.<String>empty().symmetricDifference(UnifiedSet.newSetWith("not present")));
  Verify.assertSortedSetsEqual(TreeSortedSet.newSetWith(Comparators.<Integer>reverseNaturalOrder(),1,2,3,4),SortedSets.immutable.of(Comparators.<Integer>reverseNaturalOrder()).symmetricDifference(UnifiedSet.newSetWith(1,3,2,4)).castToSortedSet());
}
