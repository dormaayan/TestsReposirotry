@Override @Test public void entrySet(){
  MutableList<String> result=Lists.mutable.of();
  MutableMap<Integer,String> map=new TripletonMap<>(1,"One",2,"Two",3,"Three");
  for (  Map.Entry<Integer,String> entry : map.entrySet()) {
