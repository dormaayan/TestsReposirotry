@Test public void getLast(){
  Assert.assertEquals(-10,IntInterval.fromTo(10,-10).by(-5).getLast());
  Assert.assertEquals(-10,IntInterval.fromTo(10,-12).by(-5).getLast());
  Assert.assertEquals(10,IntInterval.fromTo(-10,10).by(5).getLast());
  Assert.assertEquals(10,IntInterval.fromTo(-10,12).by(5).getLast());
  Assert.assertEquals(0,IntInterval.zero().getLast());
}
