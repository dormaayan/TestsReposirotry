@Override @Test public void makeString(){
  super.makeString();
  ImmutableBag<String> bag=this.newBag();
  Assert.assertEquals(FastList.newList(bag).makeString(),bag.makeString());
  Assert.assertEquals(bag.toString(),'[' + bag.makeString() + ']');
  Assert.assertEquals(bag.toString(),'[' + bag.makeString(", ") + ']');
  Assert.assertEquals(bag.toString(),bag.makeString("[",", ","]"));
}
