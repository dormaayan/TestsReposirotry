@Test public void intersect(){
  MutableSortedSet<String> set=this.newWith("4","2","1","3");
  Verify.assertSortedSetsEqual(TreeSortedSet.newSetWith("1","2"),set.intersect(UnifiedSet.newSetWith("7","5","6","1","8","2")));

}
