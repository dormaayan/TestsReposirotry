@Test public void isLetterOrDigit(){
  assertTrue(CharLists.mutable.of('A','B','C','0','1','2','3'),CharPredicate.IS_LETTER_OR_DIGIT);
  assertTrue(CharLists.mutable.of('a','b','c','0','1','2','3'),CharPredicate.IS_LETTER_OR_DIGIT);
  assertFalse(CharLists.mutable.of('.','$','*'),CharPredicate.IS_LETTER_OR_DIGIT);
}
