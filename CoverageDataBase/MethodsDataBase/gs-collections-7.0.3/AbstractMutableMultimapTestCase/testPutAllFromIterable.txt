@Test public void testPutAllFromIterable(){
  MutableMultimap<Integer,String> multimap=this.newMultimapWithKeysValues(1,"One",2,"Two");
  Assert.assertTrue(multimap.putAll(1,Lists.fixedSize.of("Three","Four").asLazy()));

}
