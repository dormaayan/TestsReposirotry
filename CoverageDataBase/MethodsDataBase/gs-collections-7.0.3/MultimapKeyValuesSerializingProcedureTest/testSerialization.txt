@Test public void testSerialization(){

  map.put("A","alpha");
  map.put("A","beta");
  FastList<? extends Serializable> expectedWrites=FastList.newListWith("A",2,"alpha","beta");

  map.toMap().forEachKeyValue(procedure);
}
