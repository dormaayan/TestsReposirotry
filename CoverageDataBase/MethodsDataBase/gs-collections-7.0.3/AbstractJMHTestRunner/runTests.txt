@Test public void runTests() throws RunnerException {
  int warmupCount=this.warmUpCount();
  int runCount=this.runCount();
  Options opts=new OptionsBuilder().include(".*" + this.getClass().getName() + ".*").warmupTime(TimeValue.seconds(2)).warmupIterations(warmupCount).measurementTime(TimeValue.seconds(2)).measurementIterations(runCount).verbosity(VerboseMode.EXTRA).forks(2).build();
  new Runner(opts).run();
}
