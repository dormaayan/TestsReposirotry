@Test public void selectMapOnEntry(){
  MutableMap<String,String> map=UnifiedMap.newWithKeysValues("1","2","2","1","3","3");

  Verify.assertContainsKeyValue("1","2",resultMap);
  Verify.assertContainsKeyValue("2","1",resultMap);
}
