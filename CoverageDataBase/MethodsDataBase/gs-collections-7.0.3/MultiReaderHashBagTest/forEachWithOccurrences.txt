@Test public void forEachWithOccurrences(){
  MutableBag<Integer> bag=MultiReaderHashBag.newBagWith(1,2,2,3,3,3);
  int[] sum=new int[1];
  bag.forEachWithOccurrences((each,occurrences) -> {
    if (occurrences > 1) {
      sum[0]+=each * occurrences;
    }
  }
);
  Assert.assertEquals(13,sum[0]);
}
