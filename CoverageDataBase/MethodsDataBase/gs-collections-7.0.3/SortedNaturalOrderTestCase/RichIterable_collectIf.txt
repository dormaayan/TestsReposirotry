@Override @Test default void RichIterable_collectIf(){
  RichIterable<Integer> iterable=this.newWith(1,1,2,2,3,3,11,11,12,12,13,13);
  assertEquals(this.getExpectedTransformed(1,1,3,3,1,1,3,3),iterable.collectIf(i -> i % 2 != 0,i -> i % 10));
  assertEquals(this.newMutableForTransform(1,1,3,3,1,1,3,3),iterable.collectIf(i -> i % 2 != 0,i -> i % 10,this.newMutableForTransform()));
}
