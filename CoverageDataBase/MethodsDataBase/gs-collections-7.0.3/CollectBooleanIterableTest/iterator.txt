@Test public void iterator(){
  long count=0;
  long isTrueCount=0;
  BooleanIterator iterator=this.booleanIterable.booleanIterator();
  while (iterator.hasNext()) {
    count++;
    if (iterator.next()) {
      isTrueCount++;
    }
  }
  Assert.assertEquals(3L,count);
  Assert.assertEquals(2L,isTrueCount);
}
