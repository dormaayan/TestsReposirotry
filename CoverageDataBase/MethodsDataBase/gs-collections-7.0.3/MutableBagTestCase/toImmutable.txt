@Override @Test public void toImmutable(){
  super.toImmutable();
  Verify.assertInstanceOf(MutableBagIterable.class,this.newWith());
  Verify.assertInstanceOf(ImmutableBagIterable.class,this.newWith().toImmutable());

}
