@Test public void testToGuavaOptional() throws Exception {
  assertFalse(Optionals.toGuavaOptional(java.util.Optional.empty()).isPresent());
  assertTrue(Optionals.toGuavaOptional(java.util.Optional.of("Foo")).isPresent());

}
