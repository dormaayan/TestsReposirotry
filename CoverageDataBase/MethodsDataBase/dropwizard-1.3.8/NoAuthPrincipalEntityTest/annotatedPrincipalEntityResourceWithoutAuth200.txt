/** 
 * When parameter is annotated then Jersey classifies such parameter as {@link org.glassfish.jersey.server.model.Parameter.Source#UNKNOWN} instead of{@link org.glassfish.jersey.server.model.Parameter.Source#ENTITY} whichis used for unannotated parameters. ValueFactoryProvider resolution logic is different for these two sources therefore must be tested separately.
 */
@Test public void annotatedPrincipalEntityResourceWithoutAuth200(){
  String principalName="Astar Seran";
  assertThat(target("/no-auth-test/annotated-principal-entity").request().header(HttpHeaders.AUTHORIZATION,"Anything here").post(Entity.entity(new JsonPrincipal(principalName),MediaType.APPLICATION_JSON)).readEntity(String.class)).isEqualTo(principalName);
}
