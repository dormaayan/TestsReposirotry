@Test public void testBuildDefault() throws Exception {
  final BiDiGzipHandler handler=new YamlConfigurationFactory<>(GzipHandlerFactory.class,BaseValidator.newValidator(),Jackson.newObjectMapper(),"dw").build(new File(Resources.getResource("yaml/default_gzip.yml").toURI())).build(null);
  assertThat(handler.getMinGzipSize()).isEqualTo(256);
  assertThat(handler.getExcludedAgentPatterns()).isEmpty();
  assertThat(handler.getIncludedMimeTypes()).isEmpty();
  assertThat(handler.getIncludedMethods()).containsOnly("GET");
  assertThat(handler.getCompressionLevel()).isEqualTo(Deflater.DEFAULT_COMPRESSION);
  assertThat(handler.isInflateNoWrap()).isTrue();
}
