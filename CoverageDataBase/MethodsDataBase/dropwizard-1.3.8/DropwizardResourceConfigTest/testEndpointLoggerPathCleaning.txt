@Test public void testEndpointLoggerPathCleaning(){
  String dirtyPath=" /this//is///a/dirty//path/     ";
  String anotherDirtyPath="a/l//p/h/  /a/b /////  e/t";
  assertThat(rc.cleanUpPath(dirtyPath)).isEqualTo("/this/is/a/dirty/path/");

}
