@Test public void testEnumViaExplicitFromStringPrivate(){
  final ParamConverter<ExplicitFromStringPrivate> converter=getConverter(ExplicitFromStringPrivate.class);
  final WebApplicationException throwable=catchThrowableOfType(() -> converter.fromString("1"),WebApplicationException.class);
  assertThat(throwable.getResponse()).extracting(Response::getEntity).matches(e -> ((ErrorMessage)e.get(0)).getCode() == 400).matches(e -> ((ErrorMessage)e.get(0)).getMessage().contains("Not permitted to call"));
}
