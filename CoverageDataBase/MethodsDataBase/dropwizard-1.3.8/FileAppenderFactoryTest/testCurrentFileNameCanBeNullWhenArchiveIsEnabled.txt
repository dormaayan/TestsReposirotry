@Test public void testCurrentFileNameCanBeNullWhenArchiveIsEnabled() throws Exception {
  FileAppenderFactory fileAppenderFactory=new FileAppenderFactory();
  fileAppenderFactory.setArchive(true);
  fileAppenderFactory.setArchivedLogFilenamePattern("name-to-be-used");
  fileAppenderFactory.setCurrentLogFilename(null);
  ImmutableList<String> errors=ConstraintViolations.format(validator.validate(fileAppenderFactory));
  assertThat(errors).isEmpty();
}
