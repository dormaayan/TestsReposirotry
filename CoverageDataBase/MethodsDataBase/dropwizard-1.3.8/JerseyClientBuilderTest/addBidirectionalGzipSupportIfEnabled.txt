@Test public void addBidirectionalGzipSupportIfEnabled() throws Exception {
  final JerseyClientConfiguration configuration=new JerseyClientConfiguration();
  configuration.setGzipEnabled(true);
  final Client client=builder.using(configuration).using(executorService,objectMapper).build("test");
  assertThat(Iterables.filter(client.getConfiguration().getInstances(),GZipDecoder.class).iterator().hasNext()).isTrue();
  assertThat(Iterables.filter(client.getConfiguration().getInstances(),ConfiguredGZipEncoder.class).iterator().hasNext()).isTrue();
  verify(apacheHttpClientBuilder,never()).disableContentCompression(true);
}
