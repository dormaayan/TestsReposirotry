@Test public void disablesGzipSupportIfDisabled() throws Exception {
  final JerseyClientConfiguration configuration=new JerseyClientConfiguration();
  configuration.setGzipEnabled(false);
  final Client client=builder.using(configuration).using(executorService,objectMapper).build("test");
  assertThat(Iterables.filter(client.getConfiguration().getInstances(),GZipDecoder.class).iterator().hasNext()).isFalse();
  assertThat(Iterables.filter(client.getConfiguration().getInstances(),ConfiguredGZipEncoder.class).iterator().hasNext()).isFalse();
  verify(apacheHttpClientBuilder).disableContentCompression(true);
}
