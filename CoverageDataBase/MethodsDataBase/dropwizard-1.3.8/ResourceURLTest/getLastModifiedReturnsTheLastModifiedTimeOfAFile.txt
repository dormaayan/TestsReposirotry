@Test public void getLastModifiedReturnsTheLastModifiedTimeOfAFile() throws Exception {
  final Path path=Files.createTempFile("resource_url_test",null);
  final URL url=path.toUri().toURL();
  final long lastModified=ResourceURL.getLastModified(url);
  assertThat(lastModified).isGreaterThan(0);
  assertThat(lastModified).isEqualTo(path.toFile().lastModified());
  Files.delete(path);
}
