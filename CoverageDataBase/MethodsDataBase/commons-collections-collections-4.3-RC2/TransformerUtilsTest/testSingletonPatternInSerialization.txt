/** 
 * Test that all Transformer singletons hold singleton pattern in serialization/deserialization process.
 */
@Test public void testSingletonPatternInSerialization(){
  final Object[] singletones=new Object[]{ExceptionTransformer.INSTANCE,NOPTransformer.INSTANCE,StringValueTransformer.stringValueTransformer()};
  for (  final Object original : singletones) {
    TestUtils.assertSameAfterSerialization("Singleton pattern broken for " + original.getClass(),original);
  }
}
