@Test public void testGetIndexed(){
  final MultiKey<Integer> mk=new MultiKey<>(ONE,TWO);
  assertSame(ONE,mk.getKey(0));
  assertSame(TWO,mk.getKey(1));
  try {
    mk.getKey(-1);
    fail();
  }
 catch (  final IndexOutOfBoundsException ex) {
  }
  try {
    mk.getKey(2);
    fail();
  }
 catch (  final IndexOutOfBoundsException ex) {
  }
}
