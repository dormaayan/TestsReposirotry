@Test public void testSize_Array(){
  final Object[] objectArray=new Object[0];
  assertEquals(0,CollectionUtils.size(objectArray));
  final String[] stringArray=new String[3];
  assertEquals(3,CollectionUtils.size(stringArray));
  stringArray[0]="a";
  stringArray[1]="b";
  stringArray[2]="c";
  assertEquals(3,CollectionUtils.size(stringArray));
}
