@Test @Deprecated public void forAllDoIterator(){
  final Closure<Collection<Integer>> testClosure=ClosureUtils.invokerClosure("clear");
  final Collection<Collection<Integer>> col=new ArrayList<>();
  col.add(collectionA);
  col.add(collectionC);
