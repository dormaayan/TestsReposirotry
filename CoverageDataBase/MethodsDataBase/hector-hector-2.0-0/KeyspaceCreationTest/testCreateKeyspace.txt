@Test public void testCreateKeyspace() throws InvalidRequestException, TException {
  List<ColumnFamilyDefinition> cf_defs=new ArrayList<ColumnFamilyDefinition>();
  List<ColumnDef> columns=new ArrayList<ColumnDef>();
  for (int i=0; i < colCount; i++) {
    String cName="col" + i;
    log.info("Creating column " + cName);
    columns.add(newIndexedColumnDef(cName,"BytesType"));
  }
  List<ColumnDefinition> columnMetadata=ThriftColumnDef.fromThriftList(columns);
  ColumnFamilyDefinition cf_def=HFactory.createColumnFamilyDefinition(KEYSPACE,"TEST_CF",ComparatorType.BYTESTYPE,columnMetadata);
  cf_defs.add(cf_def);
  makeKeyspace(KEYSPACE,cf_defs);
  checkKeyspaces();
  log.info("Done, all errors to console after this point likely due to shutdown");
}
