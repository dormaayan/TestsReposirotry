/** 
 * "http://tomcat.apache.org" is an allowed origin and "http://tomcat.apache.org:8080" is not, because ports doesn't match
 * @throws ServletException
 * @throws IOException
 */
@Test public void testCheckForPortVariance() throws ServletException, IOException {
  MockHttpServletRequest request=new MockHttpServletRequest();
  MockHttpServletResponse response=new MockHttpServletResponse();
  request.setHeader(CORSFilter.REQUEST_HEADER_ORIGIN,"http://tomcat.apache.org:8080");
  request.setMethod("GET");
  CORSFilter corsFilter=new CORSFilter();
  corsFilter.init(TestConfigs.getSpecificOriginFilterConfig());
  corsFilter.doFilter(request,response,filterChain);
  Assert.assertEquals(HttpServletResponse.SC_FORBIDDEN,response.getStatus());
}
